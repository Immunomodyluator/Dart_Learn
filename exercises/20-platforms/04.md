В Dart-веб коде есть ошибки. Найди их.

```dart
import 'dart:html';
import 'dart:io';  // bug 1 — dart:io недоступен в браузере

void main() {
  // bug 2 — querySelector может вернуть null
  final form = querySelector('#loginForm') as FormElement; // non-null assertion без проверки

  // bug 3 — синхронный XHR заблокирует UI-поток браузера
  final request = HttpRequest();
  request.open('GET', '/api/users', async: false); // sync XHR — устарел и блокирует
  request.send();
  final users = request.responseText;
  print(users);

  // bug 4 — утечка обработчика: addEventListener без removeEventListener
  void handleClick(Event e) {
    print('Clicked!');
  }
  document.querySelector('#btn')?.addEventListener('click', handleClick);
  // ... позже элемент удалён из DOM, но обработчик удерживает ссылку

  // bug 5 — innerHTML без санитизации (XSS)
  final userInput = window.location.search; // данные от пользователя
  document.querySelector('#output')!.innerHtml = '<b>$userInput</b>'; // XSS!

  // bug 6 — localStorage синхронен, но использование неправильное
  final data = window.localStorage; // OK: Storage
  data['user'] = jsonEncode({'name': 'Alice'}); // OK
  // bug 6: попытка хранить нестроковые данные напрямую
  // data['count'] = 42; // TypeError — localStorage хранит только строки
  // (неявная попытка присвоить int)

  // bug 7 — использование dart:convert не импортирован
  // (jsonEncode выше не скомпилируется)
}
```

```dart
// fix 1: убираем dart:io — он недоступен в браузере (только dart run / dart compile exe)
import 'dart:html';
import 'dart:convert'; // fix 7: нужен для jsonEncode

void main() {
  // fix 2: проверяем null перед использованием
  final form = querySelector('#loginForm');
  if (form == null) {
    print('Form element not found');
    return;
  }
  // Теперь form имеет тип Element? → Element после проверки
  final formElement = form as FormElement;

  // fix 3: асинхронный HTTP-запрос через Future-API
  HttpRequest.getString('/api/users').then((responseText) {
    print(responseText);
  }).catchError((e) => print('Error: $e'));

  // Или с await в async функции:
  // _loadUsers();

  // fix 4: сохраняем ссылку на handler для последующего удаления
  final btn = document.querySelector('#btn');
  if (btn != null) {
    void handleClick(Event e) {
      print('Clicked!');
      // После удаления элемента из DOM: btn.removeEventListener('click', handleClick);
    }
    btn.addEventListener('click', handleClick);
    // При очистке: btn.removeEventListener('click', handleClick);
    // Или используй onXxx stream с подпиской, которую можно отменить:
    // final sub = btn.onClick.listen((_) => print('Clicked!'));
    // ... позже: sub.cancel();
  }

  // fix 5: НИКОГДА не вставляй необработанный пользовательский ввод через innerHTML
  // Используй .text (textContent) — это автоматически экранирует HTML
  final rawInput = window.location.search;
  final output = document.querySelector('#output');
  if (output != null) {
    // fix: text вместо innerHtml — нет XSS
    output.text = rawInput;

    // Если нужно отобразить форматированный контент от trusted source:
    // output.setInnerHtml(trustedHtml, treeSanitizer: NodeTreeSanitizer.trusted);
    // Но ни в коем случае не с user input!
  }

  // fix 6: всегда сохраняем как строку в localStorage
  final data = window.localStorage;
  data['user'] = jsonEncode({'name': 'Alice'});   // OK: строка

  // fix: числа тоже сохраняем как строки
  var count = 0;
  data['count'] = count.toString();               // fix: toString()

  // Чтение: парсим обратно
  final storedCount = int.tryParse(data['count'] ?? '0') ?? 0;
  print('Count from storage: $storedCount');
}

// fix 3: правильный async HTTP-запрос
Future<void> _loadUsers() async {
  try {
    final response = await HttpRequest.getString('/api/users');
    final users = jsonDecode(response) as List;
    print('Loaded ${users.length} users');
  } catch (e) {
    print('Failed to load users: $e');
  }
}
```

**Семь ошибок:** 1) `dart:io` не существует в браузере — для HTTP используй `dart:html`'s `HttpRequest`. 2) `querySelector()` может вернуть `null` — cast без проверки → `TypeError`. 3) Синхронный XHR (`async: false`) блокирует JS event loop, устарел и запрещён в современных браузерах. 4) `addEventListener` без `removeEventListener` — утечка памяти если элемент удалён; используй `onClick.listen` с `cancel()`. 5) `innerHtml = userInput` — классическая XSS уязвязвимость; используй `.text`. 6) `localStorage` принимает только строки; `int` → `toString()`, обратно — `int.parse`. 7) `jsonEncode` требует `import 'dart:convert'`.
