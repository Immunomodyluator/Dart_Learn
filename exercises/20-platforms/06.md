В Dart-коде с Flutter-паттернами есть ошибки. Найди их.

```dart
// Моделирование Flutter-паттернов на чистом Dart

// bug 1 — setState вызывается после dispose
class TimerState extends State {
  int _seconds = 0;
  Timer? _timer;

  @override
  void initState() {
    _timer = Timer.periodic(Duration(seconds: 1), (_) {
      setState(() => _seconds++); // bug 1: может вызваться после dispose()
    });
  }

  @override
  Widget build() => Text('Seconds: $_seconds');

  @override
  void dispose() {
    // bug 1b: таймер не отменяется!
    print('disposed');
  }
}

// bug 2 — тяжёлые вычисления в build()
class HeavyState extends State {
  @override
  Widget build() {
    // bug 2: сортировка 100k элементов при каждом rebuild
    final data = List.generate(100000, (i) => 100000 - i)..sort();
    return Text('First: ${data.first}');
  }
}

// bug 3 — слушатель не удаляется
class ListenerState extends State {
  late ValueNotifier<int> _notifier;

  void handleChange(int value) {
    print('Changed: $value');
  }

  @override
  void initState() {
    _notifier = ValueNotifier(0);
    _notifier.addListener(handleChange); // bug 3: никогда не removeListener
  }

  @override
  Widget build() => Text('Value: ${_notifier.value}');

  @override
  void dispose() {
    _notifier.dispose(); // dispose без removeListener
  }
}

// bug 4 — создание объектов-ключей в build
```

```dart
import 'dart:async';

// fix 1: проверяем mounted перед setState + отменяем Timer в dispose
class TimerState extends State {
  int _seconds = 0;
  Timer? _timer;

  @override
  void initState() {
    _timer = Timer.periodic(const Duration(seconds: 1), (_) {
      if (!mounted) return; // fix 1: проверяем, что State ещё жив
      setState(() => _seconds++);
    });
  }

  @override
  Widget build() => Text('Seconds: $_seconds');

  @override
  void dispose() {
    _timer?.cancel(); // fix 1b: отменяем Timer в dispose
    super.dispose();
  }
}

// fix 2: тяжёлые вычисления — в initState или кешируем, не в build
class HeavyState extends State {
  late List<int> _sortedData;

  @override
  void initState() {
    // fix 2: вычисляем один раз при инициализации
    _sortedData = List.generate(100000, (i) => 100000 - i)..sort();
  }

  @override
  Widget build() {
    // build остаётся быстрым — читаем уже готовые данные
    return Text('First: ${_sortedData.first}');
  }
}

// fix 3: removeListener в dispose
class ListenerState extends State {
  late ValueNotifier<int> _notifier;

  void handleChange(int value) {
    print('Changed: $value');
  }

  @override
  void initState() {
    _notifier = ValueNotifier(0);
    _notifier.addListener(handleChange);
  }

  @override
  Widget build() => Text('Value: ${_notifier.value}');

  @override
  void dispose() {
    _notifier.removeListener(handleChange); // fix 3: убираем слушатель
    _notifier.dispose();
    super.dispose();
  }
}

// fix 4: ключи стабильны — создаём один раз или используем значение как ключ
class KeyState extends State {
  final _items = ['Apple', 'Banana', 'Cherry'];
  // fix: если нужны уникальные ключи — создаём их в initState, не в build
  final _keys = <String>[]; // ключ = сам элемент (стабильное значение)

  @override
  void initState() {
    _keys.addAll(_items); // ключи стабильны на всё время жизни State
  }

  @override
  Widget build() {
    return Column(children: List.generate(_items.length, (i) {
      // fix 4: используем стабильный ключ (например, индекс или stable id)
      return Text('${_keys[i]}: ${_items[i]}');
    }));
  }
}

// fix 5: dispose всех ValueNotifier
class LeakState extends State {
  final counter = ValueNotifier<int>(0);

  @override
  Widget build() => Text('${counter.value}');

  @override
  void dispose() {
    counter.dispose(); // fix 5: освобождаем ресурс
    super.dispose();
  }
}

// ─── Заглушки адаптированы для корректной компиляции ────────────────────────

abstract class State {
  bool _mounted = true;
  bool get mounted => _mounted;

  void initState() {}
  Widget build();

  void dispose() {
    _mounted = false;
  }

  void setState(void Function() fn) {
    fn();
    if (_mounted) build(); // rebuild
  }
}

abstract class Widget {
  String render();
}

class Text extends Widget {
  final String data;
  Text(this.data);
  @override
  String render() => data;
}

class Column extends Widget {
  final List<Widget> children;
  Column({required this.children});
  @override
  String render() => children.map((w) => w.render()).join('\n');
}

class ValueNotifier<T> {
  T _value;
  final _listeners = <void Function(T)>[];
  bool _disposed = false;

  ValueNotifier(this._value);

  T get value => _value;

  set value(T v) {
    if (_disposed) return;
    _value = v;
    for (final l in List.of(_listeners)) l(v);
  }

  void addListener(void Function(T) l) => _listeners.add(l);
  void removeListener(void Function(T) l) => _listeners.remove(l);

  void dispose() {
    _disposed = true;
    _listeners.clear();
  }
}

Future<void> main() async {
  // TimerState
  final timer = TimerState()..initState();
  await Future.delayed(const Duration(seconds: 2));
  timer.dispose();
  print('Timer state disposed, timer cancelled');

  // HeavyState
  final heavy = HeavyState()..initState();
  print(heavy.build().render()); // быстро

  // LeakState
  final leak = LeakState()..initState();
  leak.counter.value = 5;
  print(leak.build().render()); // 5
  leak.dispose(); // counter.dispose() вызван
}
```

**Пять ошибок:** 1) `setState` вызывается из Timer-callback после `dispose()` — нужна проверка `mounted` + `_timer?.cancel()` в `dispose()`. 2) Сортировка 100k элементов в `build()` — выполняется при каждом rebuild; нужно вынести в `initState`. 3) `addListener` без парного `removeListener` в `dispose` — ValueNotifier удерживает ссылку на disposed State. 4) `Object()` в `build()` создаёт новые нестабильные ключи при каждом вызове — ломает element reconciliation во Flutter. 5) `ValueNotifier` как поле класса, но `dispose()` его не вызывает — утечка памяти и слушателей.
