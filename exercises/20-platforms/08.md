В CLI-приложении есть ошибки. Найди их.

```dart
import 'dart:io';
import 'dart:convert';

// bug 1 — читаем весь stdin сразу, блокируя интерактивный режим
Future<void> readAllInput() async {
  final input = await stdin.readAsString(); // bug 1: блокирует до EOF (Ctrl+D)
  print('Got: $input');
}

// bug 2 — stdout и stderr перепутаны
void processFile(String path) {
  final file = File(path);
  if (!file.existsSync()) {
    stdout.writeln('Error: file not found: $path'); // bug 2: ошибки → stderr
    exit(0);   // bug 2b: выход с кодом 0 при ошибке
  }
  final content = file.readAsStringSync();
  stderr.writeln(content); // bug 2c: контент → stderr вместо stdout
}

// bug 3 — Process.run без обработки ошибок
Future<void> runGit(List<String> args) async {
  final result = await Process.run('git', args);
  // bug 3: игнорируем exitCode и stderr
  print(result.stdout);
}

// bug 4 — путь склеивается вручную (не работает на всех ОС)
String buildPath(String dir, String filename) {
  return '$dir/$filename'; // bug 4: разделитель '/' не работает на Windows
}

// bug 5 — exit() вызывается внутри async кода — Future не завершаются
Future<void> processAsync() async {
  await Future.delayed(Duration(seconds: 1));

  try {
    await Future.delayed(Duration(milliseconds: 100));
    print('done');
  } catch (e) {
    print('Error: $e');
    exit(1); // bug 5: exit в async обрывает event loop, pending Futures не завершаются
             // лучше бросить исключение или вернуть код через return
  }
}

// bug 6 — чтение большого файла целиком в память
Future<void> countLines(String path) async {
  // bug 6: File может быть гигабайтным — загружаем всё в строку
  final content = await File(path).readAsString();
  final lines = content.split('\n').length;
  print('Lines: $lines');
}

// bug 7 — Platform.environment без проверки ключа
void showConfig() {
  final home = Platform.environment['HOME']; // bug 7a: на Windows — null (USERPROFILE)
  final path = Platform.environment['API_KEY']!; // bug 7b: ! без проверки
  print('Home: $home, Key: $path');
}

void main(List<String> args) {
  if (args.isEmpty) {
    print('Usage: tool <path>');
    exit(0); // bug 8: при отсутствии аргументов должен быть exit(1) (usage error)
  }
  processFile(args[0]);
}
```

<summary>Решение</summary>

```dart
import 'dart:io';
import 'dart:convert';

// fix 1: читаем stdin построчно для интерактивного режима
Future<void> readInteractiveInput() async {
  print('Enter text (Ctrl+C to exit):');
  await for (final line in stdin.transform(utf8.decoder).transform(const LineSplitter())) {
    if (line.isEmpty) continue;
    print('Got: $line');
    // Для EOF: проверяем конец потока
  }
}

// fix 1 альт.: readAsString только когда stdin — не терминал (pipe/redirect)
Future<void> readAllInput() async {
  if (stdin.hasTerminal) {
    // Интерактивный режим — построчно
    await readInteractiveInput();
  } else {
    // Piped input — можно читать всё сразу
    final input = await stdin.readAsString();
    print('Got: $input');
  }
}

// fix 2: ошибки → stderr, успех → stdout, коды выхода правильные
void processFile(String path) {
  final file = File(path);
  if (!file.existsSync()) {
    stderr.writeln('Error: file not found: $path'); // fix 2: stderr ✓
    exit(1);   // fix 2b: код 1 при ошибке
    return;
  }
  final content = file.readAsStringSync();
  stdout.writeln(content); // fix 2c: контент → stdout
}

// fix 3: проверяем exitCode и обрабатываем stderr
Future<void> runGit(List<String> args) async {
  final result = await Process.run('git', args);

  if (result.exitCode != 0) {
    stderr.writeln('git error (${result.exitCode}): ${result.stderr}');
    throw ProcessException('git', args, result.stderr.toString(), result.exitCode);
  }

  final out = result.stdout.toString().trim();
  if (out.isNotEmpty) print(out);
}

// fix 4: Platform.pathSeparator / path-пакет
import 'package:path/path.dart' as p;

String buildPath(String dir, String filename) {
  return p.join(dir, filename); // fix: Platform-независимый join
  // Или без пакета: '${dir}${Platform.pathSeparator}$filename'
}

// fix 5: не используем exit() внутри async-логики — бросаем исключение
Future<void> processAsync() async {
  await Future.delayed(const Duration(seconds: 1));

  try {
    await Future.delayed(const Duration(milliseconds: 100));
    print('done');
  } catch (e) {
    stderr.writeln('Error: $e');
    // fix 5: не exit() — выбрасываем исключение или возвращаем код
    rethrow; // main обработает
  }
}

// fix 6: читаем файл потоком — не загружаем в память
Future<void> countLines(String path) async {
  var lines = 0;
  await File(path)
      .openRead()
      .transform(utf8.decoder)
      .transform(const LineSplitter())
      .forEach((_) => lines++);
  print('Lines: $lines'); // работает с файлами любого размера
}

// fix 7: безопасное чтение environment переменных
void showConfig() {
  // fix 7a: кроссплатформенный home directory
  final home = Platform.environment['HOME'] ?? // Unix
               Platform.environment['USERPROFILE'] ?? // Windows
               Directory.current.path; // fallback

  // fix 7b: проверяем наличие перед использованием
  final apiKey = Platform.environment['API_KEY'];
  if (apiKey == null || apiKey.isEmpty) {
    stderr.writeln('Error: API_KEY environment variable is not set');
    exit(1);
  }

  print('Home: $home, Key: ${apiKey.substring(0, 4)}***'); // не логируем ключ целиком
}

Future<void> main(List<String> args) async {
  if (args.isEmpty) {
    stderr.writeln('Usage: tool <path>'); // fix 8a: usage → stderr
    exit(1); // fix 8b: exit(1) — это ошибка использования
  }

  try {
    processFile(args[0]);
    await countLines(args[0]);
    await runGit(['status']);
    showConfig();
    await processAsync();
  } on ProcessException catch (e) {
    stderr.writeln('Process failed: ${e.message}');
    exit(e.errorCode);
  } catch (e) {
    stderr.writeln('Unexpected error: $e');
    exit(1);
  }
}
```

**Восемь ошибок:** 1) `stdin.readAsString()` блокирует до EOF — для интерактивного режима нужен `LineSplitter`. 2) Ошибки в `stdout`, контент в `stderr`, `exit(0)` при ошибке. 3) Игнорируется `result.exitCode` — не знаем, упал ли `git`. 4) Ручная конкатенация пути через `/` — не работает на Windows; нужен `path.join()`. 5) `exit()` в async-коде обрывает все pending Future без cleanup; лучше `rethrow`. 6) `readAsString()` для большого файла — OOM; нужен Stream + `LineSplitter`. 7) `Platform.environment['HOME']` — на Windows `USERPROFILE`; `['API_KEY']!` — `Null check` без проверки. 8) `exit(0)` при отсутствии аргументов — инструменты в UNIX возвращают код 1 при ошибке использования; `-h`/`--help` — проверь сам.
