```dart
/// Задание 1: Widget-подобное дерево
abstract class Widget {
  String render();
}

class Text extends Widget {
  final String data;
  final String? style; // e.g., 'bold'

  Text(this.data, {this.style});

  @override
  String render() {
    if (style == 'bold') return '**$data**';
    return data;
  }
}

class Padding {
  final int all;
  const Padding(this.all);
}

class Container extends Widget {
  final Widget child;
  final Padding? padding;

  Container({required this.child, this.padding});

  @override
  String render() {
    final p = padding?.all ?? 0;
    final indent = ' ' * p;
    return '$indent${child.render()}$indent';
  }
}

class Column extends Widget {
  final List<Widget> children;
  final String separator;

  Column({required this.children, this.separator = '\n'});

  @override
  String render() => children.map((w) => w.render()).join(separator);
}

class Row extends Widget {
  final List<Widget> children;

  Row({required this.children});

  @override
  String render() => children.map((w) => w.render()).join(' | ');
}

/// Задание 2: ValueNotifier
class ValueNotifier<T> {
  T _value;
  final _listeners = <void Function(T)>[];
  bool _disposed = false;

  ValueNotifier(T initial) : _value = initial;

  T get value => _value;

  set value(T newValue) {
    if (_disposed) throw StateError('ValueNotifier has been disposed');
    if (_value == newValue) return; // no change — no notify
    _value = newValue;
    _notifyListeners();
  }

  void addListener(void Function(T) listener) {
    if (_disposed) throw StateError('ValueNotifier has been disposed');
    _listeners.add(listener);
  }

  void removeListener(void Function(T) listener) {
    _listeners.remove(listener);
  }

  void _notifyListeners() {
    for (final listener in List.of(_listeners)) {
      listener(_value);
    }
  }

  void dispose() {
    _disposed = true;
    _listeners.clear();
  }
}

/// Задание 3: StatefulWidget-паттерн
abstract class State<W> {
  W? _widget;
  bool _mounted = false;

  W get widget => _widget!;
  bool get mounted => _mounted;

  void initState() {} // переопределяется subclass-ом
  Widget build();     // обязателен
  void dispose() {}   // переопределяется subclass-ом

  void setState(void Function() fn) {
    fn(); // обновляем внутреннее состояние
    _onRebuild(); // в реальном Flutter — планирует rebuild
  }

  void _onRebuild() {
    if (_mounted) {
      print('[rebuild] ${build().render()}');
    }
  }
}

// Пример конкретного stateful widget
class CounterState extends State<String> {
  int _count = 0;
  late ValueNotifier<int> _notifier;

  @override
  void initState() {
    _notifier = ValueNotifier(0);
    _notifier.addListener((v) => print('[notifier] count = $v'));
  }

  void increment() {
    setState(() {
      _count++;
      _notifier.value = _count;
    });
  }

  @override
  Widget build() {
    return Column(children: [
      Text('Counter', style: 'bold'),
      Text('Count: $_count'),
    ]);
  }

  @override
  void dispose() {
    _notifier.dispose();
    print('[dispose] CounterState cleaned up');
  }
}

// Фреймворк-функция монтирования
CounterState mountCounter(String widget) {
  final state = CounterState();
  state._widget = widget;
  state._mounted = true;
  state.initState();
  print('[mount] ${state.build().render()}');
  return state;
}

void unmount(State state) {
  state.dispose();
  state._mounted = false;
}

void main() {
  // Задание 1: Widget-дерево
  final ui = Container(
    padding: const Padding(2),
    child: Column(children: [
      Text('Title', style: 'bold'),
      Row(children: [Text('Name'), Text('Age')]),
      Text('Footer'),
    ]),
  );
  print(ui.render());
  print('---');

  // Задание 2: ValueNotifier
  final counter = ValueNotifier<int>(0);
  counter.addListener((v) => print('UI update: counter = $v'));
  counter.value = 1; // → UI update: counter = 1
  counter.value = 1; // нет изменения — нет уведомления
  counter.value = 2; // → UI update: counter = 2
  counter.dispose();

  print('---');

  // Задание 3: StatefulWidget
  final state = mountCounter('CounterWidget');
  state.increment(); // rebuild
  state.increment(); // rebuild
  unmount(state);    // dispose
}
```

**Ключевые Flutter-концепции:** 1) Widget-дерево — композиция маленьких объектов; `render()` эквивалентен `build()` в Flutter. 2) `ValueNotifier` — легковесный observable; изменение `.value` вызывает все listeners (аналог Flutter's `ValueListenableBuilder`). 3) `State.initState`/`dispose` — lifecycle методы; `setState` сигнализирует фреймворку о необходимости rebuild; `mounted` флаг защищает от вызовов после dispose.
