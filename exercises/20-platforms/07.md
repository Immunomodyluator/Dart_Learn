Напиши CLI-утилиту для работы с файлами в текущей директории.

**Требования:**

Программа `dartls` — аналог `ls` с фильтрацией:

```
Usage: dartls [options] [directory]

Options:
  -e, --ext <ext>      Filter by extension (e.g. .dart, .md)
  -s, --sort           Sort by size descending
  -r, --recursive      Include subdirectories
  -h, --help           Show help

Output:
  filename.ext    1.2 KB
  other.dart      456 B
  ---
  Total: 3 files, 2.8 KB
```

**Требования к реализации:**

- Разбор аргументов вручную (без внешних пакетов) или с `package:args`
- Форматирование размера (`B`, `KB`, `MB`)
- Красивый вывод с колонками (выравнивание по пробелам)
- Обработка ошибок (директория не найдена, нет прав доступа)
- Код завершения: 0 при успехе, 1 при ошибке

```dart
// bin/dartls.dart
import 'dart:io';

// ─── Форматирование ───────────────────────────────────────────────────────────

String formatSize(int bytes) {
  if (bytes >= 1024 * 1024) return '${(bytes / 1024 / 1024).toStringAsFixed(1)} MB';
  if (bytes >= 1024) return '${(bytes / 1024).toStringAsFixed(1)} KB';
  return '$bytes B';
}

// ─── Парсер аргументов ────────────────────────────────────────────────────────

class CliOptions {
  final String? directory;
  final String? ext;
  final bool sort;
  final bool recursive;
  final bool help;

  const CliOptions({
    this.directory,
    this.ext,
    this.sort = false,
    this.recursive = false,
    this.help = false,
  });

  static CliOptions parse(List<String> args) {
    String? directory;
    String? ext;
    var sort = false;
    var recursive = false;
    var help = false;

    for (var i = 0; i < args.length; i++) {
      final arg = args[i];
      switch (arg) {
        case '-h' || '--help':
          help = true;
        case '-s' || '--sort':
          sort = true;
        case '-r' || '--recursive':
          recursive = true;
        case '-e' || '--ext':
          if (i + 1 >= args.length) {
            throw ArgumentError('Option $arg requires a value');
          }
          ext = args[++i];
          if (!ext.startsWith('.')) ext = '.$ext';
        default:
          if (arg.startsWith('-')) {
            throw ArgumentError('Unknown option: $arg');
          }
          directory = arg;
      }
    }

    return CliOptions(
      directory: directory,
      ext: ext,
      sort: sort,
      recursive: recursive,
      help: help,
    );
  }
}

// ─── Логика ───────────────────────────────────────────────────────────────────

Future<List<File>> listFiles(
  Directory dir, {
  String? ext,
  bool recursive = false,
}) async {
  final stream = recursive
      ? dir.list(recursive: true, followLinks: false)
      : dir.list(followLinks: false);

  final files = <File>[];
  await for (final entity in stream) {
    if (entity is File) {
      if (ext == null || entity.path.endsWith(ext)) {
        files.add(entity);
      }
    }
  }
  return files;
}

// ─── Форматирование вывода ────────────────────────────────────────────────────

Future<void> printTable(List<File> files, String rootPath) async {
  if (files.isEmpty) {
    print('No files found.');
    return;
  }

  // Собираем данные
  final rows = <(String, int)>[];
  for (final file in files) {
    final name = file.path.replaceFirst('$rootPath${Platform.pathSeparator}', '');
    final size = await file.length();
    rows.add((name, size));
  }

  // Максимальная ширина имени для выравнивания
  final maxLen = rows.fold(0, (m, r) => r.$1.length > m ? r.$1.length : m);

  var totalBytes = 0;
  for (final (name, size) in rows) {
    final padding = ' ' * (maxLen - name.length + 4);
    print('$name$padding${formatSize(size)}');
    totalBytes += size;
  }

  print('─' * (maxLen + 12));
  print('Total: ${rows.length} file${rows.length == 1 ? '' : 's'}, ${formatSize(totalBytes)}');
}

const _helpText = '''
Usage: dart run bin/dartls.dart [options] [directory]

Options:
  -e, --ext <ext>    Filter by extension (e.g. .dart, .md)
  -s, --sort         Sort by size descending
  -r, --recursive    Include subdirectories
  -h, --help         Show this help
''';

// ─── Main ─────────────────────────────────────────────────────────────────────

Future<void> main(List<String> args) async {
  CliOptions opts;
  try {
    opts = CliOptions.parse(args);
  } on ArgumentError catch (e) {
    stderr.writeln('Error: $e');
    stderr.writeln('Use --help for usage.');
    exit(1);
  }

  if (opts.help) {
    print(_helpText);
    exit(0);
  }

  final dirPath = opts.directory ?? '.';
  final dir = Directory(dirPath);

  if (!await dir.exists()) {
    stderr.writeln('Error: Directory not found: $dirPath');
    exit(1);
  }

  List<File> files;
  try {
    files = await listFiles(dir, ext: opts.ext, recursive: opts.recursive);
  } on FileSystemException catch (e) {
    stderr.writeln('Error reading directory: ${e.message}');
    exit(1);
  }

  if (opts.sort) {
    // Сортируем по размеру (нужно получить размеры заранее)
    final sized = await Future.wait(
      files.map((f) async => (f, await f.length())),
    );
    sized.sort((a, b) => b.$2.compareTo(a.$2)); // по убыванию
    files = sized.map((e) => e.$1).toList();
  }

  final rootPath = dir.absolute.path;
  await printTable(files, rootPath);
}
```

**Запуск:**

```bash
# Список .dart файлов рекурсивно, отсортированных по размеру
dart run bin/dartls.dart -r -s -e .dart lib/

# Скомпилировать в нативный бинарник
dart compile exe bin/dartls.dart -o dartls
./dartls -r -e .md .
```

**Ключевые паттерны CLI:** `stdin`/`stdout`/`stderr` из `dart:io`; `exit(code)` для кода завершения; `Directory.list(recursive:)` — асинхронный Stream файловой системы; форматирование с выравниванием через пробелы. `dart compile exe` создаёт self-contained нативный бинарник без зависимостей от Dart SDK.
