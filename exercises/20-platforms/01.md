# Задание 01 (20.1): Dart на сервере (Shelf, REST API) — 🖊 Написать код

**Тема:** 20. Платформенные применения Dart | **Подтема:** 20.1 Dart на сервере (Shelf, REST API)

---

Реализуй простой REST API для управления задачами (to-do) с использованием пакета `shelf`.

**Требования:**

1. Эндпоинты:
   - `GET /tasks` — список всех задач
   - `GET /tasks/:id` — задача по ID
   - `POST /tasks` — создать задачу (тело: `{"title": "..."}`)
   - `PUT /tasks/:id` — обновить задачу (`{"title": "...", "done": true}`)
   - `DELETE /tasks/:id` — удалить задачу

2. Ответы в JSON, статус коды: 200, 201, 400, 404

3. Используй `shelf_router` для маршрутизации

**pubspec.yaml:**

```yaml
dependencies:
  shelf: ^1.4.0
  shelf_router: ^1.1.0
```

```dart
// bin/server.dart
import 'dart:convert';
import 'dart:io';
import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as io;
import 'package:shelf_router/shelf_router.dart';

// ─── Модель ───────────────────────────────────────────────────────────────────

class Task {
  final String id;
  String title;
  bool done;
  final DateTime createdAt;

  Task({required this.id, required this.title, this.done = false})
      : createdAt = DateTime.now();

  Map<String, dynamic> toJson() => {
        'id': id,
        'title': title,
        'done': done,
        'createdAt': createdAt.toIso8601String(),
      };
}

// ─── Репозиторий (in-memory) ───────────────────────────────────────────────────

class TaskRepository {
  final _tasks = <String, Task>{};
  int _nextId = 1;

  List<Task> getAll() => _tasks.values.toList();

  Task? findById(String id) => _tasks[id];

  Task create(String title) {
    final id = (_nextId++).toString();
    final task = Task(id: id, title: title);
    _tasks[id] = task;
    return task;
  }

  Task? update(String id, {String? title, bool? done}) {
    final task = _tasks[id];
    if (task == null) return null;
    if (title != null) task.title = title;
    if (done != null) task.done = done;
    return task;
  }

  bool delete(String id) => _tasks.remove(id) != null;
}

// ─── Хелперы ответов ──────────────────────────────────────────────────────────

Response jsonOk(Object data) => Response.ok(
      jsonEncode(data),
      headers: {'content-type': 'application/json'},
    );

Response jsonCreated(Object data) => Response(
      201,
      body: jsonEncode(data),
      headers: {'content-type': 'application/json'},
    );

Response jsonError(int status, String message) => Response(
      status,
      body: jsonEncode({'error': message}),
      headers: {'content-type': 'application/json'},
    );

// ─── Хендлеры ─────────────────────────────────────────────────────────────────

class TaskHandler {
  final TaskRepository _repo;
  TaskHandler(this._repo);

  // GET /tasks
  Response getAll(Request req) {
    final tasks = _repo.getAll().map((t) => t.toJson()).toList();
    return jsonOk(tasks);
  }

  // GET /tasks/:id
  Response getById(Request req, String id) {
    final task = _repo.findById(id);
    if (task == null) return jsonError(404, 'Task not found: $id');
    return jsonOk(task.toJson());
  }

  // POST /tasks
  Future<Response> create(Request req) async {
    final body = await req.readAsString();
    final Map<String, dynamic> json;
    try {
      json = jsonDecode(body) as Map<String, dynamic>;
    } on FormatException {
      return jsonError(400, 'Invalid JSON');
    }

    final title = json['title'];
    if (title == null || title is! String || title.trim().isEmpty) {
      return jsonError(400, 'Field "title" is required and must be non-empty string');
    }

    final task = _repo.create(title.trim());
    return jsonCreated(task.toJson());
  }

  // PUT /tasks/:id
  Future<Response> update(Request req, String id) async {
    final body = await req.readAsString();
    final Map<String, dynamic> json;
    try {
      json = jsonDecode(body) as Map<String, dynamic>;
    } on FormatException {
      return jsonError(400, 'Invalid JSON');
    }

    final title = json['title'] as String?;
    final done = json['done'] as bool?;

    final task = _repo.update(id, title: title, done: done);
    if (task == null) return jsonError(404, 'Task not found: $id');
    return jsonOk(task.toJson());
  }

  // DELETE /tasks/:id
  Response delete(Request req, String id) {
    final deleted = _repo.delete(id);
    if (!deleted) return jsonError(404, 'Task not found: $id');
    return Response(204); // No Content
  }
}

// ─── Маршрутизация ────────────────────────────────────────────────────────────

Handler createRouter(TaskRepository repo) {
  final handler = TaskHandler(repo);
  final router = Router();

  router.get('/tasks',        handler.getAll);
  router.get('/tasks/<id>',   handler.getById);
  router.post('/tasks',       handler.create);
  router.put('/tasks/<id>',   handler.update);
  router.delete('/tasks/<id>', handler.delete);

  return const Pipeline()
      .addMiddleware(logRequests())       // логируем запросы
      .addHandler(router.call);
}

// ─── Main ─────────────────────────────────────────────────────────────────────

Future<void> main() async {
  final repo = TaskRepository();
  final handler = createRouter(repo);

  final server = await io.serve(handler, InternetAddress.loopbackIPv4, 8080);
  print('Server running at http://localhost:${server.port}');
}
```

**Тесты вручную (curl):**

```bash
# Создать задачу
curl -X POST http://localhost:8080/tasks \
     -H 'Content-Type: application/json' \
     -d '{"title": "Buy milk"}'
# {"id":"1","title":"Buy milk","done":false,...}

# Список
curl http://localhost:8080/tasks

# Обновить
curl -X PUT http://localhost:8080/tasks/1 \
     -H 'Content-Type: application/json' \
     -d '{"done": true}'

# Удалить
curl -X DELETE http://localhost:8080/tasks/1
```

**Ключевые концепции Shelf:** `Request`/`Response` — иммутабельные объекты запроса/ответа. `Router` из `shelf_router` сопоставляет URL-паттерны с хендлерами. `Pipeline.addMiddleware` добавляет cross-cutting functionality (логирование, CORS, auth). Каждый хендлер — просто функция `(Request) → Response`.
