В коде Shelf-сервера есть ошибки. Найди их.

```dart
import 'dart:convert';
import 'dart:io';
import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as io;
import 'package:shelf_router/shelf_router.dart';

// bug 1 — хендлер не async, но читает тело запроса
Response createItem(Request request) {
  final body = request.readAsString(); // возвращает Future<String>, не String
  final json = jsonDecode(body as String) as Map; // body — это Future, не String
  return Response.ok(jsonEncode({'created': json['name']}));
}

// bug 2 — подвисший response при ошибке парсинга
Future<Response> updateItem(Request request, String id) async {
  final body = await request.readAsString();
  final json = jsonDecode(body) as Map<String, dynamic>; // FormatException не обработан
  return Response.ok(jsonEncode({'updated': id, 'data': json}));
}

// bug 3 — роутер регистрирует handler с неверным синтаксисом параметра
Handler buildRouter() {
  final router = Router();

  router.get('/items/:id', (Request req) {       // bug 3: :id — rails-style, не shelf_router
    return Response.ok('item');                   // shelf_router использует <id>
  });

  router.post('/items', createItem);

  return router.call;
}

// bug 4 — сервер слушает на всех интерфейсах вместо loopback на dev
Future<void> startServer() async {
  final server = await io.serve(
    buildRouter(),
    '0.0.0.0', // bug 4: при разработке лучше loopback (127.0.0.1)
    8080,
  );
  print('Listening on ${server.port}');
}

// bug 5 — Content-Type не указан, клиент не может распарсить JSON
Response getItems(Request request) {
  final items = [{'id': '1', 'name': 'Apple'}];
  return Response.ok(jsonEncode(items)); // bug 5: нет Content-Type: application/json
}

// bug 6 — ошибка не возвращает JSON-тело, а только статус код
Future<Response> deleteItem(Request request, String id) async {
  final deleted = await _fakeDelete(id);
  if (!deleted) {
    return Response.notFound('Not found'); // bug 6: тело — plain text, не JSON
  }
  return Response(204);
}

Future<bool> _fakeDelete(String id) async => id != '999';

void main() => startServer();
```

```dart
import 'dart:convert';
import 'dart:io';
import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as io;
import 'package:shelf_router/shelf_router.dart';

// fix 1: async хендлер + await для чтения тела
Future<Response> createItem(Request request) async {
  final body = await request.readAsString(); // fix: await
  Map<String, dynamic> json;
  try {
    json = jsonDecode(body) as Map<String, dynamic>;
  } on FormatException {
    return Response(400,
        body: jsonEncode({'error': 'Invalid JSON'}),
        headers: {'content-type': 'application/json'});
  }
  if (json['name'] == null) {
    return Response(400,
        body: jsonEncode({'error': 'Field name is required'}),
        headers: {'content-type': 'application/json'});
  }
  return Response(201, // fix: 201 Created, не 200
      body: jsonEncode({'created': json['name']}),
      headers: {'content-type': 'application/json'});
}

// fix 2: обработка FormatException
Future<Response> updateItem(Request request, String id) async {
  final body = await request.readAsString();
  Map<String, dynamic> json;
  try {
    json = jsonDecode(body) as Map<String, dynamic>; // fix: try/catch
  } on FormatException {
    return Response(400,
        body: jsonEncode({'error': 'Invalid JSON'}),
        headers: {'content-type': 'application/json'});
  }
  return Response.ok(
    jsonEncode({'updated': id, 'data': json}),
    headers: {'content-type': 'application/json'},
  );
}

// fix 3: shelf_router использует <id> не :id
Handler buildRouter() {
  final router = Router();

  router.get('/items/<id>', (Request req, String id) { // fix: <id>
    return Response.ok(
      jsonEncode({'id': id}),
      headers: {'content-type': 'application/json'},
    );
  });

  router.post('/items', createItem);
  router.put('/items/<id>', updateItem);
  router.delete('/items/<id>', deleteItem);
  router.get('/items', getItems);

  return const Pipeline()
      .addMiddleware(logRequests())
      .addHandler(router.call);
}

// fix 4: при разработке — loopback; в production передавать через env
Future<void> startServer({bool production = false}) async {
  final address = production
      ? InternetAddress.anyIPv4        // 0.0.0.0 — только в prod
      : InternetAddress.loopbackIPv4;  // fix: 127.0.0.1 при разработке

  final server = await io.serve(buildRouter(), address, 8080);
  print('Listening on http://localhost:${server.port}');
}

// fix 5: Content-Type header
Response getItems(Request request) {
  final items = [{'id': '1', 'name': 'Apple'}];
  return Response.ok(
    jsonEncode(items),
    headers: {'content-type': 'application/json'}, // fix: всегда указываем тип
  );
}

// fix 6: JSON-тело в ошибках
Future<Response> deleteItem(Request request, String id) async {
  final deleted = await _fakeDelete(id);
  if (!deleted) {
    return Response.notFound(
      jsonEncode({'error': 'Item not found', 'id': id}), // fix: JSON
      headers: {'content-type': 'application/json'},
    );
  }
  return Response(204); // No Content — тела нет
}

Future<bool> _fakeDelete(String id) async => id != '999';

void main() => startServer();
```

**Шесть ошибок:** 1) `request.readAsString()` возвращает `Future<String>` — без `await` и `async` хендлера тело запроса не будет прочитано. 2) `jsonDecode(body)` без try/catch — некорректный JSON приведёт к необработанному `FormatException` и 500. 3) `shelf_router` использует синтаксис `<paramName>`, а не `:paramName` (как в Express/Sinatra). 4) `0.0.0.0` при разработке открывает порт на всех сетевых интерфейсах — риск случайной экспозиции. 5) `Response.ok(jsonEncode(...))` без `content-type: application/json` — браузер/клиент не знает, что это JSON. 6) `Response.notFound('Not found')` — plain text тело при JSON API нарушает консистентность; клиент ожидает `{"error": "..."}`.
