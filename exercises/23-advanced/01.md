# Задание 01 (23.1): Глубокое изучение типов и компиляции — 🖊 Написать код

**Тема:** 23. Переход к продвинутым темам | **Подтема:** 23.1 Глубокое изучение типов и компиляции

---

Исследуй продвинутые возможности системы типов Dart 3 и механизм компиляции.

**Требования:**

1. **Variance демонстрация** — покажи разницу между ковариантностью и контравариантностью через примеры с `List<Animal>` vs `List<Cat>` и функциональными типами (producer covariant, consumer contravariant).

2. **`TypedRegistry<T>`** — типобезопасный реестр с регистрацией по `Type`:
   - `register<S extends T>(String name, S Function() factory)`
   - `resolve<S extends T>(String name) → S`
   - При запросе несовместимого типа — `TypeError`, не `CastError` в runtime

3. **Sealed class + exhaustive switch** — `Result<T>` с тремя вариантами: `Success<T>`, `Failure`, `Loading`. Покажи, что компилятор требует обработки всех вариантов.

4. **Compile-time vs runtime** — продемонстрируй разницу между `const`, `final` и `late final`; покажи, когда `is` и `as` необходимы.

5. **Extension types** — `UserId`, `ProductId` как extension type над `int`: один тип нельзя перепутать с другим на этапе компиляции.

```dart
// ════════════════════════════════════════════════════════════════
// 1. Variance (ковариантность и контравариантность)
// ════════════════════════════════════════════════════════════════

class Animal {
  final String name;
  Animal(this.name);
}

class Cat extends Animal {
  Cat(super.name);
  void purr() => print('$name purrs');
}

void varianceDemo() {
  // КОВАРИАНТНОСТЬ: List в Dart ковариантен (для удобства)
  final List<Cat> cats = [Cat('Luna'), Cat('Mochi')];
  final List<Animal> animals = cats;  // OK — Cat IS-A Animal

  // Но это небезопасно в runtime при записи:
  try {
    animals.add(Animal('Dog'));  // CastError в runtime!
  } on TypeError catch (e) {
    print('Covariance trap: $e');
  }

  // ПРОДЬЮСЕР (ковариантный): возвращает T — безопасно читать как Animal
  T produce<T extends Animal>(T Function() factory) => factory();
  Animal a = produce(() => Cat('Kira'));  // OK

  // КОНСЬЮМЕР (контравариантный): принимает T — требует Animal, не только Cat
  void consume(void Function(Animal) handler, Animal animal) {
    handler(animal);
  }
  // Функция (Animal → void) безопасна там, где ожидается (Cat → void)
  consume((a) => print(a.name), Cat('Tom'));
}

// ════════════════════════════════════════════════════════════════
// 2. TypedRegistry<T>
// ════════════════════════════════════════════════════════════════

class TypedRegistry<T extends Object> {
  final _factories = <String, (Type, Object Function())>{};

  void register<S extends T>(String name, S Function() factory) {
    _factories[name] = (S, factory);
  }

  S resolve<S extends T>(String name) {
    final entry = _factories[name];
    if (entry == null) throw StateError('No factory registered for "$name"');

    final (registeredType, factory) = entry;
    final instance = factory();

    // Явная type-safe проверка перед кастом
    if (instance is! S) {
      throw TypeError();
    }
    return instance;
  }

  bool has(String name) => _factories.containsKey(name);
}

abstract class Service { String get name; }
class EmailService implements Service {
  @override String get name => 'EmailService';
}
class SmsService implements Service {
  @override String get name => 'SmsService';
}

void registryDemo() {
  final registry = TypedRegistry<Service>();
  registry.register<EmailService>('email', EmailService.new);
  registry.register<SmsService>('sms', SmsService.new);

  final email = registry.resolve<EmailService>('email');
  print(email.name);  // EmailService

  // Попытка запросить неверный тип — TypeError, не CastError в runtime
  try {
    registry.resolve<SmsService>('email');  // email → EmailService, не SmsService
  } on TypeError {
    print('Type mismatch detected at resolve time');
  }
}

// ════════════════════════════════════════════════════════════════
// 3. Sealed class + exhaustive switch
// ════════════════════════════════════════════════════════════════

sealed class Result<T> {
  const Result();
}

final class Success<T> extends Result<T> {
  final T value;
  const Success(this.value);
}

final class Failure<T> extends Result<T> {
  final Object error;
  final StackTrace? stackTrace;
  const Failure(this.error, [this.stackTrace]);
}

final class Loading<T> extends Result<T> {
  const Loading();
}

// Компилятор требует обработки ВСЕХ вариантов sealed class
String describeResult<T>(Result<T> result) => switch (result) {
      Success(:final value) => 'Success: $value',
      Failure(:final error) => 'Error: $error',
      Loading() => 'Loading...',
      // Если убрать любую ветку — compile error: missing case
    };

Future<Result<String>> fetchData(bool succeed) async {
  await Future.delayed(const Duration(milliseconds: 50));
  return succeed
      ? const Success('Hello, World!')
      : const Failure('Network error');
}

// ════════════════════════════════════════════════════════════════
// 4. const vs final vs late final
// ════════════════════════════════════════════════════════════════

// CONST: вычисляется во время компиляции, единственный экземпляр
const pi = 3.14159265358979;
const greeting = 'Hello';
const list = [1, 2, 3];  // compile-time constant

class Circle {
  // const конструктор — все поля должны быть final
  final double radius;
  const Circle(this.radius);

  double get area => pi * radius * radius;
}

void constDemo() {
  const c1 = Circle(5);
  const c2 = Circle(5);
  print(identical(c1, c2));  // true — один экземпляр

  // FINAL: присваивается один раз, runtime value
  final now = DateTime.now();  // не const — значение в runtime

  // LATE FINAL: отложенная инициализация, без nullable
  late final String config;
  // config используется без инициализации — LateInitializationError
  config = 'loaded_value';
  print(config);  // OK

  // IS и AS: когда необходимы
  Object value = 'some string';
  if (value is String) {
    // В этом блоке value автоматически кастован (smart cast)
    print(value.toUpperCase());  // нет явного as
  }

  // AS нужен когда мы ТОЧНО знаем тип, но компилятор — нет
  final dynamic dyn = 42;
  final num asNum = dyn as num;  // явный downcast
  print(asNum);
}

// ════════════════════════════════════════════════════════════════
// 5. Extension types — newtype pattern
// ════════════════════════════════════════════════════════════════

extension type UserId(int value) {
  // Нельзя создать из произвольного int без явного конструктора
  bool get isValid => value > 0;
}

extension type ProductId(int value) {
  bool get isValid => value > 0;
}

// Функция принимает UserId и ProductId — нельзя перепутать в compile time
String buildOrderKey(UserId userId, ProductId productId) =>
    'order_${userId.value}_${productId.value}';

void extensionTypeDemo() {
  const userId = UserId(42);
  const productId = ProductId(7);

  print(buildOrderKey(userId, productId));  // order_42_7

  // Следующее вызовет ошибку компиляции — типы несовместимы:
  // buildOrderKey(productId, userId);  // Error: argument type 'ProductId' can't be assigned to 'UserId'
  // buildOrderKey(42, 7);              // Error: int can't be assigned to UserId

  print(userId.isValid);  // true
}

void main() {
  varianceDemo();
  registryDemo();
  constDemo();
  extensionTypeDemo();

  fetchData(true).then((r) => print(describeResult(r)));
  fetchData(false).then((r) => print(describeResult(r)));
}
```

Extension types — это **zero-cost** обёртки на этапе компиляции: `UserId` и `ProductId` полностью эквивалентны `int` в скомпилированном коде, но компилятор отличает их статически. `sealed class` гарантирует исчерпывающий `switch`— добавление нового подкласса сразу выявит все места, которые нужно обновить. `TypedRegistry` разрешает регистрацию и извлечение с compile-time safety через generics с bounded type parameters.
