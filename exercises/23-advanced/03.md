// lib/src/result.dart
/// A discriminated union that either holds a success [value] of type [T]
/// or a failure [error] of type [E].
sealed class Result<T, E extends Object> {
  const Result();

  /// Creates a successful result wrapping [value].
  const factory Result.ok(T value) = Ok<T, E>;

  /// Creates a failed result wrapping [error].
  const factory Result.err(E error) = Err<T, E>;

  /// Returns `true` if this is an [Ok].
  bool get isOk => this is Ok<T, E>;

  /// Returns `true` if this is an [Err].
  bool get isErr => this is Err<T, E>;

  /// Transforms the success value with [transform], leaving errors intact.
  ///
  /// Example:
  /// ```dart
  /// Result.ok(2).map((n) => n * 2); // Ok(4)
  /// Result.err('bad').map((n) => n * 2); // Err('bad')
  /// ```
  Result<U, E> map<U>(U Function(T value) transform) => switch (this) {
        Ok(:final value) => Ok(transform(value)),
        Err(:final error) => Err(error),
      };

  /// Chains results — [transform] must return a [Result].
  ///
  /// Example:
  /// ```dart
  /// final result = Result.ok('42')
  ///     .flatMap((s) {
  ///       final n = int.tryParse(s);
  ///       return n != null ? Result.ok(n) : Result.err('Not a number');
  ///     });
  /// ```
  Result<U, E> flatMap<U>(Result<U, E> Function(T value) transform) =>
      switch (this) {
        Ok(:final value) => transform(value),
        Err(:final error) => Err(error),
      };

  /// Transforms the error type with [transform], leaving success intact.
  Result<T, F> mapError<F extends Object>(F Function(E error) transform) =>
      switch (this) {
        Ok(:final value) => Ok(value),
        Err(:final error) => Err(transform(error)),
      };

  /// Returns the success value or [defaultValue] if this is an [Err].
  T getOrElse(T defaultValue) => switch (this) {
        Ok(:final value) => value,
        Err() => defaultValue,
      };

  /// Reduces to a single value by applying [onOk] or [onErr].
  ///
  /// Example:
  /// ```dart
  /// final msg = result.fold(
  ///   onOk: (v) => 'Got $v',
  ///   onErr: (e) => 'Failed: $e',
  /// );
  /// ```
  U fold<U>({
    required U Function(T value) onOk,
    required U Function(E error) onErr,
  }) =>
      switch (this) {
        Ok(:final value) => onOk(value),
        Err(:final error) => onErr(error),
      };

  @override
  String toString() => switch (this) {
        Ok(:final value) => 'Ok($value)',
        Err(:final error) => 'Err($error)',
      };
}

/// The success variant of [Result].
final class Ok<T, E extends Object> extends Result<T, E> {
  /// The wrapped success value.
  final T value;
  const Ok(this.value);

  @override
  bool operator ==(Object other) => other is Ok<T, E> && other.value == value;
  @override
  int get hashCode => Object.hash(Ok, value);
}

/// The failure variant of [Result].
final class Err<T, E extends Object> extends Result<T, E> {
  /// The wrapped error.
  final E error;
  const Err(this.error);

  @override
  bool operator ==(Object other) => other is Err<T, E> && other.error == error;
  @override
  int get hashCode => Object.hash(Err, error);
}

```dart
// example/result_example.dart
import 'package:dart_result/result.dart';

// Функция, возвращающая Result вместо выброса исключения
Result<int, String> divide(int a, int b) {
  if (b == 0) return const Err('Division by zero');
  return Ok(a ~/ b);
}

Result<double, String> sqrt(int n) {
  if (n < 0) return const Err('Cannot take sqrt of negative number');
  import 'dart:math' as math;
  return Ok(math.sqrt(n.toDouble()));
}

void main() {
  // 1. Basic usage
  final r1 = divide(10, 2);
  print(r1);  // Ok(5)

  final r2 = divide(10, 0);
  print(r2);  // Err(Division by zero)

  // 2. map — трансформация успешного значения
  final doubled = divide(10, 2).map((n) => n * 2);
  print(doubled);  // Ok(10)

  // 3. flatMap — цепочка операций, каждая может провалиться
  final result = divide(100, 4).flatMap(sqrt);
  print(result);  // Ok(5.0)

  // 4. getOrElse — значение по умолчанию
  final safe = divide(10, 0).getOrElse(-1);
  print(safe);  // -1

  // 5. fold — финальная обработка обоих путей
  final message = divide(10, 2).fold(
    onOk: (v) => 'Result: $v',
    onErr: (e) => 'Failed: $e',
  );
  print(message);  // Result: 5

  // 6. mapError — конвертация типа ошибки
  final withCode = divide(10, 0)
      .mapError((msg) => {'code': 400, 'message': msg});
  print(withCode);  // Err({code: 400, message: Division by zero})
}

# dart_result

A type-safe `Result<T, E>` monad for Dart that models success and failure
explicitly, inspired by Rust's `Result` type.

[![pub package](https://img.shields.io/pub/v/dart_result.svg)](https://pub.dev/packages/dart_result)

## Features

- Sealed `Result<T, E>` with `Ok<T, E>` and `Err<T, E>` variants
- `map`, `flatMap`, `mapError`, `getOrElse`, `fold` combinators
- Exhaustive switch support via sealed class
- Zero dependencies

## Getting Started

````yaml
dependencies:
  dart_result: ^1.0.0
\```

## Usage

```dart
import 'package:dart_result/result.dart';

Result<int, String> divide(int a, int b) =>
    b == 0 ? const Err('Division by zero') : Ok(a ~/ b);

void main() {
  final result = divide(10, 2)
      .map((n) => n * 10)
      .getOrElse(0);
  print(result); // 50
}
\```

## API Reference

| Method | Description |
|--------|-------------|
| `map<U>(U Function(T))` | Transform success value |
| `flatMap<U>(Result<U,E> Function(T))` | Chain fallible operations |
| `mapError<F>(F Function(E))` | Transform error type |
| `getOrElse(T default)` | Unwrap or default |
| `fold({onOk, onErr})` | Reduce to single value |
