В генераторе кода допущены ошибки, из-за которых он не будет работать. Найди их.

```dart
// lib/builder.dart
import 'package:build/build.dart';
import 'package:source_gen/source_gen.dart';
import 'package:analyzer/dart/element/element.dart';

class LoggerGenerator extends GeneratorForAnnotation<dynamic> { // bug 1: dynamic вместо типа аннотации
  @override
  String generateForAnnotatedElement(
    Element element,
    ConstantReader annotation,
    BuildStep buildStep,
  ) {
    final className = element.displayName;

    // bug 2 — нет проверки, что element является ClassElement
    final fields = (element as ClassElement).fields;

    return '''
// Сгенерировано для $className
class ${className}Logger {
  void log(String msg) => print('[$className] \$msg');
  ${fields.map((f) => 'void log${f.displayName}() => log("${f.displayName}: \${instance.${f.displayName}}");').join('\n  ')}
}
''';                  // bug 3 — 'instance' не определён в сгенерированном коде
  }
}

// bug 4 — неверная фабрика: возвращает GeneratorForAnnotation напрямую, а не Builder
Builder loggerBuilder(BuilderOptions options) => LoggerGenerator();
```

```dart
// lib/src/annotations.dart
class AddLogger {
  const AddLogger();
```

```dart
// lib/builder.dart
import 'package:build/build.dart';
import 'package:source_gen/source_gen.dart';
import 'package:analyzer/dart/element/element.dart';

import 'src/annotations.dart';

class LoggerGenerator extends GeneratorForAnnotation<AddLogger> { // fix 1: конкретный тип
  @override
  String generateForAnnotatedElement(
    Element element,
    ConstantReader annotation,
    BuildStep buildStep,
  ) {
    // fix 2: проверяем тип элемента
    if (element is! ClassElement) {
      throw InvalidGenerationSourceError(
        '@AddLogger can only be applied to classes.',
        element: element,
      );
    }

    final className = element.name;
    final fields = element.fields
        .where((f) => !f.isStatic && !f.name.startsWith('_'))
        .toList();

    // fix 3: генерируем extension с правильным параметром instance
    final logMethods = fields.map((f) =>
      '  void log${_capitalise(f.name)}(${className} instance) => '
      'log("${f.name}: \${instance.${f.name}}");'
    ).join('\n');

    return '''
class ${className}Logger {
  final String _prefix;
  const ${className}Logger([this._prefix = '$className']);

  void log(String msg) => print('[\$_prefix] \$msg');
$logMethods
}
''';
  }

  String _capitalise(String s) =>
      s.isEmpty ? s : s[0].toUpperCase() + s.substring(1);
}

// fix 4: используем SharedPartBuilder для создания .g.dart части
Builder loggerBuilder(BuilderOptions options) =>
    SharedPartBuilder([LoggerGenerator()], 'logger');
```

```yaml
# build.yaml — fix 5: добавляем build_extensions
builders:
  logger_gen:
    import: "package:my_gen/builder.dart"
    builder_factories: ["loggerBuilder"]
    build_extensions: { ".dart": [".g.dart"] } # fix 5: обязательное поле
    auto_apply: dependents
    build_to: source
    applies_builders: ["source_gen|combining_builder"]
```

**Пять багов:** 1) `GeneratorForAnnotation<dynamic>` — не находит аннотации; нужен конкретный тип аннотации. 2) `(element as ClassElement)` без проверки — выдаст `CastError` для функций/полей с этой аннотацией. 3) `instance` в сгенерированном коде не определён — метод должен принимать `instance` как параметр или быть в extension. 4) `GeneratorForAnnotation` — это `Generator`, не `Builder`; для `build_runner` нужна обёртка `SharedPartBuilder` или `LibraryBuilder`. 5) `build_extensions` обязателен в `build.yaml` — без него `build_runner` не знает, какие файлы порождает builder.
