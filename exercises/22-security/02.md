# Задание 02 (22.1): Валидация входных данных — 🐛 Найти баг

**Тема:** 22. Безопасность и защита кода | **Подтема:** 22.1 Валидация входных данных

---

В коде валидации и обработчике API-запросов есть уязвимости и баги. Найди их.

```dart
import 'dart:convert';
import 'dart:io';

// bug 1: приведение без проверки — ClassCastException при неожиданном типе
Map<String, dynamic> parseBody(String rawJson) {
  return jsonDecode(rawJson) as Map<String, dynamic>;
}

// bug 2: SQL-инъекция — имя пользователя подставляется в строку напрямую
String buildQuery(String username) {
  return "SELECT * FROM users WHERE name = '$username'";
}

// bug 3: нет валидации длины — DoS через огромные строки
bool isValidEmail(String email) {
  return RegExp(r'^[\w.+\-]+@[\w\-]+\.[a-zA-Z]{2,}$').hasMatch(email);
}

// bug 4: отражение пользовательских данных без экранирования
String renderHtml(String userName) {
  return '<h1>Welcome, $userName!</h1>';  // XSS-уязвимость
}

// bug 5: логирование чувствительных данных
void handleLogin(String username, String password) {
  print('Login attempt: username=$username, password=$password');
  // ...
}

// bug 6: небезопасное сравнение паролей (timing attack)
bool checkPassword(String input, String stored) {
  return input == stored;
}

// bug 7: возраст без upper bound — можно передать 999
int? parseAge(Object? value) {
  if (value is int && value > 0) return value;
  return null;
}

// bug 8: игнорирование ошибок валидации — сохраняем даже невалидные данные
void saveUser(Map<String, dynamic> data) {
  final email = data['email'] as String? ?? '';
  final age = data['age'] as int? ?? 0;

  isValidEmail(email);  // результат игнорируется
  parseAge(age);        // результат игнорируется

  // Сохраняем данные без проверки результатов валидации
  print('Saving: name=${data['name']}, email=$email, age=$age');
}

// bug 9: путь к файлу без санитизации (path traversal)
String readUserFile(String filename) {
  return File('uploads/$filename').readAsStringSync();
}

// bug 10: redirect URL без валидации (open redirect)
String buildRedirectUrl(String returnTo) {
  return returnTo;  // злоумышленник вернёт 'https://evil.com'
}
```

```dart
import 'dart:convert';
import 'dart:io';

// fix 1: безопасный парсинг JSON с явной обработкой ошибок типа
Map<String, dynamic> parseBody(String rawJson) {
  try {
    final decoded = jsonDecode(rawJson);
    if (decoded is! Map<String, dynamic>) {
      throw FormatException('Expected JSON object, got ${decoded.runtimeType}');
    }
    return decoded;
  } on FormatException {
    rethrow;  // пробрасываем с понятным сообщением
  }
}

// fix 2: никогда не строим SQL через конкатенацию
// Используем параметризованные запросы (placeholder-подход)
String buildQuery(String username) {
  // Реальный драйвер: db.query('SELECT * FROM users WHERE name = @name', {'name': username})
  // Здесь демонстрируем правильный шаблон
  return 'SELECT * FROM users WHERE name = ?';
  // username передаётся как параметр отдельно, не вставляется в строку
}

// fix 3: добавляем ограничение длины ДО regex (защита от ReDoS)
bool isValidEmail(String email) {
  if (email.length > 254) return false;  // RFC 5321 максимум
  return RegExp(r'^[\w.+\-]+@[\w\-]+\.[a-zA-Z]{2,}$').hasMatch(email);
}

// fix 4: HTML-экранирование пользовательских данных
String htmlEscape(String input) => input
    .replaceAll('&', '&amp;')
    .replaceAll('<', '&lt;')
    .replaceAll('>', '&gt;')
    .replaceAll('"', '&quot;')
    .replaceAll("'", '&#x27;');

String renderHtml(String userName) {
  // fix 4: экранируем перед вставкой в HTML
  return '<h1>Welcome, ${htmlEscape(userName)}!</h1>';
}

// fix 5: никогда не логируем пароли и другие чувствительные данные
void handleLogin(String username, String password) {
  // Логируем только имя пользователя, без пароля
  print('Login attempt: username=$username');
  // В production используем структурированное логирование с маскированием
}

// fix 6: константное время сравнения (защита от timing attack)
// Пакет crypto предоставляет secureCompareBytes
bool checkPassword(String input, String storedHash) {
  // Правильно: сравниваем хэши, а не plaintext пароли
  // В реальном коде: используем bcrypt/argon2 через пакет или dart:crypto HMAC
  if (input.length != storedHash.length) {
    // Намеренно продолжаем сравнение фиксированной длины
    // чтобы не раскрывать длину через timing
    return false;
  }
  var result = 0;
  for (var i = 0; i < input.length; i++) {
    result |= input.codeUnitAt(i) ^ storedHash.codeUnitAt(i);
  }
  return result == 0;
}

// fix 7: добавляем upper bound для возраста
int? parseAge(Object? value) {
  if (value is int && value >= 13 && value <= 120) return value;
  return null;
}

// fix 8: используем результаты валидации, отказываем при ошибке
void saveUser(Map<String, dynamic> data) {
  final email = data['email'] as String? ?? '';
  final age = data['age'];

  if (!isValidEmail(email)) {
    throw ArgumentError('Invalid email: $email');
  }
  final validAge = parseAge(age);
  if (validAge == null) {
    throw ArgumentError('Invalid age: $age (must be 13–120)');
  }

  print('Saving: name=${data['name']}, email=$email, age=$validAge');
}

// fix 9: санитизация пути — предотвращаем path traversal
String readUserFile(String filename) {
  // Разрешаем только простые имена файлов без разделителей пути
  if (filename.contains('/') ||
      filename.contains('\\') ||
      filename.contains('..') ||
      filename.startsWith('.')) {
    throw ArgumentError('Invalid filename: $filename');
  }

  final file = File('uploads/$filename');
  // Дополнительная проверка: файл должен оставаться внутри uploads/
  final canonical = file.resolveSymbolicLinksSync();
  final uploadsDir = Directory('uploads').resolveSymbolicLinksSync();
  if (!canonical.startsWith(uploadsDir)) {
    throw ArgumentError('Access denied: path outside uploads directory');
  }

  return file.readAsStringSync();
}

// fix 10: проверяем, что URL относительный или принадлежит нашему домену
String buildRedirectUrl(String returnTo) {
  const allowedDomains = {'example.com', 'app.example.com'};

  try {
    final uri = Uri.parse(returnTo);
    // Относительный путь — безопасен
    if (!uri.hasScheme) {
      return returnTo.startsWith('/') ? returnTo : '/';
    }
    // Абсолютный URL — проверяем домен
    if ((uri.scheme == 'https') && allowedDomains.contains(uri.host)) {
      return returnTo;
    }
  } catch (_) {
    // Невалидный URI
  }
  // Неизвестный домен или невалидный URL — возвращаем безопасный дефолт
  return '/';
}
```

**10 уязвимостей:** 1) Небезопасный каст `jsonDecode(x) as Map` — `CastError` при передаче массива или примитива. 2) SQL-инъекция через конкатенацию строк — `'; DROP TABLE users; --` уничтожит таблицу. 3) Нет ограничения длины в `isValidEmail` — ReDoS-атака через специально сформированную строку. 4) XSS: пользовательские данные вставляются в HTML без экранирования. 5) Логирование пароля в открытом виде — будет виден в логах, мониторинге, SIEM. 6) Timing attack: сравнение строк через `==` завершается раньше при первом несовпадении, расходимость времени раскрывает длину/префикс. 7) Нет `max` для возраста — `age: 999` проходит валидацию. 8) Результаты валидации игнорируются — невалидные данные сохраняются в БД. 9) Path traversal: `filename = '../../../etc/passwd"` читает системные файлы. 10) Open redirect: `returnTo = 'https://evil.com'` перенаправляет пользователя на вредоносный сайт.
