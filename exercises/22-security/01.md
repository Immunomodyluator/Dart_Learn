# Задание 01 (22.1): Валидация входных данных — 🖊 Написать код

**Тема:** 22. Безопасность и защита кода | **Подтема:** 22.1 Валидация входных данных

---

Реализуй слой валидации входных данных для REST API на Shelf.

**Требования:**

1. **`Validator<T>`** — строго типизированный класс-результат валидации с двумя состояниями: `Valid<T>(value)` и `Invalid<T>(errors)`.

2. **`StringValidator`** — строитель (builder) с методами-ограничениями, который возвращает `Validator<String>`:
   - `required()` — не пустая строка
   - `minLength(n)` / `maxLength(n)` — длина
   - `matches(RegExp)` — соответствие регулярному выражению
   - `email()` — валидный email
   - `noHtml()` — XSS-защита: запрещает `<`, `>`, `&`, `"` в сырых символах

3. **`IntValidator`** — аналогично для `int`: `min(n)`, `max(n)`, `positive()`.

4. **`validateBody(Map<String, dynamic> json, Map<String, ValidatorBuilder> schema)`** — функция, применяющая схему к JSON-объекту и возвращающая либо `Map<String, dynamic>` (проверенные данные), либо выбрасывающая `ValidationException` с полным списком ошибок по полям.

5. Пример использования: валидация тела запроса `POST /users` (поля `name`, `email`, `age`).

```dart
import 'dart:convert';

// ── Результат валидации ───────────────────────────────────────────

sealed class Validator<T> {
  const Validator();

  bool get isValid => this is Valid<T>;

  T getOrThrow() => switch (this) {
        Valid(:final value) => value,
        Invalid(:final errors) =>
          throw ValidationException({'_': errors}),
      };
}

final class Valid<T> extends Validator<T> {
  final T value;
  const Valid(this.value);
}

final class Invalid<T> extends Validator<T> {
  final List<String> errors;
  const Invalid(this.errors);
}

class ValidationException implements Exception {
  final Map<String, List<String>> fieldErrors;
  const ValidationException(this.fieldErrors);

  @override
  String toString() => 'Validation failed: $fieldErrors';

  String toJson() => jsonEncode({
        'error': 'VALIDATION_FAILED',
        'fields': fieldErrors,
      });
}

// ── Строитель валидатора строк ────────────────────────────────────

class StringValidatorBuilder {
  final List<String Function(String)> _rules = [];

  StringValidatorBuilder required() {
    _rules.add((v) => v.trim().isEmpty ? 'Field is required' : '');
    return this;
  }

  StringValidatorBuilder minLength(int n) {
    _rules.add((v) => v.length < n ? 'Minimum length is $n' : '');
    return this;
  }

  StringValidatorBuilder maxLength(int n) {
    _rules.add((v) => v.length > n ? 'Maximum length is $n' : '');
    return this;
  }

  StringValidatorBuilder matches(RegExp pattern, [String? message]) {
    _rules.add((v) =>
        !pattern.hasMatch(v) ? (message ?? 'Invalid format') : '');
    return this;
  }

  StringValidatorBuilder email() {
    return matches(
      RegExp(r'^[\w.+-]+@[\w-]+\.[a-z]{2,}$', caseSensitive: false),
      'Invalid email address',
    );
  }

  // XSS-защита: отклоняем необработанные HTML-символы
  StringValidatorBuilder noHtml() {
    _rules.add((v) {
      if (v.contains(RegExp(r'[<>&"]'))) {
        return 'HTML characters are not allowed';
      }
      return '';
    });
    return this;
  }

  Validator<String> validate(Object? raw) {
    if (raw == null || raw is! String) {
      return Invalid(['Expected a string']);
    }
    final errors =
        _rules.map((r) => r(raw)).where((e) => e.isNotEmpty).toList();
    return errors.isEmpty ? Valid(raw) : Invalid(errors);
  }
}

// ── Строитель валидатора чисел ────────────────────────────────────

class IntValidatorBuilder {
  final List<String Function(int)> _rules = [];

  IntValidatorBuilder min(int n) {
    _rules.add((v) => v < n ? 'Minimum value is $n' : '');
    return this;
  }

  IntValidatorBuilder max(int n) {
    _rules.add((v) => v > n ? 'Maximum value is $n' : '');
    return this;
  }

  IntValidatorBuilder positive() => min(1);

  Validator<int> validate(Object? raw) {
    final int value;
    if (raw is int) {
      value = raw;
    } else if (raw is String) {
      final parsed = int.tryParse(raw);
      if (parsed == null) return Invalid(['Must be an integer']);
      value = parsed;
    } else {
      return Invalid(['Expected an integer']);
    }
    final errors =
        _rules.map((r) => r(value)).where((e) => e.isNotEmpty).toList();
    return errors.isEmpty ? Valid(value) : Invalid(errors);
  }
}

// ── Схемная валидация JSON-тела ───────────────────────────────────

// Абстракция для полиморфной схемы
abstract class ValidatorBuilder<T> {
  Validator<T> validate(Object? raw);
}

extension on StringValidatorBuilder {
  ValidatorBuilder<String> asBuilder() => _StringBuilderAdapter(this);
}
extension on IntValidatorBuilder {
  ValidatorBuilder<int> asBuilder() => _IntBuilderAdapter(this);
}

class _StringBuilderAdapter implements ValidatorBuilder<String> {
  final StringValidatorBuilder _b;
  _StringBuilderAdapter(this._b);
  @override
  Validator<String> validate(Object? raw) => _b.validate(raw);
}

class _IntBuilderAdapter implements ValidatorBuilder<int> {
  final IntValidatorBuilder _b;
  _IntBuilderAdapter(this._b);
  @override
  Validator<int> validate(Object? raw) => _b.validate(raw);
}

Map<String, dynamic> validateBody(
  Map<String, dynamic> json,
  Map<String, ValidatorBuilder> schema,
) {
  final errors = <String, List<String>>{};
  final result = <String, dynamic>{};

  for (final entry in schema.entries) {
    final field = entry.key;
    final builder = entry.value;
    final validation = builder.validate(json[field]);

    switch (validation) {
      case Valid(:final value):
        result[field] = value;
      case Invalid(:final errors as List<String>):
        errors[field] = errors;
    }
  }

  if (errors.isNotEmpty) {
    throw ValidationException(errors);
  }
  return result;
}

// ── Пример: POST /users ───────────────────────────────────────────

void main() {
  // Схема валидации для POST /users
  final schema = {
    'name': StringValidatorBuilder()
        .required()
        .minLength(2)
        .maxLength(50)
        .noHtml()
        .asBuilder(),
    'email': StringValidatorBuilder()
        .required()
        .email()
        .maxLength(255)
        .asBuilder(),
    'age': IntValidatorBuilder().min(13).max(120).asBuilder(),
  };

  // Тест 1: корректные данные
  try {
    final valid = validateBody(
      {'name': 'Ivan Petrov', 'email': 'ivan@example.com', 'age': 25},
      schema,
    );
    print('Valid: $valid');
  } on ValidationException catch (e) {
    print('Error: $e');
  }

  // Тест 2: множественные ошибки
  try {
    validateBody(
      {'name': '<script>', 'email': 'not-an-email', 'age': 5},
      schema,
    );
  } on ValidationException catch (e) {
    // Выводим как JSON для REST-ответа
    print(e.toJson());
    // {"error":"VALIDATION_FAILED","fields":{"name":["HTML characters are not allowed"],"email":["Invalid email address"],"age":["Minimum value is 13"]}}
  }
}
```

`sealed class Validator<T>` гарантирует исчерпывающий `switch` — компилятор не даст забыть обработать случай `Invalid`. `noHtml()` предотвращает XSS на уровне бизнес-логики, до того как данные попадут в БД или шаблон. `validateBody` агрегирует все ошибки по полям и выбрасывает одно исключение с полной картиной нарушений — клиент получает все ошибки за один запрос.
