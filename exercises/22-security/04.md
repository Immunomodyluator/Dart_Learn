import 'dart:io';

// bug 1: секрет в исходном коде — попадёт в git-историю
const jwtSecret = 'my-super-secret-key-12345';
const dbPassword = 'admin123';

class Config {
  // bug 2: секреты в публичных полях — доступны отовсюду без ограничений
  final String databaseUrl;
  final String jwtSecret;
  final String apiKey;

  Config(this.databaseUrl, this.jwtSecret, this.apiKey);

  // bug 3: toString раскрывает секреты — риск логирования
  @override
  String toString() =>
      'Config(db=$databaseUrl, jwt=$jwtSecret, api=$apiKey)';

  // bug 4: секреты сериализуются в JSON
  Map<String, dynamic> toJson() => {
        'databaseUrl': databaseUrl,
        'jwtSecret': jwtSecret,
        'apiKey': apiKey,
      };

  // bug 5: нет валидации — приложение запускается с пустыми секретами
  factory Config.fromEnv() {
    return Config(
      Platform.environment['DATABASE_URL'] ?? '',
      Platform.environment['JWT_SECRET'] ?? '',
      Platform.environment['API_KEY'] ?? '',
    );
  }
}

void runServer(List<String> args) {
  final secret = args.isNotEmpty ? args[0] : '';

```dart
import 'dart:io';

// fix 1: НИКОГДА не хардкодим секреты в коде — только переменные окружения
// const jwtSecret = '...' — УДАЛИТЬ полностью
// Если секрет уже попал в git: git filter-branch / BFG Repo Cleaner + ротация

// fix 2: секреты за приватным полем + явный reveal()
final class SecretValue {
  final String _value;
  const SecretValue(this._value);
  String reveal() => _value;
  @override
  String toString() => 'SecretValue(***)';
}

class Config {
  final String databaseUrl;
  // fix 2: секреты через SecretValue — не раскрываются случайно
  final SecretValue jwtSecret;
  final SecretValue apiKey;
  final int port;

  Config({
    required this.databaseUrl,
    required this.jwtSecret,
    required this.apiKey,
    required this.port,
  });

  // fix 3: маскируем секреты и пароль в URL
  @override
  String toString() {
    String maskUrl(String url) {
      try {
        final uri = Uri.parse(url);
        if (uri.userInfo.isEmpty) return url;
        final user = uri.userInfo.split(':')[0];
        return url.replaceFirst(uri.userInfo, '$user:***');
      } catch (_) {
        return '***';
      }
    }
    return 'Config(db=${maskUrl(databaseUrl)}, jwt=$jwtSecret, api=$apiKey)';
  }

  // fix 4: не сериализуем секреты в JSON
  Map<String, dynamic> toJson() => {
        'databaseUrl': _maskUrl(databaseUrl),
        // jwtSecret и apiKey не включаем в JSON вообще
        'port': port,
      };

  static String _maskUrl(String url) {
    try {
      final uri = Uri.parse(url);
      if (uri.userInfo.isEmpty) return url;
      return url.replaceFirst(uri.userInfo, '${uri.userInfo.split(':')[0]}:***');
    } catch (_) {
      return '***';
    }
  }

  // fix 5: обязательная валидация — fail fast при отсутствии секретов
  factory Config.fromEnv() {
    String require(String key) {
      final v = Platform.environment[key];
      if (v == null || v.isEmpty) {
        throw StateError('Required env variable "$key" is not set');
      }
      return v;
    }

    final portStr = Platform.environment['PORT'] ?? '8080';
    final port = int.tryParse(portStr);
    if (port == null || port < 1 || port > 65535) {
      throw StateError('Invalid PORT: "$portStr"');
    }

    return Config(
      databaseUrl: require('DATABASE_URL'),
      jwtSecret: SecretValue(require('JWT_SECRET')),
      apiKey: SecretValue(require('API_KEY')),
      port: port,
    );
  }
}

// fix 6: секреты НИКОГДА не передаём в аргументах командной строки
// Используем переменные окружения или файлы с ограниченными правами
void runServer() {
  // secret берётся из Config.fromEnv(), не из аргументов
  print('Starting server...');
}

// fix 7: API-ключ не логируем — используем его хэш или только первые 8 символов
void logRequest(String path, String apiKey) {
  // Логируем только безопасный идентификатор (не сам ключ)
  final keyId = apiKey.length > 8 ? '${apiKey.substring(0, 8)}...' : '***';
  File('app.log').writeAsStringSync(
    '${DateTime.now()} GET $path apiKey=$keyId\n',
    mode: FileMode.append,
  );
}

// fix 8: секреты хранятся в переменных окружения, .env файле (в .gitignore)
// или в менеджерах секретов (HashiCorp Vault, AWS Secrets Manager и т.д.)
// secrets.dart никогда не создаём

// fix 9: константное время сравнения для API-ключей
bool isValidApiKey(String provided, String stored) {
  if (provided.length != stored.length) {
    // Симулируем фиксированное время, не раскрываем длину
    // Полноценная защита — через HMAC-сравнение
    return false;
  }
  var result = 0;
  for (var i = 0; i < provided.length; i++) {
    result |= provided.codeUnitAt(i) ^ stored.codeUnitAt(i);
  }
  return result == 0;
}

// fix 10: гарантируем удаление временного файла через try/finally
Future<void> deployWithSecret(String secret) async {
  final tmpFile = File('/tmp/deploy_secret_${DateTime.now().millisecondsSinceEpoch}.txt');
  try {
    // Устанавливаем ограниченные права до записи секрета
    await tmpFile.writeAsString(secret);
    await Process.run('chmod', ['600', tmpFile.path]);
    await Process.run('deploy_script.sh', [tmpFile.path]);
  } finally {
    // fix 10: обязательное удаление в finally
    if (await tmpFile.exists()) {
      await tmpFile.delete();
    }
  }
  // Предпочтительнее: передавать секрет через stdin или переменную окружения
}
```

**10 проблем безопасности:** 1) Хардкод секретов в исходном коде — попадут в git, доступны всем с доступом к репозиторию. 2) Публичные строковые поля — секреты доступны из любого места кода. 3) `toString()` раскрывает секреты — случайный `print(config)` сольёт JWT-ключ в логи. 4) `toJson()` включает все секреты — если объект сериализуется в ответ API, клиент получит секреты. 5) `?? ''` при отсутствии секрета — приложение запускается с пустым JWT-ключом, все токены невалидны или небезопасны. 6) Секрет в аргументах CLI — виден в `ps aux`, в истории shell, в CI-логах. 7) Полный API-ключ в лог-файле — логи часто менее защищены, чем переменные окружения. 8) `secrets.dart` в git — классическая ошибка, обнаруживается автоматическими сканерами. 9) `==` для API-ключей — timing attack раскрывает совпадающий префикс. 10) Временный файл с секретом не удаляется — остаётся доступен другим процессам.
