# Задание 03 (22.2): Управление секретами и конфигурацией — 🖊 Написать код

**Тема:** 22. Безопасность и защита кода | **Подтема:** 22.2 Управление секретами и конфигурацией

---

Реализуй безопасную систему управления конфигурацией для Dart-приложения.

**Требования:**

1. **`AppConfig`** — immutable класс конфигурации с полями (databaseUrl, redisUrl, jwtSecret, apiKey, port), загружаемый из переменных окружения. Обязательные поля выбрасывают исключение при отсутствии.

2. **`SecretValue`** — обёртка над секретом, которая:
   - Прячет значение в `toString()` → `"SecretValue(***)"` (защита от случайного логирования)
   - Предоставляет `reveal()` только явным вызовом
   - Предотвращает сериализацию через `toJson()` → `"***"`

3. **`ConfigValidator`** — проверяет конфигурацию при старте (формат URL, минимальная длина JWT secret, валидный порт).

4. Пример `main()`: загрузка конфигурации, валидация, вывод безопасного `toString()` без секретов.

```dart
// Ожидаемое поведение:
final config = AppConfig.fromEnvironment();
print(config.jwtSecret);          // SecretValue(***)
print(config.jwtSecret.reveal()); // "actual-secret-value"
print(jsonEncode(config));         // jwtSecret не попадает в JSON или маскируется
```

```dart
import 'dart:convert';
import 'dart:io';

// ── Обёртка над секретом ──────────────────────────────────────────

final class SecretValue {
  final String _value;

  const SecretValue(this._value);

  /// Единственный способ получить значение — явно вызвать reveal()
  String reveal() => _value;

  /// toString никогда не раскрывает значение
  @override
  String toString() => 'SecretValue(***)';

  /// Для JSON-сериализации возвращаем маску
  String toJson() => '***';

  /// Сравнение без раскрытия значений
  @override
  bool operator ==(Object other) =>
      other is SecretValue && _value == other._value;

  @override
  int get hashCode => _value.hashCode;
}

// ── Исключения конфигурации ───────────────────────────────────────

class ConfigException implements Exception {
  final String message;
  ConfigException(this.message);
  @override
  String toString() => 'ConfigException: $message';
}

// ── Иммутабельная конфигурация ────────────────────────────────────

final class AppConfig {
  final String databaseUrl;
  final String redisUrl;
  final SecretValue jwtSecret;
  final SecretValue apiKey;
  final int port;
  final String appEnv;

  const AppConfig({
    required this.databaseUrl,
    required this.redisUrl,
    required this.jwtSecret,
    required this.apiKey,
    required this.port,
    required this.appEnv,
  });

  /// Загрузка из переменных окружения
  factory AppConfig.fromEnvironment() {
    final env = Platform.environment;

    String require(String key) {
      final value = env[key];
      if (value == null || value.isEmpty) {
        throw ConfigException('Required environment variable "$key" is not set');
      }
      return value;
    }

    String optional(String key, String defaultValue) =>
        env[key]?.isNotEmpty == true ? env[key]! : defaultValue;

    // Парсинг порта с проверкой
    final portStr = optional('PORT', '8080');
    final port = int.tryParse(portStr);
    if (port == null) {
      throw ConfigException('PORT must be a number, got: "$portStr"');
    }

    return AppConfig(
      databaseUrl: require('DATABASE_URL'),
      redisUrl: optional('REDIS_URL', 'redis://localhost:6379'),
      jwtSecret: SecretValue(require('JWT_SECRET')),
      apiKey: SecretValue(require('API_KEY')),
      port: port,
      appEnv: optional('APP_ENV', 'development'),
    );
  }

  // toString не раскрывает секреты
  @override
  String toString() => 'AppConfig('
      'env=$appEnv, '
      'port=$port, '
      'databaseUrl=${_maskUrl(databaseUrl)}, '
      'redisUrl=$redisUrl, '
      'jwtSecret=$jwtSecret, '
      'apiKey=$apiKey'
      ')';

  // Маскируем пароль в URL: postgres://user:***@host/db
  static String _maskUrl(String url) {
    try {
      final uri = Uri.parse(url);
      if (uri.userInfo.isEmpty) return url;
      return url.replaceFirst(uri.userInfo, '${uri.userInfo.split(':')[0]}:***');
    } catch (_) {
      return '***';
    }
  }
}

// ── Валидатор конфигурации ────────────────────────────────────────

class ConfigValidator {
  static void validate(AppConfig config) {
    final errors = <String>[];

    // Проверяем формат DATABASE_URL
    try {
      final uri = Uri.parse(config.databaseUrl);
      if (!{'postgres', 'postgresql', 'mysql'}.contains(uri.scheme)) {
        errors.add('DATABASE_URL has unsupported scheme: ${uri.scheme}');
      }
    } catch (_) {
      errors.add('DATABASE_URL is not a valid URL');
    }

    // Минимальная длина JWT secret
    if (config.jwtSecret.reveal().length < 32) {
      errors.add('JWT_SECRET must be at least 32 characters long');
    }

    // Валидный диапазон порта
    if (config.port < 1 || config.port > 65535) {
      errors.add('PORT must be between 1 and 65535');
    }

    // Предупреждение в production
    if (config.appEnv == 'production' && config.port < 1024) {
      errors.add('Running on privileged port $port in production requires root — not recommended');
    }

    if (errors.isNotEmpty) {
      throw ConfigException(
        'Configuration validation failed:\n${errors.map((e) => '  - $e').join('\n')}',
      );
    }
  }
}

// ── main ─────────────────────────────────────────────────────────

void main() {
  // Симулируем переменные окружения (в реальном приложении — через OS/CI)
  // ignore: invalid_use_of_visible_for_testing_member
  // Для демонстрации задаём через Map
  final fakeEnv = {
    'DATABASE_URL': 'postgres://appuser:s3cr3t@localhost:5432/mydb',
    'JWT_SECRET': 'super-secret-jwt-key-must-be-at-least-32-chars!!',
    'API_KEY': 'key-abc-123-xyz',
    'PORT': '8080',
    'APP_ENV': 'development',
  };

  // В production используем Platform.environment напрямую
  // Здесь демонстрируем ручную сборку
  final config = AppConfig(
    databaseUrl: fakeEnv['DATABASE_URL']!,
    redisUrl: 'redis://localhost:6379',
    jwtSecret: SecretValue(fakeEnv['JWT_SECRET']!),
    apiKey: SecretValue(fakeEnv['API_KEY']!),
    port: int.parse(fakeEnv['PORT']!),
    appEnv: fakeEnv['APP_ENV']!,
  );

  // Валидируем конфигурацию при старте — fail fast
  try {
    ConfigValidator.validate(config);
    print('Configuration OK');
  } on ConfigException catch (e) {
    stderr.writeln(e);
    exit(1);
  }

  // Безопасный вывод — секреты замаскированы
  print(config);
  // AppConfig(env=development, port=8080,
  //   databaseUrl=postgres://appuser:***@localhost:5432/mydb,
  //   jwtSecret=SecretValue(***), apiKey=SecretValue(***))

  // Явное раскрытие секрета только там, где необходимо
  final jwt = config.jwtSecret.reveal();
  // Используем jwt для подписи токенов, не логируем
  print('JWT secret length: ${jwt.length} chars');
  print('SecretValue.toString(): ${config.jwtSecret}');  // SecretValue(***)
}
```

`SecretValue` защищает от случайного раскрытия секретов через `print`, логирование или `toString()`. URL-маскирование в `_maskUrl()` позволяет логировать строку подключения без пароля. Принцип fail-fast в `ConfigValidator` гарантирует, что приложение не запустится с неверной конфигурацией — ошибка обнаруживается немедленно, а не в runtime при первом использовании.
