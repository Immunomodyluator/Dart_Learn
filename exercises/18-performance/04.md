# Задание 04 (18.2): Оптимизация аллокаций — 🐛 Найти баг

**Тема:** 18. Производительность и профилирование | **Подтема:** 18.2 Оптимизация аллокаций

---

В коде с попытками оптимизации аллокаций есть ошибки или антипаттерны. Найди их.

```dart
import 'dart:typed_data';

// bug 1 — конкатенация строк в цикле (O(n²) аллокации)
String joinWithComma(List<String> items) {
  var result = '';
  for (var i = 0; i < items.length; i++) {
    result += items[i];           // new String на каждой итерации
    if (i < items.length - 1) result += ', ';
  }
  return result;
}

// bug 2 — List.from создаёт копию, хотя нам нужна только итерация
bool containsNegative(List<int> values) {
  final copy = List.from(values); // лишняя аллокация
  return copy.any((v) => v < 0);
}

// bug 3 — не используем TypedData для числовых массивов
List<double> computeDistances(List<(double, double)> points) {
  final result = <double>[];       // boxed — каждое double оборачивается
  for (final (x, y) in points) {
    result.add(x * x + y * y);
  }
  return result;
}

class Vector2 {
  double x, y;
  Vector2(this.x, this.y);

  // bug 4 — оператор + создаёт новый объект на каждое сложение
  //         в цикле physics-обновлений это тысячи аллокаций
  Vector2 operator +(Vector2 other) => Vector2(x + other.x, y + other.y);

  // bug 5 — toString вызывается при каждом логе, даже если лог отключён
  @override
  String toString() => 'Vector2($x, $y)';
}

void updatePhysics(List<Vector2> positions, List<Vector2> velocities) {
  for (var i = 0; i < positions.length; i++) {
    positions[i] = positions[i] + velocities[i]; // bug 4: new Vector2 каждый кадр
    print('Updated: ${positions[i]}');           // bug 5: toString даже в prod
  }
}

// bug 6 — кешируем результат, но каждый раз создаём новый список-обёртку
final Map<String, List<int>> _cache = {};

List<int> getCachedData(String key, List<int> Function() compute) {
  _cache.putIfAbsent(key, compute);
  return List.from(_cache[key]!); // возвращаем копию — смысл кеша теряется
}
```

```dart
import 'dart:typed_data';

// fix 1: StringBuffer — одна аллокация в конце
String joinWithComma(List<String> items) {
  return items.join(', '); // встроенный метод тоже использует StringBuffer внутри
  // Или явно:
  // final sb = StringBuffer();
  // for (var i = 0; i < items.length; i++) {
  //   if (i > 0) sb.write(', ');
  //   sb.write(items[i]);
  // }
  // return sb.toString();
}

// fix 2: убираем копию — итерируем оригинал напрямую
bool containsNegative(List<int> values) {
  return values.any((v) => v < 0); // no allocation
}

// fix 3: Float64List — unboxed хранение double (половина памяти + cache-friendly)
Float64List computeDistances(List<(double, double)> points) {
  final result = Float64List(points.length); // pre-allocated, unboxed
  for (var i = 0; i < points.length; i++) {
    final (x, y) = points[i];
    result[i] = x * x + y * y;
  }
  return result;
}

class Vector2 {
  double x, y;
  Vector2(this.x, this.y);

  // fix 4: мутирующая версия для горячих путей — нет аллокации
  void addInPlace(Vector2 other) {
    x += other.x;
    y += other.y;
  }

  // Создающая версия — только там, где нужна иммутабельность
  Vector2 operator +(Vector2 other) => Vector2(x + other.x, y + other.y);

  @override
  String toString() => 'Vector2($x, $y)';
}

// fix 5: управляем логированием флагом, чтобы не вызывать toString впустую.
// В Dart нет макросов, но можно использовать функцию-ленивый провайдер.
const _debugPhysics = false; // или bool.fromEnvironment('debug.physics')

void updatePhysics(List<Vector2> positions, List<Vector2> velocities) {
  for (var i = 0; i < positions.length; i++) {
    // fix 4: мутируем вместо создания
    positions[i].addInPlace(velocities[i]);

    // fix 5: toString не вызывается, если константа false (dead code elimination)
    if (_debugPhysics) {
      print('Updated: ${positions[i]}');
    }
  }
}

// fix 6: возвращаем немодифицируемый вид (UnmodifiableListView) — нет копии
import 'dart:collection';

final Map<String, List<int>> _cache = {};

List<int> getCachedData(String key, List<int> Function() compute) {
  final cached = _cache.putIfAbsent(key, compute);
  // Если нужна защита от внешних мутаций — UnmodifiableListView без копии
  return UnmodifiableListView(cached);
  // Если caller должен мутировать — явно документируем, что возвращается копия:
  // return List.of(cached);
}

void main() {
  print(joinWithComma(['a', 'b', 'c'])); // a, b, c

  print(containsNegative([1, -2, 3]));   // true

  final dists = computeDistances([(3.0, 4.0), (1.0, 1.0)]);
  print(dists); // [25.0, 2.0]

  final positions = [Vector2(0, 0), Vector2(1, 1)];
  final velocities = [Vector2(1, 0), Vector2(0, 1)];
  updatePhysics(positions, velocities);
  print('${positions[0].x}, ${positions[0].y}'); // 1.0, 0.0

  final data = getCachedData('key', () => [1, 2, 3]);
  print(data); // [1, 2, 3]
}
```

**Шесть ошибок:** 1) `result += item` в цикле — O(n²) аллокации; используй `StringBuffer.join`. 2) `List.from(values)` для итерации — бесполезная копия; итерируй оригинал. 3) `List<double>` в Dart хранит boxing-обёртки; `Float64List` хранит нативные double без boxing. 4) `operator +` создаёт новый `Vector2` при каждом обновлении физики; вместо него нужен `addInPlace`. 5) `${positions[i]}` вызывает `toString()` каждый кадр, даже в release; обернуть в флаг `_debugPhysics`. 6) `List.from(cached)` при каждом обращении к кешу — кеш бесполезен; `UnmodifiableListView` даёт read-only вид без копии.
