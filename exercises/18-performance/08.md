В коде есть утечки памяти и неправильное использование GC-механизмов. Найди их.

```dart
import 'dart:async';

// bug 1 — утечка Stream-подписки: subscribe без cancel
class EventLogger {
  final _log = <String>[];
  StreamSubscription? _sub;

  void startListening(Stream<String> events) {
    // bug 1: если startListening вызвать повторно — старая подписка никогда не отменяется
    _sub = events.listen((event) => _log.add(event));
  }

  void stopListening() {
    _sub?.cancel();
    _sub = null;
  }
}

// bug 2 — замыкание захватывает большой объект, мешая GC
List<int> Function() createCounter(List<int> hugeData) {
  var count = 0;
  // bug 2: hugeData захвачен замыканием — GC не может собрать список
  //         даже если вызывающий код больше не держит на него ссылку
  return () {
    count++;
    return hugeData; // зачем возвращать hugeData? — только count нужен
  };
}

// bug 3 — глобальный кеш без ограничения размера (memory leak)
final Map<String, dynamic> _globalCache = {};

void cacheResult(String key, dynamic value) {
  _globalCache[key] = value; // bug 3: кеш растёт без предела
}

// bug 4 — регистрируем finalizer на примитив (не объект)
void setupFinalizer() {
  final finalizer = Finalizer<int>((v) => print('finalized: $v'));
  final n = 42; // это int — примитив
  finalizer.attach(n, 0); // bug 4: примитивы нельзя использовать как токен объекта
                           // (точнее: int в Dart — объект, но JIT может их inline-ить)
}

// bug 5 — таймер не отменяется при dispose
class AnimationController {
  Timer? _timer;
  final List<void Function()> _listeners = [];

  void start(void Function() onTick) {
    _listeners.add(onTick); // bug 5a: listeners накапливаются
    _timer = Timer.periodic(Duration(milliseconds: 16), (_) {
      for (final l in _listeners) l();
    });
    // bug 5b: если start вызвать дважды — два Timer, оба выполняются
  }

  void dispose() {
    // bug 5c: _timer не отменяется, _listeners не очищаются
    print('disposed');
  }
}

// bug 6 — WeakReference используется для хранения важного объекта
class ServiceLocator {
  static final _services = <Type, WeakReference<Object>>{};

  static void register<T extends Object>(T service) {
    _services[T] = WeakReference(service); // bug 6: GC может собрать service
    // если никто больше не держит сильную ссылку
  }

  static T? get<T extends Object>() {
    return _services[T]?.target as T?;
  }
}

void main() {
  final logger = EventLogger();
  final ctrl = StreamController<String>();
  logger.startListening(ctrl.stream);
  logger.startListening(ctrl.stream); // дублирует подписку

  final counter = createCounter(List.generate(1000000, (i) => i));
  counter(); // hugeData удерживается в памяти

  ServiceLocator.register(Object()); // может быть сразу собрано GC
}
```

```dart
import 'dart:async';

// fix 1: отменяем предыдущую подписку перед созданием новой
class EventLogger {
  final _log = <String>[];
  StreamSubscription? _sub;

  Future<void> startListening(Stream<String> events) async {
    await _sub?.cancel(); // fix: отменяем старую подписку
    _sub = events.listen((event) => _log.add(event));
  }

  Future<void> stopListening() async {
    await _sub?.cancel();
    _sub = null;
  }

  List<String> get log => List.unmodifiable(_log);
}

// fix 2: не захватываем hugeData в замыкании — передаём только нужные данные
int Function() createCounter(List<int> hugeData) {
  // Извлекаем нужное значение сразу — не захватываем весь список
  final length = hugeData.length; // только int
  var count = 0;
  return () {
    count++;
    return count; // fix: возвращаем count, а не hugeData
  };
  // hugeData теперь не захвачена — GC может собрать её
}
// Если нужно вернуть данные — передать их напрямую, без замыкания

// fix 3: LRU-кеш с ограничением размера
class BoundedCache<K, V> {
  final int maxSize;
  final _data = <K, V>{};
  final _order = <K>[];

  BoundedCache(this.maxSize);

  void put(K key, V value) {
    if (_data.containsKey(key)) {
      _order.remove(key);
    } else if (_data.length >= maxSize) {
      final oldest = _order.removeAt(0);
      _data.remove(oldest);
    }
    _order.add(key);
    _data[key] = value;
  }

  V? get(K key) => _data[key];
  int get size => _data.length;
}

final _globalCache = BoundedCache<String, dynamic>(500); // fix: ограниченный размер

void cacheResult(String key, dynamic value) {
  _globalCache.put(key, value); // fix: автоматически вытесняет старые записи
}

// fix 4: Finalizer требует объект (не будет проблем в Dart, т.к. int — Object,
// но нет смысла финализировать примитивы — используй объект-токен)
class FinalizedToken {
  final int value;
  FinalizedToken(this.value);
}

void setupFinalizer() {
  final finalizer = Finalizer<String>((token) => print('finalized: $token'));
  final token = FinalizedToken(42);
  // fix: используем объект, а не примитив
  finalizer.attach(token, 'token_42', detach: token);
}

// fix 5: правильный AnimationController
class AnimationController {
  Timer? _timer;
  void Function()? _currentListener;
  bool _disposed = false;

  void start(void Function() onTick) {
    if (_disposed) throw StateError('AnimationController is disposed');
    // fix 5b: отменяем предыдущий таймер
    _timer?.cancel();
    _currentListener = onTick; // fix 5a: только один listener
    _timer = Timer.periodic(const Duration(milliseconds: 16), (_) {
      _currentListener?.call();
    });
  }

  void dispose() {
    _disposed = true;
    _timer?.cancel();          // fix 5c: отменяем таймер
    _timer = null;
    _currentListener = null;   // fix 5c: очищаем ссылки
    print('disposed');
  }
}

// fix 6: ServiceLocator должен держать сильные ссылки
class ServiceLocator {
  // fix: сильные ссылки — GC не сможет собрать зарегистрированные сервисы
  static final _services = <Type, Object>{};

  static void register<T extends Object>(T service) {
    _services[T] = service; // fix: не WeakReference
  }

  static T? get<T extends Object>() {
    return _services[T] as T?;
  }

  static void unregister<T>() {
    _services.remove(T);
  }
}

Future<void> main() async {
  final logger = EventLogger();
  final ctrl = StreamController<String>();
  await logger.startListening(ctrl.stream);
  await logger.startListening(ctrl.stream); // корректно: первая подписка отменена

  ctrl.add('hello');
  await Future.delayed(Duration.zero);
  print('Log: ${logger.log}'); // не дублируется

  final counter = createCounter(List.generate(100, (i) => i));
  print('Count: ${counter()}'); // 1

  final anim = AnimationController();
  anim.start(() => null);
  anim.start(() => null); // корректно: старый timer отменён
  anim.dispose();

  ServiceLocator.register(Uri.parse('scheme://host'));
  final uri = ServiceLocator.get<Uri>();
  print('Service: $uri'); // scheme://host — не null

  await ctrl.close();
  await logger.stopListening();
}
```

**Шесть ошибок:** 1) Повторный `startListening` без отмены старой подписки создаёт дублированные обработчики и удерживает объекты в памяти. 2) Замыкание захватывает `hugeData` — список не собирается GC пока активна функция-замыкание. 3) Глобальная Map без ограничения — бесконечно растёт, classic memory leak. 4) Финализация примитивов бессмысленна; Dart обработает это, но намерение некорректно. 5) Три бага AnimationController: накопление listeners, дублирование Timer, пустой `dispose()`. 6) `WeakReference` в ServiceLocator означает, что сервис может пропасть сразу после регистрации — ServiceLocator должен держать сильные ссылки.
