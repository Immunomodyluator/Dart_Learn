В коде измерения производительности есть ошибки, приводящие к неверным результатам или падению. Найди их.

```dart
import 'dart:developer';

class PerfTracker {
  final _stopwatch = Stopwatch();
  final List<Duration> _samples = [];

  // bug 1 — измеряем не то время: включаем прогрев JIT в первый замер
  void measureAll(List<Function()> workloads) {
    for (final work in workloads) {
      _stopwatch.reset();    // не запускаем перед замером
      work();
      _stopwatch.stop();
      _samples.add(_stopwatch.elapsed); // bug 1: stopwatch никогда не startSync
    }
  }

  Duration get average {
    // bug 2 — делим на 0, если список пустой
    return _samples.reduce((a, b) => a + b) ~/ _samples.length;
  }
}

// bug 3 — Timeline блоки не закрыты при исключении
void riskyOperation() {
  Timeline.startSync('risky');
  final list = <int>[];
  list[0]; // RangeError — Timeline блок "risky" останется открытым навсегда
  Timeline.finishSync();
}

// bug 4 — log с неверным уровнем (Level.FINE невалиден, нужен int)
void verboseLog(String msg) {
  log(msg, name: 'myApp', level: 'DEBUG');  // level должен быть int, не String
}

// bug 5 — создаём Stopwatch внутри замера — накладные расходы на создание объекта
//         и не сбрасываем между запусками
Duration benchmarkFunction(Function() fn, {int runs = 10}) {
  var total = Duration.zero;
  for (var i = 0; i < runs; i++) {
    final sw = Stopwatch()..start(); // создаём новый объект each iteration
    fn();
    total += sw.elapsed; // bug 5a: не вызываем sw.stop() перед чтением elapsed
  }
  return total ~/ runs;
}

void main() {
  final tracker = PerfTracker();
  tracker.measureAll([
    () => List.generate(1000, (i) => i * i),
    () => List.filled(1000, 0).fold(0, (a, b) => a + b),
  ]);

  print('Average: ${tracker.average}'); // упадёт

  riskyOperation(); // выбросит исключение, Timeline блок не закрыт

  verboseLog('test');

  final elapsed = benchmarkFunction(() => 'hello' * 1000);
  print('Function avg: $elapsed');
}
```

```dart
import 'dart:developer';

class PerfTracker {
  final _stopwatch = Stopwatch();
  final List<Duration> _samples = [];

  // fix 1: reset → start → work → stop
  void measureAll(List<Function()> workloads) {
    // Прогрев — первый запуск не считаем (JIT)
    if (workloads.isNotEmpty) workloads.first();

    for (final work in workloads) {
      _stopwatch
        ..reset()
        ..start(); // fix: запускаем перед замером
      work();
      _stopwatch.stop();
      _samples.add(_stopwatch.elapsed);
    }
  }

  Duration get average {
    // fix 2: защита от пустого списка
    if (_samples.isEmpty) return Duration.zero;
    final total = _samples.reduce((a, b) => a + b);
    return total ~/ _samples.length;
  }
}

// fix 3: Timeline блок должен закрываться даже при исключении
void riskyOperation() {
  Timeline.startSync('risky');
  try {
    final list = <int>[];
    list[0]; // выбросит RangeError
    // ... логика
  } finally {
    Timeline.finishSync(); // fix: всегда закрываем в finally
  }
}

// fix 4: level — это int, а не String (используем Level из logging или числовые константы)
void verboseLog(String msg) {
  // Уровни из пакета logging: FINEST=300, FINER=400, FINE=500,
  // CONFIG=700, INFO=800, WARNING=900, SEVERE=1000, SHOUT=1200
  log(msg, name: 'myApp', level: 500); // fix: int (FINE)
}

// fix 5: переиспользуем Stopwatch, вызываем stop() перед чтением elapsed
Duration benchmarkFunction(Function() fn, {int runs = 10}) {
  assert(runs > 0);
  final sw = Stopwatch(); // создаём один раз
  var total = Duration.zero;

  // Прогрев
  fn();

  for (var i = 0; i < runs; i++) {
    sw.reset();
    sw.start();
    fn();
    sw.stop(); // fix 5a: останавливаем перед чтением
    total += sw.elapsed;
  }
  return total ~/ runs;
}

void main() {
  final tracker = PerfTracker();
  tracker.measureAll([
    () => List.generate(1000, (i) => i * i),
    () => List.filled(1000, 0).fold(0, (a, b) => a + b),
  ]);

  print('Average: ${tracker.average}');

  try {
    riskyOperation();
  } on RangeError {
    print('Caught RangeError — Timeline block properly closed');
  }

  verboseLog('test');

  final elapsed = benchmarkFunction(() => 'hello' * 1000);
  print('Function avg: $elapsed');
}
```

**Пять проблем:** 1) `Stopwatch` никогда не запускался через `.start()` — все замеры были `Duration.zero`. 2) `.reduce()` на пустом списке бросает `StateError`; нужна проверка `isEmpty`. 3) `Timeline.startSync` без `try/finally` — при исключении блок остаётся открытым, что ломает DevTools Timeline граф. 4) `log(..., level: 'DEBUG')` — `level` принимает `int`, не `String`; будет `TypeError`. 5) `sw.elapsed` читается без предварительного `sw.stop()` — в Dart `elapsed` работает и без `stop()`, но это нечёткое измерение; главная проблема — новый `Stopwatch` создаётся на каждой итерации (лишние аллокации).
