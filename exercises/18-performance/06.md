В асинхронном коде есть проблемы производительности и корректности. Найди их.

```dart
import 'dart:async';

// bug 1 — await в цикле когда можно распараллелить
Future<Map<String, int>> countWords(List<String> urls) async {
  final counts = <String, int>{};
  for (final url in urls) {
    final text = await downloadText(url); // bug 1: последовательно
    counts[url] = text.split(' ').length;
  }
  return counts;
}

// bug 2 — Future.wait без обработки ошибок: одна ошибка убивает все
Future<List<String>> fetchAll(List<String> urls) {
  return Future.wait(urls.map(downloadText)); // bug 2: no eagerError: false
}

// bug 3 — создаём новый StreamController для каждого вызова listen
Stream<int> counterStream(int max) {
  final controller = StreamController<int>(); // bug 3: leak если не cancel
  Timer.periodic(const Duration(milliseconds: 100), (timer) {
    controller.add(timer.tick);
    if (timer.tick >= max) {
      timer.cancel();
      // bug 3a: controller.close() не вызывается
    }
  });
  return controller.stream;
}

// bug 4 — sync completer в async-контексте
Future<String> asyncOperation() {
  final completer = Completer<String>.sync(); // bug 4: sync completer опасен
  Future.delayed(Duration(seconds: 1), () {
    // Sync completer вызывает listener синхронно в стеке delayed-callback
    // Если listener бросит — exception уйдёт в зону delayed, а не вызывающего
    completer.complete('done');
  });
  return completer.future;
}

// bug 5 — блокирующее вычисление в main isolate
Future<List<int>> processData(List<int> rawData) async {
  // Тяжёлое O(n²) вычисление блокирует event loop
  final result = <int>[];
  for (var i = 0; i < rawData.length; i++) {
    for (var j = 0; j < rawData.length; j++) {
      if (rawData[i] > rawData[j]) result.add(rawData[i] - rawData[j]);
    }
  }
  return result;
}

// bug 6 — fire-and-forget Future без обработки ошибок
void startBackgroundSync() {
  _syncLoop(); // bug 6: ошибки потеряны — не await, не .then/.catchError
}

Future<void> _syncLoop() async {
  while (true) {
    await Future.delayed(Duration(seconds: 30));
    await syncWithServer(); // может бросить
  }
}

Future<void> syncWithServer() async {}
Future<String> downloadText(String url) async => 'sample text';
```

```dart
import 'dart:async';
import 'dart:isolate';

// fix 1: параллельная загрузка через Future.wait
Future<Map<String, int>> countWords(List<String> urls) async {
  final futures = urls.map((url) async {
    final text = await downloadText(url);
    return MapEntry(url, text.split(' ').length);
  });
  final entries = await Future.wait(futures);
  return Map.fromEntries(entries);
}

// fix 2: eagerError: false — collect all results, report errors individually
Future<List<String?>> fetchAll(List<String> urls) async {
  // Вариант 1: eagerError: false ждёт все Future, потом бросает первую ошибку
  // return Future.wait(urls.map(downloadText), eagerError: false);

  // Вариант 2 (рекомендуемый): обернуть в nullable — partial success
  final results = await Future.wait(
    urls.map((url) => downloadText(url).then<String?>((v) => v).onError((_, __) => null)),
  );
  return results;
}

// fix 3: используем StreamController с onListen/onCancel + закрываем controller
Stream<int> counterStream(int max) {
  late StreamController<int> controller;
  Timer? timer;

  controller = StreamController<int>(
    onListen: () {
      timer = Timer.periodic(const Duration(milliseconds: 100), (t) {
        controller.add(t.tick);
        if (t.tick >= max) {
          t.cancel();
          controller.close(); // fix 3a: закрываем поток
        }
      });
    },
    onCancel: () {
      timer?.cancel(); // fix: отменяем Timer при cancel подписки
    },
  );

  return controller.stream;
}

// fix 4: обычный (async) Completer
Future<String> asyncOperation() {
  final completer = Completer<String>(); // fix: без .sync()
  Future.delayed(const Duration(seconds: 1), () {
    completer.complete('done');
  });
  return completer.future;
  // Примечание: Completer.sync() нужен только в очень узких случаях
  // (например, внутри реализации StreamController), и то с осторожностью
}

// fix 5: тяжёлое вычисление → в Isolate
Future<List<int>> processData(List<int> rawData) {
  return Isolate.run(() {
    final result = <int>[];
    for (var i = 0; i < rawData.length; i++) {
      for (var j = 0; j < rawData.length; j++) {
        if (rawData[i] > rawData[j]) result.add(rawData[i] - rawData[j]);
      }
    }
    return result;
  });
}

// fix 6: обрабатываем ошибки в background loop
void startBackgroundSync() {
  unawaited(_syncLoop()); // явно обозначаем fire-and-forget через unawaited()
  // или сохраняем ссылку:
  // _syncFuture = _syncLoop();
}

Future<void> _syncLoop() async {
  while (true) {
    await Future.delayed(const Duration(seconds: 30));
    try {
      await syncWithServer();
    } catch (e, st) {
      // fix 6: логируем — не теряем молча
      print('Sync error: $e\n$st');
      // при необходимости: await Future.delayed(retryDelay);
    }
  }
}

// Импорт для unawaited
import 'package:meta/meta.dart'; // или просто void-игнорирование:
// ignore: avoid_unawaited_futures
// _syncLoop();

Future<void> syncWithServer() async {}
Future<String> downloadText(String url) async => 'sample text from $url';

Future<void> main() async {
  final urls = ['url1', 'url2', 'url3'];

  print(await countWords(urls));
  print(await fetchAll(urls));

  final stream = counterStream(3);
  await for (final tick in stream) {
    print('tick: $tick');
  }

  print(await asyncOperation());
  print(await processData([1, 2, 3, 4, 5]));
}
```

**Шесть проблем:** 1) `await` в цикле = последовательные запросы; `Future.wait` + `map` распараллеливает. 2) `Future.wait` с `eagerError: true` (по умолчанию) отменяет результаты при первой ошибке; `eagerError: false` или ручная обработка сохраняет partial results. 3) `Timer.periodic` без `controller.close()` — поток никогда не завершится + утечка Timer при отмене подписки. 4) `Completer.sync()` вызывает callbacks синхронно в стеке complete() — опасно и почти никогда не нужно. 5) O(n²) в `async` функции блокирует event loop; нужен `Isolate.run`. 6) Fire-and-forget без `.catchError`/`try-catch` — исключения теряются молча, нужно хотя бы логировать.
