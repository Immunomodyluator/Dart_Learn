<summary>Решение</summary>

```dart
// Часть 1:
// $ dart run --observe main.dart
//   или для выбора порта:
// $ dart run --observe=8181 --pause-isolates-on-start main.dart
//
// (A) _stopwatch.start()
// (B) _stopwatch.stop()
// (C) _stopwatch.elapsed
// (D) _stopwatch.stop()
// (E) _stopwatch.reset()

class Benchmark {
  final _stopwatch = Stopwatch();

  void start() => _stopwatch.start();  // (A)
  void stop()  => _stopwatch.stop();   // (B)

  Duration get elapsed => _stopwatch.elapsed; // (C)

  void reset() {
    _stopwatch.stop();  // (D) — сначала останавливаем
    _stopwatch.reset(); // (E) — затем сбрасываем
  }

  void printResult(String label) {
    print('$label: ${elapsed.inMicroseconds} µs '
          '(${elapsed.inMilliseconds} ms)');
  }
}

// Часть 3: Timeline API из dart:developer
import 'dart:developer';

void profiledSort(List<int> data) {
  Timeline.startSync('sort');  // (F)

  data.sort();

  Timeline.finishSync();       // (G) — finishSync, не endSync
}

// Часть 4:
void processItems(List<String> items) {
  for (final item in items) {
    log('Processing: $item', name: 'processing'); // (H) — функция log из dart:developer
    // ... бизнес-логика ...
  }
}

// Часть 5:
// $ dart devtools   (I)  — открывает DevTools в браузере
//
// Или подключиться к уже запущенному Observatory:
// $ dart devtools --vm-uri=http://127.0.0.1:8181/...

// --- Демонстрация ---
void main() {
  final bench = Benchmark();

  final data = List.generate(100000, (i) => 100000 - i);

  bench.start();
  profiledSort(data);
  bench.stop();
  bench.printResult('sort 100k elements');

  bench.reset();

  bench.start();
  processItems(['alpha', 'beta', 'gamma']);
  bench.stop();
  bench.printResult('processItems 3 strings');
}
```

**Ключевые команды:** `dart run --observe` запускает Observatory; `Timeline.startSync`/`Timeline.finishSync` добавляют именованные блоки в Timeline-трек DevTools; `log()` из `dart:developer` виден в DevTools Logging, а не в stdout; `dart devtools` запускает веб-интерфейс DevTools.
