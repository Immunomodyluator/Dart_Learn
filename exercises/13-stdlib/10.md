Код должен дождаться завершения запроса и вернуть результат через `Completer`. Найди все баги.

```dart
import 'dart:async';

class RequestQueue {
  final _controller = StreamController<String>();  // bug 1 — должен быть broadcast

  Stream<String> get responses => _controller.stream;

  final _pending = <String, Completer<String>>{};

  void send(String requestId, String payload) {
    final completer = Completer<String>();
    _pending[requestId] = completer;
    print('Sent: $payload');
    // симулируем асинхронный ответ
    Future.delayed(const Duration(milliseconds: 50), () {
      receive(requestId, 'response to $payload');
    });
  }

  void receive(String requestId, String response) {
    _controller.add(response);
    final completer = _pending[requestId];
    completer.complete(response);   // bug 2 — completer может быть null
    // bug 3 — не удаляем из _pending → утечка памяти
  }

  Future<String> sendAndWait(String id, String payload) {
    send(id, payload);
    return _pending[id]!.future;
  }

  Future<void> close() async {
    _controller.close();           // bug 4 — нет await
  }
}

void main() async {
  final queue = RequestQueue();

  // подписка 1
  queue.responses.listen((r) => print('Sub1: $r'));
  // подписка 2 — bug 1 даст StateError: Stream already has subscriber
  queue.responses.listen((r) => print('Sub2: $r'));

  final r1 = await queue.sendAndWait('req1', 'hello');
  final r2 = await queue.sendAndWait('req2', 'world');
  print('$r1, $r2');
  await queue.close();
}
```

```dart
import 'dart:async';

class RequestQueue {
  // fix 1: broadcast stream для нескольких подписчиков
  final _controller = StreamController<String>.broadcast();

  Stream<String> get responses => _controller.stream;

  final _pending = <String, Completer<String>>{};

  void send(String requestId, String payload) {
    final completer = Completer<String>();
    _pending[requestId] = completer;
    print('Sent: $payload');
    Future.delayed(const Duration(milliseconds: 50), () {
      receive(requestId, 'response to $payload');
    });
  }

  void receive(String requestId, String response) {
    _controller.add(response);
    final completer = _pending.remove(requestId); // fix 2+3: remove вернёт null если нет + удаляет
    completer?.complete(response);                 // null-safe вызов
  }

  Future<String> sendAndWait(String id, String payload) {
    send(id, payload);
    return _pending[id]!.future;
  }

  Future<void> close() async {
    await _controller.close();     // fix 4: await
  }
}

void main() async {
  final queue = RequestQueue();

  queue.responses.listen((r) => print('Sub1: $r'));
  queue.responses.listen((r) => print('Sub2: $r')); // теперь работает

  final r1 = await queue.sendAndWait('req1', 'hello');
  final r2 = await queue.sendAndWait('req2', 'world');
  print('$r1, $r2');
  await queue.close();
}
```

**Четыре бага:** 1) Обычный `StreamController` допускает только одного подписчика; для нескольких используй `.broadcast()`. 2) `_pending[requestId]` может вернуть `null` если `requestId` неизвестен — без null-проверки будет NPE. 3) После завершения Completer'а запись в `_pending` нужно удалять (`remove`), иначе растёт утечка памяти. 4) `_controller.close()` возвращает `Future<void>`; без `await` закрытие может не завершиться перед концом `main`.
