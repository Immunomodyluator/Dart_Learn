```dart
import 'dart:collection';

List<String> sortedWords(String text) {
  final words =
      text.toLowerCase().split(RegExp(r'[\s,\.!?;:]+'))
          .where((w) => w.isNotEmpty)
          .toSet()   // убираем дубликаты
          .toList();
  words.sort();      // Comparable.compareTo для String
  return words;
}

class LruCache<K, V> {
  final int capacity;
  final _map = LinkedHashMap<K, V>();

  LruCache(this.capacity);

  V? get(K key) {
    if (!_map.containsKey(key)) return null;
    // переместить в конец (MRU позиция)
    final value = _map.remove(key) as V;
    _map[key] = value;
    return value;
  }

  void put(K key, V value) {
    if (_map.containsKey(key)) _map.remove(key);
    if (_map.length >= capacity) _map.remove(_map.keys.first); // удаляем LRU
    _map[key] = value;
  }

  @override
  String toString() => _map.toString();
}

SplayTreeMap<String, int> wordFrequency(List<String> words) {
  final freq = SplayTreeMap<String, int>();
  for (final w in words) {
    freq[w] = (freq[w] ?? 0) + 1;
  }
  return freq;
}

void main() {
  print(sortedWords('Hello world, hello Dart!'));
  // [dart, hello, world]

  final cache = LruCache<String, int>(3);
  cache.put('a', 1);
  cache.put('b', 2);
  cache.put('c', 3);
  cache.get('a');       // a теперь MRU
  cache.put('d', 4);   // b вытесняется (LRU)
  print(cache);         // {c: 3, a: 1, d: 4}

  final freq = wordFrequency(['apple', 'banana', 'apple', 'cherry', 'banana', 'apple']);
  print(freq);          // {apple: 3, banana: 2, cherry: 1} — отсортировано по ключу
}
```

`LinkedHashMap` сохраняет порядок вставки — это основа LRU кэша: самый ранний ключ (LRU) всегда `_map.keys.first`. `SplayTreeMap` хранит ключи в отсортированном порядке и выполняет операции за амортизированное O(log n).
