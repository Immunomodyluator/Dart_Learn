<summary>Решение</summary>

```dart
import 'dart:math';

void mathBasics() {
  print(pi);           // 3.141592653589793
  print(e);            // 2.718281828459045

  print(sqrt(16));     // 4.0
  print(pow(2, 10));   // 1024.0
  print(log(100));     // 4.605170185988092
  print(log(1000) / log(10)); // 2.9999... ≈ 3.0

  print(3.7.floor());  // 3  (или: print(3.7.floorToDouble()) )
  print(3.2.ceil());   // 4
  print(max(-5, 3));   // 3
}

double degreesToRadians(double deg) => deg * pi / 180;

void trigonometry() {
  final angle = degreesToRadians(30);
  print(sin(angle).toStringAsFixed(4)); // 0.5000
  print(cos(angle).toStringAsFixed(4)); // 0.8660
}

void randomNumbers() {
  final rng = Random(42);
  print(rng.nextInt(99) + 1);        // nextInt(n) → 0..n-1, поэтому +1 для 1..99
  print(rng.nextDouble().toStringAsFixed(3));

  final list = [1, 2, 3, 4, 5];
  list.shuffle(rng);
  print(list);
}

void main() {
  mathBasics();
  trigonometry();
  randomNumbers();
}
```

`dart:math` не имеет функции `log(x, base)` — логарифм по произвольному основанию считается через `log(x) / log(base)`. `nextInt(n)` возвращает `[0, n)`, а не `[1, n]`; для диапазона `[min, max]` используй `min + rng.nextInt(max - min + 1)`. `list.shuffle(rng)` принимает опциональный `Random` для воспроизводимости.
