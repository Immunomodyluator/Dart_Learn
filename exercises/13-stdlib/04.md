# Задание 04 (13.2): dart:convert — 🐛 Найти баг

**Тема:** 13. Стандартная библиотека Dart | **Подтема:** 13.2 dart:convert

---

Код должен сериализовать/десериализовать конфигурацию. Найди все баги.

```dart
import 'dart:convert';

class Config {
  final String host;
  final int port;
  final bool ssl;

  Config(this.host, this.port, this.ssl);

  // bug 1 — jsonEncode принимает Map, не объект
  String toJson() => jsonEncode(this);

  factory Config.fromJson(String json) {
    // bug 2 — используем dynamic без приведения типов
    final map = jsonDecode(json);
    return Config(map['host'], map['port'], map['ssl']);
  }
}

void main() {
  final cfg = Config('localhost', 8080, true);
  final json = cfg.toJson();
  print(json);

  final cfg2 = Config.fromJson(json);
  print('${cfg2.host}:${cfg2.port} ssl=${cfg2.ssl}');

  // трансформация байт
  final bytes = utf8.encode('Привет');   // List<int>
  final back = utf8.decode(bytes);
  print(back == 'Привет');

  // bug 3 — base64 декодирование с неправильной строкой
  final b64 = base64.encode(bytes);
  final decoded = base64.decode(b64 + '!!');  // '!!' — невалидные base64 символы
  print(String.fromCharCodes(decoded));
}
```

<summary>Решение</summary>

```dart
import 'dart:convert';

class Config {
  final String host;
  final int port;
  final bool ssl;

  Config(this.host, this.port, this.ssl);

  // fix 1 — сначала конвертируем в Map, потом в JSON
  String toJson() => jsonEncode({
        'host': host,
        'port': port,
        'ssl': ssl,
      });

  factory Config.fromJson(String json) {
    // fix 2 — явное приведение типов
    final map = jsonDecode(json) as Map<String, dynamic>;
    return Config(
      map['host'] as String,
      map['port'] as int,
      map['ssl'] as bool,
    );
  }
}

void main() {
  final cfg = Config('localhost', 8080, true);
  final json = cfg.toJson();
  print(json); // {"host":"localhost","port":8080,"ssl":true}

  final cfg2 = Config.fromJson(json);
  print('${cfg2.host}:${cfg2.port} ssl=${cfg2.ssl}');
  // localhost:8080 ssl=true

  final bytes = utf8.encode('Привет');
  final back = utf8.decode(bytes);
  print(back == 'Привет'); // true

  // fix 3 — только валидные base64 символы
  final b64 = base64.encode(bytes);
  final decoded = base64.decode(b64);   // убираем '!!'
  print(utf8.decode(decoded));          // Привет
}
```

**Три бага:** 1) `jsonEncode(this)` — `jsonEncode` не умеет сериализовать произвольные объекты Dart; нужен `Map<String, dynamic>` или объект с методом `toJson()`, вызываемым через конвертер. 2) Результат `jsonDecode` — `dynamic`; без `as Map<String, dynamic>` обращение к полям не даст нужных типов и может упасть в рантайме. 3) Строка base64 не должна содержать символы вне алфавита base64 (`A–Z`, `a–z`, `0–9`, `+`, `/`, `=`) — `base64.decode` бросит `FormatException`.
