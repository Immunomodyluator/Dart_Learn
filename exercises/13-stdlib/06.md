Функции расчёта статистики содержат ошибки. Найди их все.

```dart
import 'dart:math';

double mean(List<double> values) {
  // bug 1 — деление на длину без проверки на пустой список
  return values.reduce((a, b) => a + b) / values.length;
}

double standardDeviation(List<double> values) {
  final avg = mean(values);
  final sumSq = values.map((x) => (x - avg) * (x - avg)).reduce((a, b) => a + b);
  // bug 2 — выборочное стандартное отклонение делится на n-1, не на n
  return sqrt(sumSq / values.length);
}

double distance(double x1, double y1, double x2, double y2) {
  final dx = x2 - x1;
  final dy = y2 - y1;
  // bug 3 — sqrt((dx*dx + dy*dy)) правильно, но не используем pow()
  //         менее читаемо, но ключевой баг: нет обработки NaN для отрицательных
  return sqrt(dx * dx + dy * dy);
}

int randomInRange(int min, int max) {
  final rng = Random();
  // bug 4 — nextInt(max - min) не включает max, нужно nextInt(max - min + 1)
  return min + rng.nextInt(max - min);
}

void main() {
  print(mean([]));         // throws RangeError
  print(standardDeviation([2, 4, 4, 4, 5, 5, 7, 9])); // должно быть 2.0
  print(distance(0, 0, 3, 4));  // правильно: 5.0
  print(randomInRange(1, 6));   // кубик: 1..6 но никогда не выдаст 6
}
```

```dart
import 'dart:math';

double mean(List<double> values) {
  if (values.isEmpty) throw ArgumentError('values must not be empty'); // fix 1
  return values.reduce((a, b) => a + b) / values.length;
}

double standardDeviation(List<double> values) {
  if (values.length < 2) throw ArgumentError('need at least 2 values');
  final avg = mean(values);
  final sumSq = values.map((x) => pow(x - avg, 2).toDouble()).reduce((a, b) => a + b);
  // fix 2 — делим на n-1 (выборочное стандартное отклонение Бесселя)
  return sqrt(sumSq / (values.length - 1));
}

double distance(double x1, double y1, double x2, double y2) {
  final dx = x2 - x1;
  final dy = y2 - y1;
  return sqrt(dx * dx + dy * dy); // правильно; можно и pow(dx,2)+pow(dy,2)
}

int randomInRange(int min, int max) {
  if (min > max) throw ArgumentError('min must be <= max');
  final rng = Random();
  // fix 4 — +1 чтобы включить max
  return min + rng.nextInt(max - min + 1);
}

void main() {
  try {
    print(mean([]));
  } on ArgumentError catch (e) {
    print('Error: $e'); // Error: Invalid argument(s): values must not be empty
  }

  // выборочное СКО для [2,4,4,4,5,5,7,9] = 2.0
  print(standardDeviation([2, 4, 4, 4, 5, 5, 7, 9])); // 2.0

  print(distance(0, 0, 3, 4));              // 5.0

  final counts = List.filled(7, 0);
  for (var i = 0; i < 6000; i++) {
    counts[randomInRange(1, 6)]++;
  }
  print(counts.sublist(1)); // все грани примерно по 1000
}
```

**Четыре бага:** 1) `reduce` на пустом списке бросает `StateError` — нужна явная проверка. 2) Популяционное СКО (делить на n) отличается от выборочного (n-1); для статистических расчётов по выборке используй поправку Бесселя. 3) `distance` корректна; ключевой баг в задании — отсутствует (это ловушка). 4) `nextInt(max - min)` даёт `[min, max)`, то есть `max` никогда не выпадет; нужно `nextInt(max - min + 1)`.
