# Задание 06 (19.3): Immutable data и value objects — 🐛 Найти баг

**Тема:** 19. Архитектура кода и паттерны | **Подтема:** 19.3 Immutable data и value objects

---

В «неизменяемых» объектах есть ошибки, нарушающие иммутабельность. Найди их.

```dart
// bug 1 — список внутри «иммутабельного» класса мутабелен
class Config {
  final String host;
  final List<String> allowedOrigins; // bug 1: список можно изменить снаружи

  const Config({required this.host, required this.allowedOrigins});
}

// bug 2 — copyWith изменяет оригинал вместо создания нового объекта
class UserProfile {
  String name;        // bug 2a: поля не final
  String? bio;

  UserProfile({required this.name, this.bio});

  UserProfile copyWith({String? name, String? bio}) {
    this.name = name ?? this.name; // bug 2b: мутирует this!
    this.bio  = bio  ?? this.bio;
    return this;                   // bug 2c: возвращает тот же объект
  }
}

// bug 3 — Equality по ссылке для value object
class Coordinate {
  final double lat;
  final double lon;

  const Coordinate(this.lat, this.lon);
  // bug 3: нет == и hashCode — сравнение по identity
}

// bug 4 — строитель не создаёт независимую копию
class OrderBuilder {
  final List<String> _items = [];
  String? _customerId;

  OrderBuilder customer(String id) {
    _customerId = id;
    return this;
  }

  OrderBuilder addItem(String item) {
    _items.add(item);
    return this;
  }

  Order build() {
    return Order(_customerId!, _items); // bug 4: Order получает ссылку на _items
    // дальнейшие вызовы addItem изменят уже созданный Order
  }
}

class Order {
  final String customerId;
  final List<String> items;

  Order(this.customerId, this.items); // bug 4b: items не скопирован и не защищён
}

// bug 5 — DateTime не иммутабелен? (тонкость)
class Event {
  final String name;
  final DateTime time;  // DateTime в Dart — immutable!
  final List<DateTime> _schedule;

  Event(this.name, this.time, List<DateTime> schedule)
      : _schedule = schedule; // bug 5: внешний список не скопирован

  List<DateTime> get schedule => _schedule; // bug 5b: возвращает изменяемый список
}
```

```dart
import 'dart:collection';

// fix 1: защитить список от внешних мутаций
class Config {
  final String host;
  final List<String> allowedOrigins;

  Config({required this.host, required List<String> allowedOrigins})
      // fix: создаём неизменяемую копию при конструировании
      : allowedOrigins = List.unmodifiable(allowedOrigins);

  // Или через const + literal:
  // const Config({required this.host, required this.allowedOrigins});
  // тогда вызывать с const-литералом: Config(host: 'x', allowedOrigins: ['y'])
}

// fix 2: final поля + правильный copyWith
class UserProfile {
  final String name; // fix 2a: final
  final String? bio;

  const UserProfile({required this.name, this.bio});

  UserProfile copyWith({String? name, String? bio}) {
    // fix 2b+2c: создаём НОВЫЙ объект, не мутируем this
    return UserProfile(
      name: name ?? this.name,
      bio:  bio  ?? this.bio,
    );
  }

  @override
  bool operator ==(Object other) =>
      other is UserProfile && name == other.name && bio == other.bio;

  @override
  int get hashCode => Object.hash(name, bio);

  @override
  String toString() => 'UserProfile(name: $name, bio: $bio)';
}

// fix 3: Value object требует == и hashCode
class Coordinate {
  final double lat;
  final double lon;

  const Coordinate(this.lat, this.lon);

  // fix: equality by value
  @override
  bool operator ==(Object other) =>
      other is Coordinate && lat == other.lat && lon == other.lon;

  @override
  int get hashCode => Object.hash(lat, lon);

  @override
  String toString() => 'Coordinate($lat, $lon)';
}

// fix 4: строитель и итоговый объект независимы
class OrderBuilder {
  final List<String> _items = [];
  String? _customerId;

  OrderBuilder customer(String id) {
    _customerId = id;
    return this;
  }

  OrderBuilder addItem(String item) {
    _items.add(item);
    return this;
  }

  Order build() {
    if (_customerId == null) throw StateError('Customer ID is required');
    // fix 4: передаём КОПИЮ списка
    return Order(_customerId!, List.unmodifiable(_items));
  }
}

class Order {
  final String customerId;
  final List<String> items; // защищён List.unmodifiable в build()

  const Order(this.customerId, this.items);

  @override
  String toString() => 'Order($customerId, $items)';
}

// fix 5: Event с неизменяемым расписанием
class Event {
  final String name;
  final DateTime time; // DateTime уже immutable — ОК
  final List<DateTime> _schedule;

  Event(this.name, this.time, List<DateTime> schedule)
      // fix 5: копируем и защищаем при конструировании
      : _schedule = List.unmodifiable(schedule);

  // fix 5b: возвращаем UnmodifiableListView вместо мутабельного списка
  List<DateTime> get schedule => UnmodifiableListView(_schedule);
  // Или: List<DateTime> get schedule => List.unmodifiable(_schedule);

  @override
  String toString() => 'Event($name at $time, schedule: $_schedule)';
}

void main() {
  // fix 1: нельзя мутировать allowedOrigins
  final cfg = Config(host: 'localhost', allowedOrigins: ['http://a.com']);
  try {
    (cfg.allowedOrigins as List).add('http://evil.com'); // UnsupportedError
  } on UnsupportedError {
    print('Config.allowedOrigins — защищён!');
  }

  // fix 2: copyWith создаёт новый объект
  final profile = const UserProfile(name: 'Alice', bio: 'Dev');
  final updated = profile.copyWith(name: 'Bob');
  assert(profile.name == 'Alice', 'Оригинал не изменился');
  assert(updated.name == 'Bob');
  assert(!identical(profile, updated));

  // fix 3: value equality
  const c1 = Coordinate(55.75, 37.61);
  const c2 = Coordinate(55.75, 37.61);
  assert(c1 == c2);     // теперь работает
  assert(!identical(c1, c2)); // разные объекты

  // fix 4: builder не влияет на созданный Order
  final builder = OrderBuilder()
    ..customer('user-1')
    ..addItem('apple');
  final order = builder.build();
  builder.addItem('banana'); // не влияет на order!
  assert(order.items.length == 1, 'Order не изменился после build');

  // fix 5: schedule защищён
  final event = Event('Meeting', DateTime.now(), [DateTime(2024, 1, 1)]);
  try {
    event.schedule.add(DateTime(2024, 12, 31)); // UnsupportedError
  } on UnsupportedError {
    print('Event.schedule — защищён!');
  }

  print('Все проверки иммутабельности пройдены!');
}
```

**Пять нарушений иммутабельности:** 1) `final List` только запрещает переприсвоение, но не мутацию содержимого — нужен `List.unmodifiable`. 2) `copyWith` мутирует `this` и возвращает ту же ссылку; нужно создавать `new UserProfile(...)`. 3) Value object без `==`/`hashCode` — два объекта с одинаковыми координатами будут «разными» в Map/Set. 4) Builder отдаёт внутренний `_items` напрямую — дальнейшие вызовы `addItem` незаметно меняют уже «готовый» объект. 5) `Event` не копирует входящий список и возвращает изменяемый геттер — внешний код может поменять расписание.
