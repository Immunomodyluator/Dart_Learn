**Исходный код с проблемами:**

```dart
// Богатый объект (God Object) + магические числа + длинные методы
class Manager {
  var d = <Map>[];
  var u = '';
  var s = 0;

  void proc(String t, String c, double a) {
    if (s == 0) {
      d.add({'t': t, 'c': c, 'a': a, 'dt': DateTime.now().toString()});
      if (d.length > 1000) d.removeAt(0);
      if (a > 10000) {
        print('ALERT: $a');
        s = 1;
      }
    } else if (s == 1) {
      print('blocked');
    }
  }

  double calc() {
    var r = 0.0;
    for (var i in d) {
      if (i['c'] == u) r += (i['a'] as double);
    }
    return r;
  }
}
```

**Задание:** Рефактори в соответствии с принципами:

1. Single Responsibility — разбить God Object
2. Понятные имена — нет однобуквенных идентификаторов
3. Нет магических чисел — именованные константы
4. Маленькие методы — каждый делает одно
5. Правильные типы — строгая типизация

```dart
import 'dart:collection';

// --- Доменный объект ---

enum TransactionStatus { pending, completed, flagged }

class Transaction {
  final String id;
  final String type;          // 't' → type
  final String currency;      // 'c' → currency
  final double amount;        // 'a' → amount
  final DateTime createdAt;   // 'dt' → createdAt
  final TransactionStatus status;

  const Transaction({
    required this.id,
    required this.type,
    required this.currency,
    required this.amount,
    required this.createdAt,
    this.status = TransactionStatus.pending,
  });

  @override
  String toString() =>
      'Transaction($id: $amount $currency [$status])';
}

// --- Политика мониторинга (Single Responsibility) ---

abstract interface class FraudPolicy {
  bool isSuspicious(Transaction tx);
}

class LargeAmountFraudPolicy implements FraudPolicy {
  static const _threshold = 10000.0; // именованная константа вместо 10000

  @override
  bool isSuspicious(Transaction tx) => tx.amount > _threshold;
}

// --- Алерт-сервис (Single Responsibility) ---

abstract interface class AlertService {
  void sendAlert(String message);
}

class ConsoleAlertService implements AlertService {
  @override
  void sendAlert(String message) => print('⚠ ALERT: $message');
}

// --- Репозиторий транзакций (Single Responsibility) ---

class TransactionRepository {
  static const _maxHistorySize = 1000; // именованная константа

  final _transactions = ListQueue<Transaction>();

  void add(Transaction tx) {
    if (_transactions.length >= _maxHistorySize) {
      _transactions.removeFirst(); // FIFO eviction
    }
    _transactions.addLast(tx);
  }

  List<Transaction> findByCurrency(String currency) {
    return _transactions
        .where((tx) => tx.currency == currency)
        .toList();
  }

  int get count => _transactions.length;
}

// --- Сервис аккаунта (Single Responsibility) ---

enum AccountStatus { active, blocked }

class AccountService {
  final TransactionRepository _repository;
  final FraudPolicy _fraudPolicy;
  final AlertService _alertService;

  AccountStatus _status = AccountStatus.active;
  String _defaultCurrency;

  AccountService({
    required TransactionRepository repository,
    required FraudPolicy fraudPolicy,
    required AlertService alertService,
    String defaultCurrency = 'USD',
  })  : _repository = repository,
        _fraudPolicy = fraudPolicy,
        _alertService = alertService,
        _defaultCurrency = defaultCurrency;

  AccountStatus get status => _status;

  /// Обрабатывает транзакцию.
  /// Возвращает false если аккаунт заблокирован.
  bool processTransaction(Transaction transaction) {
    if (_status == AccountStatus.blocked) {
      print('Account is blocked. Transaction rejected.');
      return false;
    }

    _repository.add(transaction);
    _checkForFraud(transaction);
    return true;
  }

  double totalAmountInCurrency(String currency) {
    return _repository
        .findByCurrency(currency)
        .fold(0.0, (sum, tx) => sum + tx.amount);
  }

  double get defaultCurrencyTotal => totalAmountInCurrency(_defaultCurrency);

  void _checkForFraud(Transaction transaction) {
    if (_fraudPolicy.isSuspicious(transaction)) {
      _alertService.sendAlert(
        'Suspicious transaction: ${transaction.amount} ${transaction.currency}',
      );
      _status = AccountStatus.blocked;
    }
  }
}

// --- Демонстрация ---
void main() {
  final service = AccountService(
    repository: TransactionRepository(),
    fraudPolicy: LargeAmountFraudPolicy(),
    alertService: ConsoleAlertService(),
    defaultCurrency: 'USD',
  );

  service.processTransaction(Transaction(
    id: 'tx-1',
    type: 'purchase',
    currency: 'USD',
    amount: 250.0,
    createdAt: DateTime.now(),
  ));

  print('Total USD: \$${service.defaultCurrencyTotal}'); // 250.0

  service.processTransaction(Transaction(
    id: 'tx-2',
    type: 'wire',
    currency: 'USD',
    amount: 15000.0, // > 10000 → fraud alert → blocked
    createdAt: DateTime.now(),
  ));

  print('Status: ${service.status}'); // AccountStatus.blocked

  final rejected = service.processTransaction(Transaction(
    id: 'tx-3',
    type: 'purchase',
    currency: 'USD',
    amount: 10.0,
    createdAt: DateTime.now(),
  ));
  print('tx-3 accepted: $rejected'); // false — blocked
}
```

**Применённые принципы:** 1) God Object `Manager` разбит на `Transaction`, `TransactionRepository`, `AccountService`, `FraudPolicy`, `AlertService` — у каждого одна ответственность. 2) Однобуквенные `d/u/s/t/c/a/r/i` → `_transactions/defaultCurrency/_status/type/currency/amount/sum/tx`. 3) Магические числа `0`, `1`, `1000`, `10000` → `AccountStatus.*`, `_maxHistorySize`, `_threshold`. 4) `proc()` разбит на `processTransaction`/`_checkForFraud`/`totalAmountInCurrency`. 5) `Map` → `Transaction`; `var` → строго типизированные поля.
