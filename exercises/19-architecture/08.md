// Файл: everything.dart — всё в одном файле

import 'dart:io';
import 'dart:convert';

// anti-pattern 1: Singleton через глобальные переменные
var db;
var logger;
var config;

void init() {
  db = {};
  logger = print;
  config = {'debug': true, 'maxRetries': 3};
}

// anti-pattern 2: Feature Envy — метод в не том классе
class User {
  String name;
  String email;
  DateTime? lastLogin;
  List<String> permissions;

  User(this.name, this.email, this.permissions, {this.lastLogin});

  // Метод слишком много знает о внешней системе
  void sendWelcomeEmail() {
    // Логика отправки email прямо в User
    final subject = 'Welcome, $name!';
    final body = 'Your account: $email\nPermissions: $permissions';
    print('SMTP: $subject\n$body'); // обращается к внешнему сервису
    logger('Email sent to $email');
    db['email_log'] ??= [];
    db['email_log'].add({'to': email, 'subject': subject});
  }

  bool canAccess(String resource) {
    // правила авторизации прямо в User
    if (permissions.contains('admin')) return true;
    if (resource.startsWith('/public')) return true;
    return permissions.contains('read:$resource');
  }
}

// anti-pattern 3: Primitive Obsession
class Order {
  String userId;       // ID пользователя — просто строка
  String status;       // 'pending', 'shipped', 'delivered', 'cancelled' — строки!
  double totalPrice;   // деньги как double
  String address;      // полный адрес в одной строке

  Order(this.userId, this.status, this.totalPrice, this.address);

  void ship() {
    if (status == 'peding') { // anti-pattern 3b: typo нельзя поймать на этапе компиляции
      status = 'shipped';
    }
  }
}

// anti-pattern 4: Long Parameter List
void createReport(
  String userId,
  String reportType,
  DateTime startDate,
  DateTime endDate,
  String currency,
  bool includeVat,
  bool groupByMonth,
  String format,
  String? email,
  int? maxRows,
) {
  // 10 параметров — сложно использовать
}

// anti-pattern 5: Train Wreck (Law of Demeter violation)
class Invoice {
  Order order;
  Invoice(this.order);

  String getCity() {
    return order.address.split(',')[2].trim(); // цепочка из 3 знаний о внутреннем устройстве
  }
}

void main() {
  init();
  final user = User('Alice', 'alice@test.com', ['read:reports']);
  user.sendWelcomeEmail();

```dart
import 'dart:io';

// fix 1: избегаем глобальных переменных — используем DI
// (см. задание 19.2 — Container или просто передача через конструктор)

// fix 2: Feature Envy — убираем из User всё, что не относится к данным пользователя
class User {
  final String id;
  final String name;
  final String email;
  final DateTime? lastLogin;
  final List<String> permissions;

  const User({
    required this.id,
    required this.name,
    required this.email,
    required this.permissions,
    this.lastLogin,
  });
}

// fix 2a: Email-логика → EmailService
abstract interface class EmailService {
  void sendWelcome(User user);
}

class SmtpEmailService implements EmailService {
  final EmailLog _log;
  const SmtpEmailService(this._log);

  @override
  void sendWelcome(User user) {
    final subject = 'Welcome, ${user.name}!';
    print('SMTP: $subject to ${user.email}');
    _log.record(to: user.email, subject: subject);
  }
}

class EmailLog {
  final _entries = <Map<String, String>>[];
  void record({required String to, required String subject}) {
    _entries.add({'to': to, 'subject': subject});
  }
  int get count => _entries.length;
}

// fix 2b: авторизация → AuthorizationService
class AuthorizationService {
  bool canAccess(User user, String resource) {
    if (user.permissions.contains('admin')) return true;
    if (resource.startsWith('/public')) return true;
    return user.permissions.contains('read:$resource');
  }
}

// fix 3: Primitive Obsession → value types
final class UserId {
  final String value;
  UserId(this.value) {
    if (value.isEmpty) throw ArgumentError('UserId cannot be empty');
  }
  @override String toString() => value;
  @override bool operator ==(Object o) => o is UserId && value == o.value;
  @override int get hashCode => value.hashCode;
}

// fix 3a: enum вместо строк — typo невозможен
enum OrderStatus { pending, shipped, delivered, cancelled }

// fix 3c: деньги в целых единицах (центы)
class Money {
  final int cents;
  final String currency;
  const Money(this.cents, this.currency);
  @override String toString() => '${cents / 100} $currency';
}

// fix 3d: StreetAddress вместо одной строки
class StreetAddress {
  final String street;
  final String city;
  final String country;
  const StreetAddress({required this.street, required this.city, required this.country});
}

class Order {
  final UserId userId;
  OrderStatus status;
  final Money totalPrice;
  final StreetAddress address;

  Order({
    required this.userId,
    required this.totalPrice,
    required this.address,
    this.status = OrderStatus.pending,
  });

  void ship() {
    if (status == OrderStatus.pending) { // fix: нет опечаток — enum
      status = OrderStatus.shipped;
    }
  }
}

// fix 4: Parameter Object вместо длинного списка
class ReportRequest {
  final String userId;
  final String reportType;
  final DateTime startDate;
  final DateTime endDate;
  final String currency;
  final bool includeVat;
  final bool groupByMonth;
  final String format;
  final String? email;
  final int? maxRows;

  const ReportRequest({
    required this.userId,
    required this.reportType,
    required this.startDate,
    required this.endDate,
    required this.currency,
    this.includeVat = false,
    this.groupByMonth = false,
    this.format = 'pdf',
    this.email,
    this.maxRows,
  });
}

void createReport(ReportRequest request) {
  // fix 4: один параметр, понятный named-параметры
  print('Creating ${request.reportType} (${request.currency})');
}

// fix 5: Law of Demeter — Invoice знает только о своих прямых зависимостях
class Invoice {
  final Order order;
  Invoice(this.order);

  // fix 5: Invoice спрашивает address.city — не лезет внутрь address
  String get city => order.address.city; // не split(',')[2].trim()
}

void main() {
  final log = EmailLog();
  final emailSvc = SmtpEmailService(log);
  final authSvc = AuthorizationService();

  final user = User(
    id: 'user-1',
    name: 'Alice',
    email: 'alice@test.com',
    permissions: ['read:reports'],
  );

  emailSvc.sendWelcome(user);
  print('Emails sent: ${log.count}');

  print('Can access /reports: ${authSvc.canAccess(user, 'reports')}'); // true
  print('Can access /admin: ${authSvc.canAccess(user, 'admin')}');     // false

  final order = Order(
    userId: UserId('user-1'),
    totalPrice: Money(9999, 'USD'),
    address: StreetAddress(street: '1 Main St', city: 'Boston', country: 'US'),
  );
  order.ship();
  print('Order status: ${order.status}'); // OrderStatus.shipped

  final invoice = Invoice(order);
  print('Invoice city: ${invoice.city}'); // Boston

  createReport(ReportRequest(
    userId: 'user-1',
    reportType: 'sales',
    startDate: DateTime(2024, 1),
    endDate: DateTime(2024, 12),
    currency: 'USD',
    groupByMonth: true,
  ));
}
```

**Пять антипаттернов:** 1) Глобальные `var db/logger/config` — невозможно тестировать, скрытые зависимости. 2) Feature Envy — `User.sendWelcomeEmail()` знает о SMTP, логировании и БД; `canAccess` реализует авторизацию. 3) Primitive Obsession — string/double для ID, статуса, денег, адреса; опечатка `'peding'` не поймана компилятором. 4) Long Parameter List из 10 параметров — заменяется Parameter Object. 5) Train Wreck `order.address.split(',')[2].trim()` — Invoice знает о внутреннем формате строки адреса через три уровня обращений.
