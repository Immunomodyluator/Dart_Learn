```dart
// lib/src/domain/user.dart
// (E) — final поля + const конструктор = immutable класс
class User {
  final String id;       // (E1) final
  final String name;     // (E2) final
  final String? email;   // (E3) final

  const User({           // (E4) const конструктор с именованными параметрами
    required this.id,
    required this.name,
    this.email,
  });

  // (F) copyWith — шаблон для immutable объектов
  User copyWith({String? name, String? email}) {
    return User(
      id: id,
      name: name ?? this.name,
      email: email ?? this.email,
    );
  }

  @override
  String toString() => 'User($id, $name, $email)';

  @override
  bool operator ==(Object other) =>
      other is User && id == other.id && name == other.name && email == other.email;

  @override
  int get hashCode => Object.hash(id, name, email);
}

// lib/src/domain/get_user_usecase.dart
// (G) — абстрактный интерфейс: domain не знает о реализации
abstract interface class UserRepository {  // (G)
  Future<User?> findById(String id);
}

class GetUserUseCase {
  final UserRepository _repo; // (H) зависит от абстракции, не от impl
  const GetUserUseCase(this._repo);

  Future<User?> call(String id) => _repo.findById(id); // (I)
}

// lib/src/data/user_repository_impl.dart
class InMemoryUserRepository implements UserRepository {
  final Map<String, User> _store;

  const InMemoryUserRepository(this._store);

  @override
  Future<User?> findById(String id) async => _store[id];
}

// lib/my_library.dart  (C) — barrel file
export 'src/domain/user.dart';           // (J)
export 'src/domain/get_user_usecase.dart';
// src/data НЕ экспортируется — детали реализации скрыты

// test/domain/get_user_usecase_test.dart  (D)
import 'package:test/test.dart';
import 'package:my_library/my_library.dart';
import '../lib/src/data/user_repository_impl.dart'; // тест может знать о реализации

void main() {
  group('GetUserUseCase', () {
    late GetUserUseCase useCase;

    setUp(() {
      final repo = InMemoryUserRepository({
        '1': const User(id: '1', name: 'Alice', email: 'alice@example.com'),
      });
      useCase = GetUserUseCase(repo);
    });

    test('returns user when found', () async {
      final user = await useCase('1');
      expect(user?.name, 'Alice');
    });

    test('returns null when not found', () async {
      final user = await useCase('999');
      expect(user, isNull);
    });
  });
}
```

**Ключевые принципы:** `final` + `const`-конструктор делает объект immutable; `copyWith` — стандартный паттерн для неизменяемых значений. `abstract interface class` разделяет домен от инфраструктуры (DIP). Barrel-файл экспортирует только публичный API — детали реализации в `src/` остаются приватными.
