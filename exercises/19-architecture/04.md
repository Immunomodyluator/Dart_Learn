В коде с DI есть ошибки. Найди их.

```dart
// bug 1 — Singleton через статическое поле (не DI-совместим)
class Logger {
  static final Logger _instance = Logger._();
  Logger._();
  static Logger get instance => _instance; // нельзя заменить в тестах

  void log(String msg) => print('[LOG] $msg');
}

// bug 2 — создание зависимостей внутри конструктора (hardcoded)
class ReportService {
  late final Logger _logger;
  late final DatabaseClient _db;

  ReportService() {
    _logger = Logger.instance;    // bug 2a: hardcoded singleton
    _db = DatabaseClient.connect('localhost:5432'); // bug 2b: side effect в конструкторе
  }

  Future<String> generate(String reportId) async {
    _logger.log('Generating $reportId');
    final data = await _db.query('SELECT * FROM reports WHERE id=?', [reportId]);
    return data.toString();
  }
}

// bug 3 — циклическая зависимость
class ServiceA {
  final ServiceB b;
  ServiceA(this.b);
}

class ServiceB {
  final ServiceC c;
  ServiceB(this.c);
}

class ServiceC {
  final ServiceA a; // bug 3: C → A → B → C — цикл!
  ServiceC(this.a);
}

// bug 4 — контейнер теряет типы (возвращает Object вместо T)
class BadContainer {
  final _store = <String, Object>{};

  void register(String name, Object instance) {
    _store[name] = instance;
  }

  Object resolve(String name) { // bug 4: теряем типобезопасность
    return _store[name] ?? (throw StateError('Not found: $name'));
  }
}

// bug 5 — lazy singleton создаётся заново при concurrent вызовах
class LazyContainer {
  final _instances = <Type, Object>{};
  final _factories = <Type, Object Function()>{};

  void singleton<T extends Object>(T Function() factory) {
    _factories[T] = factory;
  }

  Future<T> resolveLazy<T extends Object>() async {
    if (_instances.containsKey(T)) return _instances[T] as T;
    // bug 5: между проверкой и присвоением — другой await может создать ещё один instance
    await Future.delayed(Duration.zero); // симуляция async инициализации
    final instance = (_factories[T]!() as T);
    _instances[T] = instance;
    return instance;
  }
}

class DatabaseClient {
  static DatabaseClient connect(String url) => DatabaseClient._(url);
  DatabaseClient._(this.url);
  final String url;
  Future<List<dynamic>> query(String sql, List params) async => [];
}
```

<summary>Решение</summary>

```dart
// fix 1: Logger как интерфейс — внедряется через конструктор
abstract interface class Logger {
  void log(String msg);
}

class ConsoleLogger implements Logger {
  @override
  void log(String msg) => print('[LOG] $msg');
}

// В тестах можно использовать NoOpLogger:
class NoOpLogger implements Logger {
  @override
  void log(String msg) {}
}

// fix 2: все зависимости — через конструктор (Constructor Injection)
class ReportService {
  final Logger _logger;
  final DatabaseClient _db;

  // fix 2a: зависимость передаётся извне
  // fix 2b: нет side effect в конструкторе — инициализация отделена
  const ReportService(this._logger, this._db);

  Future<String> generate(String reportId) async {
    _logger.log('Generating $reportId');
    final data = await _db.query('SELECT * FROM reports WHERE id=?', [reportId]);
    return data.toString();
  }
}

// fix 3: решения циклической зависимости
// Вариант A: извлечь зависимость в интерфейс + ленивая инициализация
class ServiceA {
  final ServiceB b;
  const ServiceA(this.b);
  void doA() => print('A');
}

class ServiceB {
  final ServiceC c;
  const ServiceB(this.c);
  void doB() => print('B');
}

// fix 3: ServiceC зависит от интерфейса, который A реализует
abstract interface class AInterface {
  void doA();
}

class ServiceC {
  final AInterface a; // fix: зависимость через абстракцию
  const ServiceC(this.a);
}

// Сборка: A → B → C → AInterface (реализует A) — нет цикла
// Порядок создания: B(C(A)), A реализует AInterface

// fix 4: типобезопасный контейнер через Type как ключ
class GoodContainer {
  final _store = <Type, Object>{};

  void register<T extends Object>(T instance) {
    _store[T] = instance;
  }

  T resolve<T extends Object>() { // fix: возвращаем T
    final instance = _store[T];
    if (instance == null) throw StateError('Not found: $T');
    return instance as T;
  }
}

// fix 5: double-checked locking (в Dart — через Completer)
class LazyContainer {
  final _instances = <Type, Object>{};
  final _factories = <Type, Object Function()>{};
  final _pending = <Type, Future<Object>>{};

  void singleton<T extends Object>(T Function() factory) {
    _factories[T] = factory;
  }

  Future<T> resolveLazy<T extends Object>() async {
    // fix: если уже готово — вернуть сразу
    if (_instances.containsKey(T)) return _instances[T] as T;

    // fix: если уже инициализируется — ждать тот же Future
    if (_pending.containsKey(T)) return await _pending[T]! as T;

    // Иначе: начать инициализацию и запомнить Future
    final future = Future.microtask(() async {
      await Future.delayed(Duration.zero); // симуляция async init
      return _factories[T]!() as Object;
    });
    _pending[T] = future;

    final instance = await future as T;
    _instances[T] = instance;
    _pending.remove(T);
    return instance;
  }
}

class DatabaseClient {
  final String url;
  DatabaseClient(this.url);
  Future<List<dynamic>> query(String sql, List params) async => [];
}

Future<void> main() async {
  // fix 1+2: сборка зависимостей в одном месте (composition root)
  final logger = ConsoleLogger();
  final db = DatabaseClient('localhost:5432');
  final service = ReportService(logger, db);
  await service.generate('report-1');

  // fix 4: типобезопасный контейнер
  final container = GoodContainer();
  container.register<Logger>(ConsoleLogger());
  final resolvedLogger = container.resolve<Logger>(); // тип Logger, не Object
  resolvedLogger.log('Hello from container');

  // fix 5: lazy singleton без дублирования
  final lazy = LazyContainer();
  lazy.singleton<DatabaseClient>(() => DatabaseClient('db:5432'));

  // Параллельные resolve — один экземпляр
  final results = await Future.wait([
    lazy.resolveLazy<DatabaseClient>(),
    lazy.resolveLazy<DatabaseClient>(),
  ]);
  assert(identical(results[0], results[1]), 'Должен быть один singleton');
  print('Все проверки DI пройдены!');
}
```

**Пять проблем:** 1) `static final` singleton нельзя подменить в тестах — используй интерфейс + constructor injection. 2a) `Logger.instance` в конструкторе — hardcoded зависимость. 2b) Сетевое соединение в конструкторе — side effect, нельзя создать объект без реального подключения. 3) Циклическая зависимость `A→B→C→A` — невозможно создать объекты; решается через интерфейс или lazy setter. 4) `Object resolve(String name)` — потеря типов, нужны; дженерики `T resolve<T>()` + `Type` как ключ. 5) Async `resolveLazy` без синхронизации — параллельные вызовы создают несколько экземпляров singleton; нужно кешировать Future пока инициализация не завершена.
