Реализуй набор неизменяемых value objects и продемонстрируй их применение.

**Задание 1 — Value Object: Money**

Реализуй `Money` — неизменяемый объект-значение для денег:

- Хранит сумму в целых копейках/центах (избегаем floating point)
- Операции: `+`, `-`, `*` (на коэффициент), сравнение
- **Equality by value** (не by reference)
- Фабричный конструктор `Money.fromDouble(double amount, String currency)`
- Выбрасывает `ArgumentError` при отрицательной сумме (Money — не долг)

**Задание 2 — Immutable коллекция**

Реализуй `ImmutableList<T>` — обёртку над `List<T>` с неизменяемым API:

- `ImmutableList.of(List<T>)` — конструктор из обычного списка
- `operator []` для чтения
- `add(T item)` — возвращает **новый** `ImmutableList<T>` с добавленным элементом
- `where(bool Function(T) test)` — возвращает **новый** `ImmutableList<T>`
- `length`, `isEmpty`, `toList()`

**Задание 3 — Record как value object**

Используй Dart 3 `record` для координаты и диапазона:

```dart
typedef Point = ({double x, double y});
typedef Range = ({double min, double max});
```

Напиши функции `distance(Point a, Point b)` и `clamp(double v, Range r)`.

```dart
import 'dart:math';

/// Задание 1: Money Value Object
class Money implements Comparable<Money> {
  final int _cents; // всегда неотрицательный
  final String currency;

  const Money._(this._cents, this.currency);

  factory Money(int cents, String currency) {
    if (cents < 0) throw ArgumentError('Money cannot be negative: $cents');
    return Money._(cents, currency);
  }

  factory Money.fromDouble(double amount, String currency) {
    if (amount < 0) throw ArgumentError('Money cannot be negative: $amount');
    return Money._((amount * 100).round(), currency);
  }

  // Арифметика — возвращает новый объект
  Money operator +(Money other) {
    _assertSameCurrency(other);
    return Money._(_cents + other._cents, currency);
  }

  Money operator -(Money other) {
    _assertSameCurrency(other);
    final result = _cents - other._cents;
    if (result < 0) throw ArgumentError('Money cannot be negative after subtraction');
    return Money._(result, currency);
  }

  Money operator *(num factor) {
    if (factor < 0) throw ArgumentError('Factor cannot be negative');
    return Money._((_cents * factor).round(), currency);
  }

  bool operator <(Money other) { _assertSameCurrency(other); return _cents < other._cents; }
  bool operator >(Money other) { _assertSameCurrency(other); return _cents > other._cents; }
  bool operator <=(Money other) { _assertSameCurrency(other); return _cents <= other._cents; }
  bool operator >=(Money other) { _assertSameCurrency(other); return _cents >= other._cents; }

  @override
  int compareTo(Money other) {
    _assertSameCurrency(other);
    return _cents.compareTo(other._cents);
  }

  // Equality by VALUE
  @override
  bool operator ==(Object other) =>
      other is Money && _cents == other._cents && currency == other.currency;

  @override
  int get hashCode => Object.hash(_cents, currency);

  double get amount => _cents / 100.0;
  int get cents => _cents;

  @override
  String toString() => '${amount.toStringAsFixed(2)} $currency';

  void _assertSameCurrency(Money other) {
    if (currency != other.currency) {
      throw ArgumentError('Currency mismatch: $currency vs ${other.currency}');
    }
  }
}

/// Задание 2: ImmutableList
class ImmutableList<T> {
  final List<T> _inner;

  ImmutableList._(this._inner);

  factory ImmutableList.of(List<T> source) {
    return ImmutableList._(List.unmodifiable(source));
  }

  static ImmutableList<T> empty<T>() => ImmutableList._([]);

  T operator [](int index) => _inner[index];

  int get length => _inner.length;
  bool get isEmpty => _inner.isEmpty;
  bool get isNotEmpty => _inner.isNotEmpty;

  ImmutableList<T> add(T item) {
    return ImmutableList._(List.unmodifiable([..._inner, item]));
  }

  ImmutableList<T> where(bool Function(T) test) {
    return ImmutableList._(List.unmodifiable(_inner.where(test)));
  }

  ImmutableList<R> map<R>(R Function(T) f) {
    return ImmutableList._(List.unmodifiable(_inner.map(f)));
  }

  List<T> toList() => List.of(_inner);

  @override
  String toString() => 'ImmutableList($_inner)';

  @override
  bool operator ==(Object other) =>
      other is ImmutableList<T> &&
      _inner.length == other._inner.length &&
      List.generate(_inner.length, (i) => _inner[i] == other._inner[i])
          .every((e) => e);

  @override
  int get hashCode => Object.hashAll(_inner);
}

/// Задание 3: Dart 3 records
typedef Point = ({double x, double y});
typedef Range = ({double min, double max});

double distance(Point a, Point b) {
  final dx = b.x - a.x;
  final dy = b.y - a.y;
  return sqrt(dx * dx + dy * dy);
}

double clamp(double v, Range r) {
  assert(r.min <= r.max, 'Range.min must be <= Range.max');
  return v.clamp(r.min, r.max);
}

void main() {
  // Money
  final price = Money.fromDouble(19.99, 'USD');
  final tax   = Money.fromDouble(1.60, 'USD');
  final total = price + tax;
  print('Total: $total'); // 21.59 USD

  final doubled = price * 2;
  print('Doubled: $doubled'); // 39.98 USD

  // Value equality
  final m1 = Money(1999, 'USD');
  final m2 = Money(1999, 'USD');
  assert(m1 == m2);
  assert(!identical(m1, m2));

  // ImmutableList
  final list = ImmutableList.of([1, 2, 3]);
  final list2 = list.add(4).add(5);
  assert(list.length == 3);  // оригинал не изменился
  assert(list2.length == 5);

  final evens = list2.where((x) => x % 2 == 0);
  print('Evens: $evens'); // ImmutableList([2, 4])

  // Records
  final p1 = (x: 0.0, y: 0.0);
  final p2 = (x: 3.0, y: 4.0);
  print('Distance: ${distance(p1, p2)}'); // 5.0

  final range = (min: 0.0, max: 100.0);
  print('Clamp -5: ${clamp(-5, range)}');   // 0.0
  print('Clamp 150: ${clamp(150, range)}'); // 100.0
}
```

**Ключевые паттерны immutable objects:** 1) Храни деньги в целых единицах (центы) — нет ошибок плавающей точки. 2) Все методы возвращают новые объекты — внешний код не может изменить внутреннее состояние. 3) Переопределяй `==` и `hashCode` для семантики «равенства по значению» — value objects должны быть равны если их поля равны. 4) `List.unmodifiable()` защищает от мутации через `toList()`-копии. 5) Records — встроенный value type в Dart 3; `==` и `hashCode` генерируются автоматически.
