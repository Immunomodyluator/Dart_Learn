Реализуй простой Dependency Injection-контейнер для Dart без внешних пакетов.

**Задание 1 — Простой DI-контейнер**

```dart
class Container {
  // Зарегистрировать фабрику (каждый раз новый экземпляр)
  void register<T>(T Function(Container) factory);

  // Зарегистрировать синглтон (один экземпляр на всё время жизни контейнера)
  void singleton<T>(T Function(Container) factory);

  // Получить зависимость
  T resolve<T>();
}
```

**Задание 2 — Применение к трёхуровневой архитектуре**

Зарегистрируй и разреши зависимости для:

- `EmailService` (singleton)
- `UserRepository` (singleton)
- `UserService` (factory — новый при каждом resolve)
- `OrderService` зависящий от `UserService` и `EmailService`

**Задание 3 — Scoped-контейнер**

Реализуй `Container.scope()` — дочерний контейнер, который:

- Наследует регистрации родителя
- Создаёт собственные синглтоны (не разделяет с родителем)

```dart
/// Задание 1: DI-контейнер
class Container {
  final Container? _parent;
  final Map<Type, Object Function(Container)> _factories = {};
  final Map<Type, Object Function(Container)> _singletonFactories = {};
  final Map<Type, Object> _singletonInstances = {};

  Container({Container? parent}) : _parent = parent;

  void register<T extends Object>(T Function(Container) factory) {
    _factories[T] = factory;
  }

  void singleton<T extends Object>(T Function(Container) factory) {
    _singletonFactories[T] = factory;
  }

  T resolve<T extends Object>() {
    // Проверяем синглтоны этого контейнера
    if (_singletonInstances.containsKey(T)) {
      return _singletonInstances[T] as T;
    }
    if (_singletonFactories.containsKey(T)) {
      final instance = _singletonFactories[T]!(this) as T;
      _singletonInstances[T] = instance;
      return instance;
    }
    // Проверяем обычные фабрики
    if (_factories.containsKey(T)) {
      return _factories[T]!(this) as T;
    }
    // Ищем у родителя
    if (_parent != null) {
      return _parent.resolve<T>();
    }
    throw StateError('No registration found for type $T');
  }

  /// Задание 3: Scoped-контейнер
  Container scope() => Container(parent: this);
}

/// Задание 2: Трёхуровневая архитектура

abstract interface class IEmailService {
  void send(String to, String subject);
}

class EmailService implements IEmailService {
  int sentCount = 0;
  @override
  void send(String to, String subject) {
    sentCount++;
    print('Email to $to: $subject');
  }
}

abstract interface class IUserRepository {
  String? findName(String id);
}

class InMemoryUserRepository implements IUserRepository {
  final _users = {'1': 'Alice', '2': 'Bob'};
  @override
  String? findName(String id) => _users[id];
}

class UserService {
  final IUserRepository _repo;
  const UserService(this._repo);
  String? getUserName(String id) => _repo.findName(id);
}

class OrderService {
  final UserService _users;
  final IEmailService _email;
  const OrderService(this._users, this._email);

  void placeOrder(String userId, String item) {
    final name = _users.getUserName(userId) ?? 'Unknown';
    _email.send('$name@example.com', 'Order confirmed: $item');
  }
}

Container buildContainer() {
  final c = Container();

  c.singleton<IEmailService>((_) => EmailService());
  c.singleton<IUserRepository>((_) => InMemoryUserRepository());

  // factory — новый UserService при каждом resolve
  c.register<UserService>(
    (c) => UserService(c.resolve<IUserRepository>()),
  );

  c.register<OrderService>(
    (c) => OrderService(
      c.resolve<UserService>(),
      c.resolve<IEmailService>(),
    ),
  );

  return c;
}

void main() {
  final container = buildContainer();

  // Синглтоны — одинаковые объекты
  final e1 = container.resolve<IEmailService>();
  final e2 = container.resolve<IEmailService>();
  assert(identical(e1, e2), 'EmailService должен быть синглтоном');

  // Фабрика — разные объекты
  final u1 = container.resolve<UserService>();
  final u2 = container.resolve<UserService>();
  assert(!identical(u1, u2), 'UserService должен быть новым каждый раз');

  // Использование OrderService
  final order = container.resolve<OrderService>();
  order.placeOrder('1', 'Dart book');

  // Scoped-контейнер
  final scope = container.scope();
  // Регистрируем в scope собственный EmailService (не влияет на parent)
  scope.singleton<IEmailService>((_) => EmailService());

  final scopedEmail = scope.resolve<IEmailService>();
  assert(!identical(e1, scopedEmail), 'Scoped EmailService — другой объект');

  // Но UserRepository наследуется из родителя
  final scopedRepo = scope.resolve<IUserRepository>();
  assert(identical(scopedRepo, container.resolve<IUserRepository>()),
      'IUserRepository наследуется из родителя');

  print('Все проверки DI прошли!');
}
```

**Ключевые концепции DI:** Container хранит две карты — фабрики (новый объект при каждом `resolve`) и singleton-фабрики (создаёт один раз и кешируется). Scoped-контейнер делегирует `resolve` родителю для незарегистрированных типов — объекты области видимости изолированы. Всё построено на `Type` как ключе — безопасно в AOT благодаря дженерикам без рефлексии.
