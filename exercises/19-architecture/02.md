// ИСПРАВЛЕННАЯ СТРУКТУРА

// lib/src/domain/user.dart
class User {
  final String id;
  final String name;
  final String? email;
  const User({required this.id, required this.name, this.email});
}

// lib/src/domain/user_repository.dart
// fix 1: абстрактный интерфейс в domain — не импортирует никакую инфраструктуру
abstract interface class UserRepository {
  Future<List<User>> getAllUsers();
}

// lib/src/domain/user_service.dart
// fix 1 + 2: domain зависит только от абстракций
class UserService {
  final UserRepository _repo; // fix 2: инъекция зависимости через конструктор

  const UserService(this._repo); // не создаём зависимость вручную

  Future<List<String>> getUserNames() async {
    final users = await _repo.getAllUsers();
    return users.map((u) => u.name).toList();
  }
}

// lib/src/data/sqlite_user_repository.dart
// fix: инфраструктура знает о domain, но не наоборот
import 'package:sqlite3/sqlite3.dart';
import '../domain/user.dart';
import '../domain/user_repository.dart';

class SqliteUserRepository implements UserRepository {
  final Database _db; // зависимость приходит снаружи

  SqliteUserRepository(this._db);

  @override
  Future<List<User>> getAllUsers() async {
    final rows = _db.select('SELECT id, name, email FROM users');
    return rows
        .map((r) => User(
              id: r['id'] as String,
              name: r['name'] as String,
              email: r['email'] as String?,
            ))
        .toList();
  }
}

// lib/src/presentation/user_screen.dart
// fix 3: presentation зависит только от domain-сервиса
// fix 4: зависимость инъектируется через конструктор
class UserScreen {
  final UserService _service; // fix 3+4: зависимость через интерфейс/конструктор

  const UserScreen(this._service);

  Future<void> render() async {
    final names = await _service.getUserNames(); // fix: нет дублирования логики
    for (final name in names) {
      print(name);
    }
  }
}

**Пять нарушений:** 1) `domain` импортирует `data`-слой и внешний пакет `sqlite3` — нарушение Dependency Rule (зависимости должны направляться внутрь, к domain). 2) `Database.instance` (singleton) в domain — нет возможности подменить в тестах. 3) `presentation` напрямую использует `Database` — нарушение слоистости. 4) `UserService()` и `Database.instance` создаются внутри классов — hardcoded зависимости. 5) Barrel-файл экспортирует инфраструктурные детали — внешние потребители пакета не должны видеть БД-реализацию.
