```yaml

version: 1.0.0

environment:
  sdk: '___'          # минимальная версия SDK: ^3.0.0

dependencies:
  # ...

  ___: ^1.25.0        # основной пакет для тестирования
  ___: ^3.0.0         # пакет для coverage HTML-отчётов
```

**2. Команды в терминале:**

```bash
# Запустить все тесты

# Запустить тесты с измерением покрытия
dart ___ --enable-coverage ___ test

# Сгенерировать HTML-отчёт из LCOV файла
dart pub ___ covtool
genhtml ___ -o coverage/html

# Запустить только тесты с тегом 'unit'
dart test ___=unit
```

<details>
<summary>Решение</summary>

**1. pubspec.yaml:**

```yaml
name: my_dart_app
version: 1.0.0

environment:
  sdk: '>=3.0.0 <4.0.0'   # диапазон версий SDK

dependencies:
  # ...

dev_dependencies:
  test: ^1.25.0
  coverage: ^3.0.0
```

**2. Команды:**

```bash
# Запустить все тесты
dart test

# Запустить тесты с измерением покрытия (встроено в dart test)
dart test --coverage=coverage

# Сгенерировать HTML-отчёт из LCOV файла
dart pub global activate coverage
dart pub global run coverage:format_coverage \
  --lcov --in=coverage --out=coverage/lcov.info --packages=.dart_tool/package_config.json
genhtml coverage/lcov.info -o coverage/html

# Запустить только тесты с тегом 'unit'
dart test --tags=unit
# или: dart test -t unit
```

**3. GitHub Actions:**

```yaml
name: Tests
on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dart-lang/setup-dart@v1
        with:
          sdk: stable
      - run: dart pub get
      - run: dart analyze
      - run: dart test --coverage=coverage
      - name: Generate coverage report
        run: |
          dart pub global activate coverage
          dart pub global run coverage:format_coverage \
            --lcov --in=coverage --out=coverage/lcov.info \
            --packages=.dart_tool/package_config.json
      - uses: coverallsapp/github-action@v2
        with:
          path-to-lcov: coverage/lcov.info
```

`dart test --coverage=coverage` создаёт raw coverage данные; `format_coverage` преобразует их в LCOV формат. Для GitHub Actions используй `dart-lang/setup-dart@v1` для установки SDK. Теги (`--tags`) позволяют разделить медленные интеграционные тесты от быстрых unit-тестов.
