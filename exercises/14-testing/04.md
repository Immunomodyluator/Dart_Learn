Тесты написаны, но проверяют не то, что нужно. Найди все проблемы.

import 'package:test/test.dart';

abstract interface class EmailService {
  Future<void> sendEmail(String to, String subject, String body);
}

class FakeEmailService implements EmailService {
  final List<String> sentEmails = [];

  @override
  Future<void> sendEmail(String to, String subject, String body) async {
    sentEmails.add(to);
  }
}

class NotificationService {
  final EmailService emailService;
  NotificationService(this.emailService);

  Future<void> notifyUser(String email, String message) =>
      emailService.sendEmail(email, 'Notification', message);

  Future<void> notifyAll(List<String> emails, String message) async {
    // bug in production code? нет — намеренно только первый
    await emailService.sendEmail(emails.first, 'Broadcast', message);
  }
}

void main() {
  late FakeEmailService fake;
  late NotificationService service;

  setUp(() {
    service = NotificationService(fake);  // bug 1 — fake ещё не инициализирован!
    fake = FakeEmailService();
  });

  test('notifyUser sends email', () async {
    await service.notifyUser('user@test.com', 'hello');
    // bug 2 — проверяем длину, но не содержимое
    expect(fake.sentEmails.length, 1);
  });

  test('notifyAll sends to all users', () async {
    final emails = ['a@test.com', 'b@test.com', 'c@test.com'];
    await service.notifyAll(emails, 'broadcast');
    // bug 3 — тест ожидает 3 письма, но реализация отправляет только первое
    expect(fake.sentEmails, hasLength(3));
  });

  // bug 4 — тест не изолирован: fake не сбрасывается между тестами
  test('notifyUser after notifyAll', () async {
    await service.notifyUser('c@test.com', 'direct');
    expect(fake.sentEmails, contains('c@test.com'));  // может содержать старые данные
  });
}

<summary>Решение</summary>

```dart
import 'package:test/test.dart';

abstract interface class EmailService {
  Future<void> sendEmail(String to, String subject, String body);
}

class FakeEmailService implements EmailService {
  final List<String> sentEmails = [];

  @override
  Future<void> sendEmail(String to, String subject, String body) async {
    sentEmails.add(to);
  }
}

class NotificationService {
  final EmailService emailService;
  NotificationService(this.emailService);

  Future<void> notifyUser(String email, String message) =>
      emailService.sendEmail(email, 'Notification', message);

  Future<void> notifyAll(List<String> emails, String message) async {
    // реализация отправляет только первому — тест должен это отражать
    await emailService.sendEmail(emails.first, 'Broadcast', message);
  }
}

void main() {
  late FakeEmailService fake;
  late NotificationService service;

  setUp(() {
    fake = FakeEmailService();             // fix 1: сначала fake, потом service
    service = NotificationService(fake);
  });                                      // fix 4: setUp пересоздаёт оба — изоляция гарантирована

  test('notifyUser sends email to correct address', () async {
    await service.notifyUser('user@test.com', 'hello');
    expect(fake.sentEmails, equals(['user@test.com']));  // fix 2: проверяем содержимое
  });

  test('notifyAll sends email only to first user', () async {
    // fix 3: тест соответствует реальному поведению реализации
    final emails = ['a@test.com', 'b@test.com', 'c@test.com'];
    await service.notifyAll(emails, 'broadcast');
    expect(fake.sentEmails, equals(['a@test.com'])); // только первый!
    expect(fake.sentEmails, hasLength(1));
  });

  test('notifyUser sends independently', () async {
    await service.notifyUser('c@test.com', 'direct');
    expect(fake.sentEmails, equals(['c@test.com'])); // только это письмо
  });
}
```

**Четыре бага:** 1) В `setUp` `fake` инициализируется **после** `service` — `NotificationService(fake)` передаёт неинициализированную переменную; `LateInitializationError` гарантирован. 2) `hasLength(1)` не проверяет, что письмо отправлено правильному получателю — нужно сравнивать содержимое. 3) Тест ожидает 3 письма, но реализация отправляет только одно — тест должен описывать **текущее** поведение, а не желаемое (или нужно исправить реализацию). 4) `setUp` создаёт новый `FakeEmailService` — это правильный паттерн изоляции; баг был в порядке инициализации.
