Напиши интеграционный тест, проверяющий взаимодействие нескольких компонентов.

**Система:**

```dart
// В памяти реализация (без внешних зависимостей)
class InMemoryProductRepo {
  final _products = <int, Product>{};
  int _nextId = 1;

  Future<Product> add(String name, double price) async {
    final p = Product(_nextId++, name, price);
    _products[p.id] = p;
    return p;
  }
  Future<Product?> findById(int id) async => _products[id];
  Future<List<Product>> findAll() async => _products.values.toList();
}

class Product {
  final int id;
  final String name;
  final double price;
  Product(this.id, this.name, this.price);
}

class Cart {
  final InMemoryProductRepo repo;
  Cart(this.repo);

  final _items = <int, int>{};  // productId → quantity

  Future<void> addItem(int productId, int qty) async {
    final p = await repo.findById(productId);
    if (p == null) throw ArgumentError('Product $productId not found');
    _items[productId] = (_items[productId] ?? 0) + qty;
  }

  Future<double> total() async {
    double sum = 0;
    for (final entry in _items.entries) {
      final p = (await repo.findById(entry.key))!;
      sum += p.price * entry.value;
    }
    return sum;
  }
}

import 'package:test/test.dart';

// --- Весь production code встроен для примера ---
class Product {
  final int id;
  final String name;
  final double price;
  Product(this.id, this.name, this.price);
  @override
  String toString() => 'Product($id, $name, \$$price)';
}

class InMemoryProductRepo {
  final _products = <int, Product>{};
  int _nextId = 1;

  Future<Product> add(String name, double price) async {
    final p = Product(_nextId++, name, price);
    _products[p.id] = p;
    return p;
  }
  Future<Product?> findById(int id) async => _products[id];
  Future<List<Product>> findAll() async => _products.values.toList();
}

class Cart {
  final InMemoryProductRepo repo;
  Cart(this.repo);
  final _items = <int, int>{};

  Future<void> addItem(int productId, int qty) async {
    final p = await repo.findById(productId);
    if (p == null) throw ArgumentError('Product $productId not found');
    _items[productId] = (_items[productId] ?? 0) + qty;
  }

  Future<double> total() async {
    double sum = 0;
    for (final entry in _items.entries) {
      final p = (await repo.findById(entry.key))!;
      sum += p.price * entry.value;
    }
    return sum;
  }
}

// --- Тесты ---
void main() {
  late InMemoryProductRepo repo;
  late Cart cart;

  setUp(() {
    repo = InMemoryProductRepo();
    cart = Cart(repo);
  });

  group('Cart integration', () {
    test('calculates total of two products', () async {
      final apple = await repo.add('Apple', 1.5);
      final bread = await repo.add('Bread', 2.0);

      await cart.addItem(apple.id, 3);  // 3 × 1.5 = 4.5
      await cart.addItem(bread.id, 2);  // 2 × 2.0 = 4.0

      expect(await cart.total(), closeTo(8.5, 0.001));
    });

    test('throws ArgumentError for unknown product', () async {
      await expectLater(
        () => cart.addItem(999, 1),
        throwsA(isA<ArgumentError>()),
      );
    });

    test('accumulates quantity when same product added twice', () async {
      final milk = await repo.add('Milk', 1.0);
      await cart.addItem(milk.id, 2);
      await cart.addItem(milk.id, 3);  // итого 5 × 1.0 = 5.0
      expect(await cart.total(), closeTo(5.0, 0.001));
    });

    test('empty cart has total of 0', () async {
      expect(await cart.total(), equals(0.0));
    });
  });
}
```

Интеграционный тест проверяет несколько компонентов вместе (`Cart` + `InMemoryProductRepo`). Используй `closeTo(value, delta)` при сравнении чисел с плавающей точкой — прямое `equals` может быть ненадёжным из-за округления. `setUp` гарантирует чистое состояние перед каждым тестом.
