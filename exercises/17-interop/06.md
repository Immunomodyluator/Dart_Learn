В коде есть ошибки, связанные с неправильным использованием `dart:mirrors` и неверными предположениями о рефлексии. Найди их.

```dart
import 'dart:mirrors';

class Config {
  String host;
  int port;
  bool _debug;          // приватное поле

  Config(this.host, this.port, this._debug);
}

// bug 1 — пытаемся сериализовать объект через mirrors в AOT-скомпилированном коде
// (например, скомпилированном как dart compile exe)
Map<String, dynamic> toJson(Object obj) {
  final mirror = reflect(obj);
  final result = <String, dynamic>{};

  mirror.type.declarations.forEach((symbol, decl) {
    if (decl is VariableMirror) {  // bug 2 — включаем и приватные поля (они не должны попадать в JSON)
      final name = MirrorSystem.getName(symbol);
      result[name] = mirror.getField(symbol).reflectee;
    }
  });
  return result;
}

// bug 3 — создание экземпляра класса по имени строки в среде без mirrors
Object createInstance(String className, List<dynamic> args) {
  // В AOT-компиляции classMirror будет null и программа упадёт
  final classMirror = currentMirrorSystem()
      .findLibrary(Symbol('dart.core'))
      .declarations[Symbol(className)] as ClassMirror;  // bug 3a: dart.core не содержит пользовательские классы
                                                         // bug 3b: приведение без проверки

  return classMirror.newInstance(Symbol(''), args).reflectee;
}

class PluginManager {
  final Map<String, Object> _plugins = {};

  // bug 4 — getField по имени не работает для методов, только для переменных
  void callMethod(Object plugin, String methodName) {
    final mirror = reflect(plugin);
    // getField работает только для полей/геттеров, а не методов
    final result = mirror.getField(Symbol(methodName));
    print('Result: $result');
  }

  // bug 5 — setCField вызывается на финальном поле
  void resetPlugin(Object plugin) {
    final mirror = reflect(plugin);
    // Config.host — нефинальное, но Config.port сделаем "финальным" в мыслях
    mirror.setField(Symbol('host'), 'new_host');
    mirror.setField(Symbol('_debug'), true); // bug 5: попытка изменить приватное поле другого класса
  }
}

void main() {
  final cfg = Config('localhost', 8080, false);

  final json = toJson(cfg);
  print(json);  // будет содержать _debug: false — это утечка приватных данных

  // bug 6 — полагаемся на стабильность порядка symbols в declarations (порядок не гарантирован)
  final keys = reflect(cfg).type.declarations.keys
      .map(MirrorSystem.getName)
      .take(2)  // "берём первые 2 поля" — но порядок не определён
      .toList();
  print('First two fields: $keys');
}
```

```dart
// Исправленная версия — без dart:mirrors, AOT-совместимая

// fix 1 & 2: Не используем dart:mirrors в продакшн/AOT коде.
// Вместо этого — явная сериализация через toJson():
class Config {
  final String host;
  final int port;
  final bool _debug;

  Config(this.host, this.port, this._debug);

  // fix 1+2: явный toJson — только публичные поля, AOT-совместимо
  Map<String, dynamic> toJson() => {
        'host': host,
        'port': port,
        // _debug намеренно НЕ включаем — это приватная деталь реализации
      };
}

// fix 3: создание экземпляров по имени строки невозможно без mirrors.
// AOT-совместимый подход — реестр фабрик:
typedef Factory<T> = T Function(List<dynamic> args);

class PluginRegistry {
  final Map<String, Factory<Object>> _factories = {};

  void register<T extends Object>(String name, Factory<T> factory) {
    _factories[name] = factory;
  }

  Object create(String className, List<dynamic> args) {
    final factory = _factories[className];
    if (factory == null) throw ArgumentError('Unknown plugin: $className');
    return factory(args);
  }
}

abstract class Plugin {
  String get name;
  void initialize();
  Map<String, dynamic> getStatus();
}

class LogPlugin implements Plugin {
  @override
  final String name = 'LogPlugin';

  @override
  void initialize() => print('LogPlugin initialized');

  @override
  Map<String, dynamic> getStatus() => {'active': true, 'level': 'info'};
}

class PluginManager {
  // fix 4: вместо getField используем полиморфизм/интерфейс
  void callMethod(Plugin plugin, String methodName) {
    switch (methodName) {
      case 'initialize':
        plugin.initialize();
      case 'getStatus':
        print('Status: ${plugin.getStatus()}');
      default:
        throw ArgumentError('Unknown method: $methodName');
    }
  }

  // fix 5: не меняем приватные поля других классов через mirrors.
  // Если поле нужно изменить — предоставь публичный метод в классе.
  void resetPlugin(Plugin plugin) {
    // Используем публичный API — если нужно, добавляем метод reset()
    print('Use Plugin.reset() interface method instead of reflection');
  }
}

void main() {
  final cfg = Config('localhost', 8080, false);

  // fix 1+2: только публичные данные
  final json = cfg.toJson();
  print(json); // {host: localhost, port: 8080} — без _debug

  // fix 3: реестр фабрик
  final registry = PluginRegistry();
  registry.register<LogPlugin>('LogPlugin', (_) => LogPlugin());

  final plugin = registry.create('LogPlugin', []) as Plugin;

  // fix 4: через интерфейс
  final manager = PluginManager();
  manager.callMethod(plugin, 'initialize');
  manager.callMethod(plugin, 'getStatus');

  // fix 6: порядок полей определён явно в toJson(), а не через mirrors
  final keys = cfg.toJson().keys.toList();
  print('Fields: $keys'); // ['host', 'port'] — всегда в этом порядке
}
```

**Шесть проблем:** 1) `dart:mirrors` не работает при AOT-компиляции и в Flutter. 2) `toJson` включал приватное поле `_debug` — утечка данных. 3a) `findLibrary(Symbol('dart.core'))` не содержит пользовательские классы. 3b) `as ClassMirror` без проверки — `NoSuchMethodError` или `TypeError` в runtime. 4) `mirror.getField()` работает только для полей/геттеров, а не для методов; для вызова методов нужен `mirror.invoke(symbol, args)`. 5) `setField` на приватном поле `_debug` другой библиотеки — `UnsupportedError` (приватные поля других библиотек защищены даже через mirrors). 6) Порядок `declarations.keys` не гарантирован; использование `.take(2)` возвращает непредсказуемые поля.
