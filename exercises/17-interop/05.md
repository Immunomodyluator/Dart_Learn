```dart
// ЧАСТЬ 1: Mirror API — работает только в dart run (JIT)
// Запуск: dart run inspect_mirror.dart (НЕ dart compile)
import 'dart:mirrors';
import 'dart:math';

void inspectWithMirrors(Object obj) {
  final im = reflect(obj);
  final cm = im.type;

  print('=== ${MirrorSystem.getName(cm.simpleName)} ===');

  // Поля
  cm.declarations.forEach((symbol, decl) {
    if (decl is VariableMirror && !decl.isPrivate && !decl.isStatic) {
      final name = MirrorSystem.getName(symbol);
      final value = im.getField(symbol).reflectee;
      print('  $name: $value');
    }
  });

  // Методы (без операторов и конструкторов)
  final methods = cm.declarations.values
      .whereType<MethodMirror>()
      .where((m) => !m.isPrivate && !m.isOperator && !m.isConstructor && !m.isStatic)
      .map((m) => MirrorSystem.getName(m.simpleName))
      .toList();

  if (methods.isNotEmpty) {
    print('  methods: ${methods.join(', ')}');
  }
}

// ЧАСТЬ 2: AOT-совместимый интерфейс
abstract interface class Inspectable {
  String get className;
  Map<String, Object?> get fields;
  List<String> get methodNames;
}

// Вспомогательный миксин для вывода
mixin InspectableMixin implements Inspectable {
  void inspect() {
    print('=== $className ===');
    fields.forEach((k, v) => print('  $k: $v'));
    if (methodNames.isNotEmpty) {
      print('  methods: ${methodNames.join(', ')}');
    }
  }
}

// ЧАСТЬ 3: Вручную реализованный Inspectable — как это делает build_runner
// dart:mirrors недоступен в AOT (dart compile) и Flutter,
// потому что tree shaking требует знать используемые символы на этапе компиляции,
// а рефлексия делает это невозможным (любой код может быть вызван через mirrors).
// Поэтому в production используют кодогенерацию: freezed, json_serializable и т. д.
class Point with InspectableMixin {
  final double x;
  final double y;
  final String label;

  const Point(this.x, this.y, this.label);

  double distanceTo(Point other) =>
      sqrt(pow(x - other.x, 2) + pow(y - other.y, 2));

  @override
  String toString() => 'Point($x, $y)';

  // --- Сгенерированный / написанный вручную код ---
  @override
  String get className => 'Point';

  @override
  Map<String, Object?> get fields => {'x': x, 'y': y, 'label': label};

  @override
  List<String> get methodNames => ['distanceTo', 'toString'];
}

void main() {
  final p = Point(3.0, 4.0, 'origin');

  print('--- Mirror API (только JIT) ---');
  inspectWithMirrors(p);

  print('');
  print('--- AOT-совместимый интерфейс ---');
  p.inspect();

  // Практический тест
  assert(p.fields['x'] == 3.0);
  assert(p.methodNames.contains('distanceTo'));
  print('Все проверки пройдены!');
}
```

**Ключевой вывод:** `dart:mirrors` работает только в JIT-режиме (`dart run`). При AOT-компиляции (`dart compile exe`, Flutter) дерево вызовов стряхивается (tree shaking) — рефлексия не позволяет определить, какие методы реально нужны, поэтому `dart:mirrors` в этих режимах отключён. Кодогенераторы (`build_runner`) создают тот же `Inspectable`-код статически во время разработки, а не в runtime.
