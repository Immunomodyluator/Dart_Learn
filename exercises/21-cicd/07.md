# Задание 07 (21.4): Автоматическая публикация пакетов — 🖊 Написать код

**Тема:** 21. CI/CD и развёртывание | **Подтема:** 21.4 Автоматическая публикация пакетов

---

Реализуй полный пайплайн автоматической публикации Dart-пакета на pub.dev.

**Требования:**

1. **`tool/bump_version.dart`** — скрипт для обновления версии в `pubspec.yaml`:
   - Аргументы: `major` | `minor` | `patch`
   - Читает текущую версию, инкрементирует нужную часть
   - Записывает обновлённую версию обратно в `pubspec.yaml`
   - Выводит: `Bumped 1.2.3 → 1.2.4`

2. **`tool/update_changelog.dart`** — добавляет запись в `CHANGELOG.md`:
   - Принимает версию как аргумент
   - Вставляет секцию `## X.Y.Z (YYYY-MM-DD)` с placeholder для изменений
   - Не затирает предыдущие записи

3. **`.github/workflows/publish.yml`** — публикация при создании тега `v*`:
   - Проверяет, что версия в `pubspec.yaml` совпадает с тегом
   - Запускает `dart pub publish` через OIDC (без ручных токенов)
   - Создаёт GitHub Release с содержимым `CHANGELOG.md` для этой версии

```dart
// tool/bump_version.dart
import 'dart:io';

void main(List<String> args) {
  if (args.isEmpty || !{'major', 'minor', 'patch'}.contains(args[0])) {
    stderr.writeln('Usage: dart tool/bump_version.dart <major|minor|patch>');
    exit(1);
  }

  final bumpType = args[0];
  final pubspec = File('pubspec.yaml');

  if (!pubspec.existsSync()) {
    stderr.writeln('pubspec.yaml not found');
    exit(1);
  }

  final content = pubspec.readAsStringSync();
  final versionMatch = RegExp(r'^version:\s*(\d+\.\d+\.\d+)', multiLine: true)
      .firstMatch(content);

  if (versionMatch == null) {
    stderr.writeln('Could not find version in pubspec.yaml');
    exit(1);
  }

  final current = versionMatch.group(1)!;
  final parts = current.split('.').map(int.parse).toList();

  switch (bumpType) {
    case 'major':
      parts[0]++;
      parts[1] = 0;
      parts[2] = 0;
    case 'minor':
      parts[1]++;
      parts[2] = 0;
    case 'patch':
      parts[2]++;
  }

  final next = parts.join('.');
  final updated = content.replaceFirst(
    'version: $current',
    'version: $next',
  );

  pubspec.writeAsStringSync(updated);
  print('Bumped $current → $next');
}
```

```dart
// tool/update_changelog.dart
import 'dart:io';

void main(List<String> args) {
  if (args.isEmpty) {
    stderr.writeln('Usage: dart tool/update_changelog.dart <version>');
    exit(1);
  }

  final version = args[0];
  final today = DateTime.now();
  final date =
      '${today.year}-${today.month.toString().padLeft(2, '0')}-${today.day.toString().padLeft(2, '0')}';

  final changelog = File('CHANGELOG.md');
  final existing = changelog.existsSync() ? changelog.readAsStringSync() : '';

  // Проверяем, нет ли уже записи для этой версии
  if (existing.contains('## $version')) {
    stderr.writeln('Version $version already exists in CHANGELOG.md');
    exit(1);
  }

  const newEntry = '''
## {version} ({date})

### Added
-

### Changed
-

### Fixed
-

''';

  final entry = newEntry
      .replaceAll('{version}', version)
      .replaceAll('{date}', date);

  // Вставляем ПЕРЕД существующим содержимым (новые записи вверху)
  final header = existing.startsWith('# Changelog')
      ? existing.substring(0, existing.indexOf('\n') + 1)
      : '# Changelog\n';
  final rest = existing.startsWith('# Changelog')
      ? existing.substring(existing.indexOf('\n') + 1)
      : existing;

  changelog.writeAsStringSync('$header\n$entry$rest');
  print('Added CHANGELOG entry for $version');
}
```

```yaml
# .github/workflows/publish.yml
name: Publish to pub.dev

on:
  push:
    tags:
      - "v[0-9]+.[0-9]+.[0-9]+" # точный шаблон: v1.2.3

jobs:
  validate-and-publish:
    runs-on: ubuntu-latest
    permissions:
      contents: write # для создания GitHub Release
      id-token: write # OIDC — для авторизации на pub.dev без токенов

    steps:
      - uses: actions/checkout@v4

      - uses: dart-lang/setup-dart@v1
        with:
          sdk: stable

      - name: Install dependencies
        run: dart pub get

      - name: Run tests
        run: dart test

      - name: Analyze
        run: dart analyze --fatal-infos

      # Проверяем, что версия в pubspec.yaml совпадает с тегом
      - name: Verify version matches tag
        run: |
          PUBSPEC_VERSION=$(grep '^version:' pubspec.yaml | awk '{print $2}')
          TAG_VERSION=${GITHUB_REF_NAME#v}  # убираем 'v' из 'v1.2.3'
          echo "pubspec version: $PUBSPEC_VERSION"
          echo "tag version:     $TAG_VERSION"
          if [ "$PUBSPEC_VERSION" != "$TAG_VERSION" ]; then
            echo "ERROR: version mismatch!"
            exit 1
          fi

      # Dry-run перед реальной публикацией
      - name: Publish dry-run
        run: dart pub publish --dry-run

      # Публикация через OIDC — не нужны PUB_TOKEN секреты
      - name: Publish to pub.dev
        run: dart pub publish --force
        # --force нужен в неинтерактивном режиме CI
        # OIDC настраивается через dart-lang/setup-dart и разрешение id-token: write

      # Извлекаем секцию changelog для текущей версии
      - name: Extract changelog section
        id: changelog
        run: |
          VERSION=${GITHUB_REF_NAME#v}
          NOTES=$(awk "/^## $VERSION /,/^## [0-9]/{if(/^## [0-9]/ && !/^## $VERSION /) exit; print}" CHANGELOG.md)
          # Используем multiline output через EOF-делиметр
          echo "notes<<EOF" >> "$GITHUB_OUTPUT"
          echo "$NOTES" >> "$GITHUB_OUTPUT"
          echo "EOF" >> "$GITHUB_OUTPUT"

      # Создаём GitHub Release с содержимым changelog
      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2
        with:
          body: ${{ steps.changelog.outputs.notes }}
          generate_release_notes: false
          # tag_name берётся автоматически из GITHUB_REF
```

```bash
# Типичный сценарий выпуска новой версии:
# 1. Обновляем версию
dart run tool/bump_version.dart patch  # 1.2.3 → 1.2.4

# 2. Обновляем CHANGELOG
dart run tool/update_changelog.dart 1.2.4
# Редактируем CHANGELOG.md — заполняем Added/Changed/Fixed

# 3. Коммитируем
git add pubspec.yaml CHANGELOG.md
git commit -m "chore: release 1.2.4"

# 4. Создаём тег — это запускает workflow publish.yml
git tag v1.2.4
git push origin main --tags
```

OIDC-авторизация (`id-token: write`) позволяет pub.dev верифицировать, что публикация пришла именно из GitHub Actions вашего репозитория, без необходимости хранить долгоживущий `PUB_TOKEN` в секретах. Проверка совпадения версии тега и `pubspec.yaml` предотвращает случайную публикацию не той версии.
