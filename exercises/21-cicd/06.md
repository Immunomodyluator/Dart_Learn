В Dockerfile и конфигурациях есть ошибки. Найди их.

```dockerfile
# Dockerfile (намеренно испорченная версия)

FROM dart:stable AS build

WORKDIR /app

# bug 1: копируем ВСЕ файлы сразу — кэш зависимостей инвалидируется при любом изменении кода
COPY . .
COPY pubspec.yaml pubspec.lock ./

RUN dart pub get
RUN dart compile exe bin/server.dart -o bin/server

# bug 2: используем полный образ Dart в runtime вместо минимального
FROM dart:stable AS runtime

WORKDIR /app

# bug 3: копируем исходники вместо скомпилированного бинарника
COPY --from=build /app .

# bug 4: процесс запускается от root (нет useradd / USER инструкции)

# bug 5: нет EXPOSE — порт не задокументирован и не предсказуем
# bug 6: нет HEALTHCHECK

CMD ["dart", "bin/server.dart"]  # bug 7: запускаем через Dart VM, а не скомпилированный бинарник
```

```yaml
# docker-compose.yml (намеренно испорченная версия)
version: "3.9"

services:
  app:
    build: .
    ports:
      - "8080:8080"
    environment:
      - DATABASE_URL=postgres://admin:secret123@db:5432/prod # bug 8: хардкод секрета
    depends_on:
      - redis # bug 9: нет проверки готовности (не ждём healthcheck redis)
    restart: no # bug 10: контейнер не перезапускается при падении

  redis:
    image: redis:latest # bug 11: latest-тег нестабилен, версия может сломать API
    # bug 12: нет персистентных volumes для Redis — данные теряются при рестарте
```

```yaml
# .github/workflows/deploy.yml (намеренно испорченный)
jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      # bug 13: нет шага тестирования — деплоим непроверенный код

        run: docker build -t myapp .

      - name: Push image
        run: |
          echo $DOCKER_PASSWORD | docker login -u $DOCKER_USER --password-stdin

          DOCKER_PASSWORD: ${{ secrets.DOCKER_PASSWORD }}
```

<details>
<summary>Решение</summary>

```dockerfile
# Dockerfile — исправленная версия

FROM dart:stable AS build

WORKDIR /app

# fix 1: сначала только манифесты — кэш зависимостей не инвалидируется
#         при изменении .dart-файлов
COPY pubspec.yaml pubspec.lock ./
RUN dart pub get

# Только потом исходники
COPY . .
RUN dart compile exe bin/server.dart -o bin/server

# fix 2: минимальный runtime-образ (финальный размер ~30 MB vs ~500 MB)
FROM debian:bookworm-slim AS runtime

RUN apt-get update && apt-get install -y --no-install-recommends \
    libssl3 ca-certificates curl \
  && rm -rf /var/lib/apt/lists/*

# fix 4: добавляем non-root пользователя
RUN useradd --create-home appuser
WORKDIR /home/appuser/app

# fix 3: копируем только скомпилированный бинарник
COPY --from=build --chown=appuser:appuser /app/bin/server ./server

USER appuser

# fix 5: документируем порт
EXPOSE 8080

# fix 6: добавляем healthcheck
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
  CMD curl -f http://localhost:8080/health || exit 1

# fix 7: запускаем скомпилированный бинарник, не через Dart VM
CMD ["./server"]
```

```yaml
# docker-compose.yml — исправленная версия
version: "3.9"

services:
  app:
    build: .
    ports:
      - "8080:8080"
    environment:
      # fix 8: секреты через переменные окружения / .env файл
      - DATABASE_URL=${DATABASE_URL}
    env_file:
      - .env # .env в .gitignore — никогда не коммитить
    depends_on:
      redis:
        # fix 9: ждём, пока redis пройдёт healthcheck
        condition: service_healthy
    # fix 10: перезапуск при падении
    restart: unless-stopped

  redis:
    # fix 11: фиксируем мажорную версию — стабильное API
    image: redis:7-alpine
    restart: unless-stopped
    # fix 12: персистентные данные Redis
    volumes:
      - redis_data:/data
    command: redis-server --appendonly yes
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 5

volumes:
  redis_data:
```

```yaml
# .github/workflows/deploy.yml — исправленный
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dart-lang/setup-dart@v1
        with:
          sdk: stable
      - run: dart pub get
      - run: dart analyze --fatal-infos
      - run: dart test

  deploy:
    # fix 13: деплоим только если тесты прошли
    needs: test
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Log in to registry
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          # Встроенный GITHUB_TOKEN — не нужны внешние секреты на логин
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          # fix 14: тегируем по git SHA + latest — история образов сохраняется
          tags: |
            ghcr.io/${{ github.repository }}:${{ github.sha }}
            ghcr.io/${{ github.repository }}:latest

      - name: Deploy via SSH
        # fix 15: используем специализированный action — ключ не попадает в логи
        uses: appleboy/ssh-action@v1
        with:
          host: ${{ secrets.DEPLOY_HOST }}
          username: deploy
          key: ${{ secrets.SSH_KEY }}
          script: |
            docker pull ghcr.io/${{ github.repository }}:${{ github.sha }}
            cd /app
            IMAGE_TAG=${{ github.sha }} docker-compose up -d --no-deps app
```

**15 ошибок:** 1) `COPY . .` перед `COPY pubspec.*` — любое изменение кода инвалидирует кэш зависимостей, каждая сборка скачивает пакеты заново. 2) `FROM dart:stable` в runtime — образ ~500 MB вместо ~30 MB; Dart VM не нужна для нативного бинарника. 3) `COPY --from=build /app .` — копируем весь исходный код в production-образ. 4) Нет non-root пользователя — процесс работает как root внутри контейнера. 5) Нет `EXPOSE` — порт не задокументирован. 6) Нет `HEALTHCHECK`. 7) `CMD ["dart", "bin/server.dart"]` — запуск через VM, не скомпилированный бинарник. 8) `DATABASE_URL` с паролем прямо в Compose — попадает в историю git. 9) `depends_on: [redis]` без `condition` — приложение стартует до готовности Redis. 10) `restart: no` — контейнер не восстанавливается после падения. 11) `redis:latest` — нестабильный тег, может привести к неожиданному обновлению. 12) Нет `volumes` для Redis — данные удаляются при остановке. 13) Деплой без предварительных тестов. 14) `docker push myapp` без тега версии — нет истории образов, нельзя откатиться. 15) SSH-ключ через `${{ secrets.SSH_KEY }}` напрямую в `run` — может попасть в логи; следует использовать специализированный action.

</details>
