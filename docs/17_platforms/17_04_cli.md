# 17.4 CLI-приложения на Dart

## 1. Формальное определение

**CLI-приложение (Command-Line Interface)** — программа, работающая в терминале. Dart отлично подходит для CLI благодаря AOT-компиляции в нативный бинарник (быстрый старт, ~5 мс) и богатой стандартной библиотеке (`dart:io`).

## 2. Зачем это нужно

- **Быстрый старт** — AOT-бинарник запускается мгновенно (в отличие от JIT).
- **Кроссплатформенность** — один код для Windows, macOS, Linux.
- **Распространение** — через `dart pub global activate` или как бинарник.
- **Экосистема** — доступ ко всем Dart-пакетам.

## 3. Создание CLI-приложения

```bash
dart create -t console my_cli
```

```
my_cli/
├── bin/
```

### Базовый пример

```dart
// bin/my_cli.dart
import 'dart:io';

void main(List<String> arguments) {
  if (arguments.isEmpty) {
    print('Usage: my_cli <name>');
    exit(1);
  }

  final name = arguments.first;
  print('Hello, $name!');
}
```

## 4. Парсинг аргументов (package:args)

```yaml
dependencies:
  args: ^2.4.0
```

```dart
import 'dart:io';
import 'package:args/args.dart';
import 'package:args/command_runner.dart';

void main(List<String> arguments) {
  final parser = ArgParser()
    ..addOption('output', abbr: 'o', help: 'Output file path')
    ..addOption('format', abbr: 'f', defaultsTo: 'json',
        allowed: ['json', 'csv', 'xml'])
    ..addFlag('verbose', abbr: 'v', defaultsTo: false)
    ..addFlag('help', abbr: 'h', negatable: false);

  try {
    final results = parser.parse(arguments);

    if (results['help'] as bool) {
      print('Usage: my_cli [options] <input>');
      print(parser.usage);
      exit(0);
    }

    final verbose = results['verbose'] as bool;
    final format = results['format'] as String;
    final output = results['output'] as String?;
    final rest = results.rest; // Позиционные аргументы

    if (verbose) print('Format: $format, Output: $output');
    processFiles(rest, format: format, output: output);
  } on FormatException catch (e) {
    print('Error: ${e.message}');
    print(parser.usage);
    exit(1);
  }
}
```

### CommandRunner — подкоманды

```dart
import 'package:args/command_runner.dart';

void main(List<String> arguments) {
  final runner = CommandRunner<void>('my_cli', 'My CLI tool')
    ..addCommand(InitCommand())
    ..addCommand(BuildCommand())
    ..addCommand(ServeCommand());

  runner.run(arguments);
}

class InitCommand extends Command<void> {
  @override String get name => 'init';
  @override String get description => 'Initialize a new project';

  InitCommand() {
    argParser.addOption('template', abbr: 't', defaultsTo: 'default');
  }

  @override
  void run() {
    final template = argResults!['template'] as String;
    print('Initializing project with template: $template');
  }
}

class BuildCommand extends Command<void> {
  @override String get name => 'build';
  @override String get description => 'Build the project';

  @override
  void run() {
    print('Building...');
  }
}

// Использование:
// my_cli init --template=web
// my_cli build
// my_cli --help
```

## 5. Работа с файлами и I/O

```dart
import 'dart:io';
import 'dart:convert';

// Чтение stdin
void readInput() {
  stdout.write('Enter your name: ');
  final name = stdin.readLineSync(encoding: utf8);
  print('Hello, $name!');
}

// Работа с файлами
Future<void> processFile(String path) async {
  final file = File(path);

  if (!await file.exists()) {
    stderr.writeln('File not found: $path');
    exit(1);
  }

  // Построчное чтение (эффективно для больших файлов)
  final lines = file.openRead()
      .transform(utf8.decoder)
      .transform(const LineSplitter());

  var lineCount = 0;
  await for (final line in lines) {
    lineCount++;
    // Обработка строки
  }

  print('Processed $lineCount lines');
}

// Рекурсивный обход директорий
Future<void> listFiles(String path) async {
  final dir = Directory(path);
  await for (final entity in dir.list(recursive: true)) {
    if (entity is File && entity.path.endsWith('.dart')) {
      print(entity.path);
    }
  }
}
```

## 6. Прогресс и цвета

```dart
// Цветной вывод через ANSI-коды
void printColored() {
  print('\x1B[32m✓ Success\x1B[0m');   // Зелёный
  print('\x1B[31m✗ Error\x1B[0m');     // Красный
  print('\x1B[33m⚠ Warning\x1B[0m');   // Жёлтый
  print('\x1B[1mBold text\x1B[0m');    // Жирный
}

// Простой прогресс-бар
void showProgress(int current, int total) {
  final percent = (current / total * 100).toInt();
  final bar = '█' * (percent ~/ 2) + '░' * (50 - percent ~/ 2);
  stdout.write('\r[$bar] $percent%');
  if (current == total) print('');
}
```

## 7. Компиляция и распространение

```bash
# Компиляция в нативный бинарник
dart compile exe bin/my_cli.dart -o my_cli

# Кросс-компиляция не поддерживается — компилируйте на целевой ОС

# Распространение через pub global
# В pubspec.yaml:
# executables:
#   my_cli: my_cli  ← bin/my_cli.dart

dart pub global activate my_cli         # из pub.dev
dart pub global activate --source path . # локально
```

### pubspec.yaml для CLI

```yaml
name: my_cli
version: 1.0.0
description: A CLI tool for ...

environment:
  sdk: ^3.0.0

dependencies:
  args: ^2.4.0

executables:
  my_cli: my_cli # bin/my_cli.dart
```

## 8. Exit codes

```dart
import 'dart:io';

// Стандартные коды возврата
void main(List<String> args) {
  try {
    final result = process(args);
    exit(0);  // Успех
  } on FormatException {
    exit(64); // Usage error
  } on FileSystemException {
    exit(66); // Input error
  } catch (e) {
    stderr.writeln('Error: $e');
    exit(1);  // Общая ошибка
  }
}
```
